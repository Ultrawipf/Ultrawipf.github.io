81 static void _prep_out_transaction (cdcd_interface_t* p_cdc)
82 {
83  uint8_t const rhport = TUD_OPT_RHPORT;
84  uint16_t available = tu_fifo_remaining(&p_cdc->rx_ff);
85 
86  // Prepare for incoming data but only allow what we can store in the ring buffer.
87  // TODO Actually we can still carry out the transfer, keeping count of received bytes
88  // and slowly move it to the FIFO when read().
89  // This pre-check reduces endpoint claiming
90  TU_VERIFY(available >= sizeof(p_cdc->epout_buf), );
91 
92  // claim endpoint
93  TU_VERIFY(usbd_edpt_claim(rhport, p_cdc->ep_out), );
94 
95  // fifo can be changed before endpoint is claimed
96  available = tu_fifo_remaining(&p_cdc->rx_ff);
97 
98  if ( available >= sizeof(p_cdc->epout_buf) )
99  {
100  usbd_edpt_xfer(rhport, p_cdc->ep_out, p_cdc->epout_buf, sizeof(p_cdc->epout_buf));
101  }else
102  {
103  // Release endpoint since we don't make any transfer
104  usbd_edpt_release(rhport, p_cdc->ep_out);
105  }
106 }

164 uint32_t tud_cdc_n_write(uint8_t itf, void const* buffer, uint32_t bufsize)
165 {
166  cdcd_interface_t* p_cdc = &_cdcd_itf[itf];
167  uint16_t ret = tu_fifo_write_n(&p_cdc->tx_ff, buffer, bufsize);
168 
169  // flush if queue more than packet size
170  if ( tu_fifo_count(&p_cdc->tx_ff) >= BULK_PACKET_SIZE )
171  {
172  tud_cdc_n_write_flush(itf);
173  }
174

178 uint32_t tud_cdc_n_write_flush (uint8_t itf)
179 {
180  cdcd_interface_t* p_cdc = &_cdcd_itf[itf];
181 
182  // Skip if usb is not ready yet
183  TU_VERIFY( tud_ready(), 0 );
184 
185  // No data to send
186  if ( !tu_fifo_count(&p_cdc->tx_ff) ) return 0;
187 
188  uint8_t const rhport = TUD_OPT_RHPORT;
189 
190  // Claim the endpoint
191  TU_VERIFY( usbd_edpt_claim(rhport, p_cdc->ep_in), 0 );
192 
193  // Pull data from FIFO
194  uint16_t const count = tu_fifo_read_n(&p_cdc->tx_ff, p_cdc->epin_buf, sizeof(p_cdc->epin_buf));
195 
196  if ( count )
197  {
198  TU_ASSERT( usbd_edpt_xfer(rhport, p_cdc->ep_in, p_cdc->epin_buf, count), 0 );
199  return count;
200  }else
201  {
202  // Release endpoint since we don't make any transfer
203  // Note: data is dropped if terminal is not connected
204  usbd_edpt_release(rhport, p_cdc->ep_in);
205  return 0;
206  }
207 }

222 void cdcd_init(void)
223 {
224  tu_memclr(_cdcd_itf, sizeof(_cdcd_itf));
225 
226  for(uint8_t i=0; i<CFG_TUD_CDC; i++)
227  {
228  cdcd_interface_t* p_cdc = &_cdcd_itf[i];
229 
230  p_cdc->wanted_char = -1;
231 
232  // default line coding is : stop bit = 1, parity = none, data bits = 8
233  p_cdc->line_coding.bit_rate = 115200;
234  p_cdc->line_coding.stop_bits = 0;
235  p_cdc->line_coding.parity = 0;
236  p_cdc->line_coding.data_bits = 8;
237 
238  // Config RX fifo
239  tu_fifo_config(&p_cdc->rx_ff, p_cdc->rx_ff_buf, TU_ARRAY_SIZE(p_cdc->rx_ff_buf), 1, false);
240 
241  // Config TX fifo as overwritable at initialization and will be changed to non-overwritable
242  // if terminal supports DTR bit. Without DTR we do not know if data is actually polled by terminal.
243  // In this way, the most current data is prioritized.
244  tu_fifo_config(&p_cdc->tx_ff, p_cdc->tx_ff_buf, TU_ARRAY_SIZE(p_cdc->tx_ff_buf), 1, true);
245 
246 #if CFG_FIFO_MUTEX
247  tu_fifo_config_mutex(&p_cdc->rx_ff, osal_mutex_create(&p_cdc->rx_ff_mutex));
248  tu_fifo_config_mutex(&p_cdc->tx_ff, osal_mutex_create(&p_cdc->tx_ff_mutex));
249 #endif
250  }
251 }

253 void cdcd_reset(uint8_t rhport)
254 {
255  (void) rhport;
256 
257  for(uint8_t i=0; i<CFG_TUD_CDC; i++)
258  {
259  cdcd_interface_t* p_cdc = &_cdcd_itf[i];
260 
261  tu_memclr(p_cdc, ITF_MEM_RESET_SIZE);
262  tu_fifo_clear(&p_cdc->rx_ff);

268 uint16_t cdcd_open(uint8_t rhport, tusb_desc_interface_t const * itf_desc, uint16_t max_len)
269 {
270  // Only support ACM subclass
271  TU_VERIFY( TUSB_CLASS_CDC == itf_desc->bInterfaceClass &&
272  CDC_COMM_SUBCLASS_ABSTRACT_CONTROL_MODEL == itf_desc->bInterfaceSubClass, 0);
273 
274  // Note: 0xFF can be used with RNDIS
275  TU_VERIFY(tu_within(CDC_COMM_PROTOCOL_NONE, itf_desc->bInterfaceProtocol, CDC_COMM_PROTOCOL_ATCOMMAND_CDMA), 0);
276 
277  // Find available interface
278  cdcd_interface_t * p_cdc = NULL;
279  for(uint8_t cdc_id=0; cdc_id<CFG_TUD_CDC; cdc_id++)
280  {
281  if ( _cdcd_itf[cdc_id].ep_in == 0 )
282  {
283  p_cdc = &_cdcd_itf[cdc_id];
284  break;
285  }
286  }
287  TU_ASSERT(p_cdc, 0);
288 
289  //------------- Control Interface -------------//
290  p_cdc->itf_num = itf_desc->bInterfaceNumber;
291 
292  uint16_t drv_len = sizeof(tusb_desc_interface_t);
293  uint8_t const * p_desc = tu_desc_next( itf_desc );
294 
295  // Communication Functional Descriptors
296  while ( TUSB_DESC_CS_INTERFACE == tu_desc_type(p_desc) && drv_len <= max_len )
297  {
298  drv_len += tu_desc_len(p_desc);
299  p_desc = tu_desc_next(p_desc);
300  }
301 
302  if ( TUSB_DESC_ENDPOINT == tu_desc_type(p_desc) )
303  {
304  // notification endpoint if any
305  TU_ASSERT( usbd_edpt_open(rhport, (tusb_desc_endpoint_t const *) p_desc), 0 );
306 
307  p_cdc->ep_notif = ((tusb_desc_endpoint_t const *) p_desc)->bEndpointAddress;
308 
309  drv_len += tu_desc_len(p_desc);
310  p_desc = tu_desc_next(p_desc);
311  }
312 
313  //------------- Data Interface (if any) -------------//
314  if ( (TUSB_DESC_INTERFACE == tu_desc_type(p_desc)) &&
315  (TUSB_CLASS_CDC_DATA == ((tusb_desc_interface_t const *) p_desc)->bInterfaceClass) )
316  {
317  // next to endpoint descriptor
318  drv_len += tu_desc_len(p_desc);
319  p_desc = tu_desc_next(p_desc);
320 
321  // Open endpoint pair
322  TU_ASSERT( usbd_open_edpt_pair(rhport, p_desc, 2, TUSB_XFER_BULK, &p_cdc->ep_out, &p_cdc->ep_in), 0 );
323 
324  drv_len += 2*sizeof(tusb_desc_endpoint_t);
325  }
326 
327  // Prepare for incoming data
328  _prep_out_transaction(p_cdc);
329 
330  return drv_len;
331 }

333 // Invoked when a control transfer occurred on an interface of this class
334 // Driver response accordingly to the request and the transfer stage (setup/data/ack)
335 // return false to stall control endpoint (e.g unsupported request)
336 bool cdcd_control_xfer_cb(uint8_t rhport, uint8_t stage, tusb_control_request_t const * request)
337 {
338  // Handle class request only
339  TU_VERIFY(request->bmRequestType_bit.type == TUSB_REQ_TYPE_CLASS);
340 
341  uint8_t itf = 0;
342  cdcd_interface_t* p_cdc = _cdcd_itf;
343 
344  // Identify which interface to use
345  for ( ; ; itf++, p_cdc++)
346  {
347  if (itf >= TU_ARRAY_SIZE(_cdcd_itf)) return false;
348 
349  if ( p_cdc->itf_num == request->wIndex ) break;
350  }
351 
352  switch ( request->bRequest )
353  {
354  case CDC_REQUEST_SET_LINE_CODING:
355  if (stage == CONTROL_STAGE_SETUP)
356  {
357  TU_LOG2(" Set Line Coding\r\n");
358  tud_control_xfer(rhport, request, &p_cdc->line_coding, sizeof(cdc_line_coding_t));
359  }
360  else if ( stage == CONTROL_STAGE_ACK)
361  {
362  if ( tud_cdc_line_coding_cb ) tud_cdc_line_coding_cb(itf, &p_cdc->line_coding);
363  }
364  break;
365 
366  case CDC_REQUEST_GET_LINE_CODING:
367  if (stage == CONTROL_STAGE_SETUP)
368  {
369  TU_LOG2(" Get Line Coding\r\n");
370  tud_control_xfer(rhport, request, &p_cdc->line_coding, sizeof(cdc_line_coding_t));
371  }
372  break;
373 
374  case CDC_REQUEST_SET_CONTROL_LINE_STATE:
375  if (stage == CONTROL_STAGE_SETUP)
376  {
377  tud_control_status(rhport, request);
378  }
379  else if (stage == CONTROL_STAGE_ACK)
380  {
381  // CDC PSTN v1.2 section 6.3.12
382  // Bit 0: Indicates if DTE is present or not.
383  // This signal corresponds to V.24 signal 108/2 and RS-232 signal DTR (Data Terminal Ready)
384  // Bit 1: Carrier control for half-duplex modems.
385  // This signal corresponds to V.24 signal 105 and RS-232 signal RTS (Request to Send)
386  bool const dtr = tu_bit_test(request->wValue, 0);
387  bool const rts = tu_bit_test(request->wValue, 1);
388 
389  p_cdc->line_state = (uint8_t) request->wValue;
390 
391  // Disable fifo overwriting if DTR bit is set
392  tu_fifo_set_overwritable(&p_cdc->tx_ff, !dtr);
393 
394  TU_LOG2(" Set Control Line State: DTR = %d, RTS = %d\r\n", dtr, rts);
395 
396  // Invoke callback
397  if ( tud_cdc_line_state_cb ) tud_cdc_line_state_cb(itf, dtr, rts);
398  }
399  break;
400  case CDC_REQUEST_SEND_BREAK:
401  if (stage == CONTROL_STAGE_SETUP)
402  {
403  tud_control_status(rhport, request);
404  }
405  else if (stage == CONTROL_STAGE_ACK)
406  {
407  TU_LOG2(" Send Break\r\n");
408  if ( tud_cdc_send_break_cb ) tud_cdc_send_break_cb(itf, request->wValue);
409  }
410  break;
411 
412  default: return false; // stall unsupported request
413  }
414 
415  return true;
416 }

418 bool cdcd_xfer_cb(uint8_t rhport, uint8_t ep_addr, xfer_result_t result, uint32_t xferred_bytes)
419 {
420  (void) result;
421 
422  uint8_t itf;
423  cdcd_interface_t* p_cdc;
424 
425  // Identify which interface to use
426  for (itf = 0; itf < CFG_TUD_CDC; itf++)
427  {
428  p_cdc = &_cdcd_itf[itf];
429  if ( ( ep_addr == p_cdc->ep_out ) || ( ep_addr == p_cdc->ep_in ) ) break;
430  }
431  TU_ASSERT(itf < CFG_TUD_CDC);
432 
433  // Received new data
434  if ( ep_addr == p_cdc->ep_out )
435  {
436  tu_fifo_write_n(&p_cdc->rx_ff, &p_cdc->epout_buf, xferred_bytes);
437 
438  // Check for wanted char and invoke callback if needed
439  if ( tud_cdc_rx_wanted_cb && (((signed char) p_cdc->wanted_char) != -1) )
440  {
441  for ( uint32_t i = 0; i < xferred_bytes; i++ )
442  {
443  if ( (p_cdc->wanted_char == p_cdc->epout_buf[i]) && !tu_fifo_empty(&p_cdc->rx_ff) )
444  {
445  tud_cdc_rx_wanted_cb(itf, p_cdc->wanted_char);
446  }
447  }
448  }
449 
450  // invoke receive callback (if there is still data)
451  if (tud_cdc_rx_cb && !tu_fifo_empty(&p_cdc->rx_ff) ) tud_cdc_rx_cb(itf);
452 
453  // prepare for OUT transaction
454  _prep_out_transaction(p_cdc);
455  }
456 
457  // Data sent to host, we continue to fetch from tx fifo to send.
458  // Note: This will cause incorrect baudrate set in line coding.
459  // Though maybe the baudrate is not really important !!!
460  if ( ep_addr == p_cdc->ep_in )
461  {
462  // invoke transmit callback to possibly refill tx fifo
463  if ( tud_cdc_tx_complete_cb ) tud_cdc_tx_complete_cb(itf);
464 
465  if ( 0 == tud_cdc_n_write_flush(itf) )
466  {
467  // If there is no data left, a ZLP should be sent if
468  // xferred_bytes is multiple of EP Packet size and not zero
469  if ( !tu_fifo_count(&p_cdc->tx_ff) && xferred_bytes && (0 == (xferred_bytes & (BULK_PACKET_SIZE-1))) )
470  {
471  if ( usbd_edpt_claim(rhport, p_cdc->ep_in) )
472  {
473  usbd_edpt_xfer(rhport, p_cdc->ep_in, NULL, 0);
474  }
475  }
476  }
477  }
478 
479  // nothing to do with notif endpoint for now
480 
481  return true;
482 }
