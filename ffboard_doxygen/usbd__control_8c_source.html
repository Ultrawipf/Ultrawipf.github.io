65 // Queue ZLP status transaction
66 static inline bool _status_stage_xact(uint8_t rhport, tusb_control_request_t const * request)

84 // Queue a transaction in Data Stage
85 // Each transaction has up to Endpoint0's max packet size.
86 // This function can also transfer an zero-length packet
87 static bool _data_stage_xact(uint8_t rhport)
88 {
89  uint16_t const xact_len = tu_min16(_ctrl_xfer.data_len - _ctrl_xfer.total_xferred, CFG_TUD_ENDPOINT0_SIZE);
90 
91  uint8_t ep_addr = EDPT_CTRL_OUT;
92 
93  if ( _ctrl_xfer.request.bmRequestType_bit.direction == TUSB_DIR_IN )
94  {
95  ep_addr = EDPT_CTRL_IN;
96  if ( xact_len ) memcpy(_usbd_ctrl_buf, _ctrl_xfer.buffer, xact_len);
97  }
98 
99  return usbd_edpt_xfer(rhport, ep_addr, xact_len ? _usbd_ctrl_buf : NULL, xact_len);
100 }

102 // Transmit data to/from the control endpoint.
103 // If the request's wLength is zero, a status packet is sent instead.
104 bool tud_control_xfer(uint8_t rhport, tusb_control_request_t const * request, void* buffer, uint16_t len)
105 {
106  _ctrl_xfer.request = (*request);
107  _ctrl_xfer.buffer = (uint8_t*) buffer;
108  _ctrl_xfer.total_xferred = 0U;
109  _ctrl_xfer.data_len = tu_min16(len, request->wLength);
110 
111  if (request->wLength > 0U)
112  {
113  if(_ctrl_xfer.data_len > 0U)
114  {
115  TU_ASSERT(buffer);
116  }
117 
118 // TU_LOG2(" Control total data length is %u bytes\r\n", _ctrl_xfer.data_len);
119 
120  // Data stage
121  TU_ASSERT( _data_stage_xact(rhport) );
122  }
123  else
124  {
125  // Status stage
126  TU_ASSERT( _status_stage_xact(rhport, request) );
127  }
128 
129  return true;
130 }

161 // callback when a transaction complete on
162 // - DATA stage of control endpoint or
163 // - Status stage
164 bool usbd_control_xfer_cb (uint8_t rhport, uint8_t ep_addr, xfer_result_t result, uint32_t xferred_bytes)
165 {
166  (void) result;
167 
168  // Endpoint Address is opposite to direction bit, this is Status Stage complete event
169  if ( tu_edpt_dir(ep_addr) != _ctrl_xfer.request.bmRequestType_bit.direction )
170  {
171  TU_ASSERT(0 == xferred_bytes);
172 
173  // invoke optional dcd hook if available
174  if (dcd_edpt0_status_complete) dcd_edpt0_status_complete(rhport, &_ctrl_xfer.request);
175 
176  if (_ctrl_xfer.complete_cb)
177  {
178  // TODO refactor with usbd_driver_print_control_complete_name
179  _ctrl_xfer.complete_cb(rhport, CONTROL_STAGE_ACK, &_ctrl_xfer.request);
180  }
181 
182  return true;
183  }
184 
185  if ( _ctrl_xfer.request.bmRequestType_bit.direction == TUSB_DIR_OUT )
186  {
187  TU_VERIFY(_ctrl_xfer.buffer);
188  memcpy(_ctrl_xfer.buffer, _usbd_ctrl_buf, xferred_bytes);
189  }
190 
191  _ctrl_xfer.total_xferred += xferred_bytes;
192  _ctrl_xfer.buffer += xferred_bytes;
193 
194  // Data Stage is complete when all request's length are transferred or
195  // a short packet is sent including zero-length packet.
196  if ( (_ctrl_xfer.request.wLength == _ctrl_xfer.total_xferred) || (xferred_bytes < CFG_TUD_ENDPOINT0_SIZE) )
197  {
198  // DATA stage is complete
199  bool is_ok = true;
200 
201  // invoke complete callback if set
202  // callback can still stall control in status phase e.g out data does not make sense
203  if ( _ctrl_xfer.complete_cb )
204  {
205  #if CFG_TUSB_DEBUG >= 2
206  usbd_driver_print_control_complete_name(_ctrl_xfer.complete_cb);
207  #endif
208 
209  is_ok = _ctrl_xfer.complete_cb(rhport, CONTROL_STAGE_DATA, &_ctrl_xfer.request);
210  }
211 
212  if ( is_ok )
213  {
214  // Send status
215  TU_ASSERT( _status_stage_xact(rhport, &_ctrl_xfer.request) );
216  }else
217  {
218  // Stall both IN and OUT control endpoint
219  dcd_edpt_stall(rhport, EDPT_CTRL_OUT);
220  dcd_edpt_stall(rhport, EDPT_CTRL_IN);
221  }
222  }
223  else
224  {
225  // More data to transfer
226  TU_ASSERT( _data_stage_xact(rhport) );
227  }
228 
229  return true;
230 }
