181  inline CommandStatus handleGetSet(const ParsedCommand& cmd,std::vector<CommandReply>& replies,TVal& value){
182  if(cmd.type == CMDtype::set){
183  value = static_cast<TVal>(cmd.val);
184  }else if(cmd.type == CMDtype::get){
185  replies.push_back(CommandReply(value));
186  }else{
187  return CommandStatus::ERR;
188  }
189  return CommandStatus::OK;
