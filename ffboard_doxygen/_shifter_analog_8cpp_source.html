39 void ShifterAnalog::calculateGear() {
40  gear = 0;
41 
42  switch (mode)
43  {
44  case ShifterMode::G29_seq:
45  case ShifterMode::G27_seq:
46  // Sequential mode
47  if(x_val < X_12){
48  gear = 1;
49  }else if(x_val > X_56){
50  gear = 2;
51  }
52  break;
53  case ShifterMode::G27_H:
54  case ShifterMode::G29_H:
55  // Calculate h shifter gears by thresholds
56  if(x_val < X_12){
57  if(y_val > Y_135)
58  gear=1; // 1st gear
59 
60  if(y_val < Y_246)
61  gear=2; // 2nd gear
62 
63  }else if(x_val > X_56){
64  if(y_val > Y_135)
65  gear=5; // 5th gear
66 
67  if(y_val < Y_246)
68  gear=6; // 6th gear
69 
70  }else{
71  if(y_val > Y_135)
72  gear=3; // 3rd gear
73 
74  if(y_val < Y_246)
75  gear=4; // 4th gear
76  }
77 
78  if(gear == 6 && reverseButtonState){
79  gear = 7; // Reverse
80  }
81  }
82 }

101 uint8_t ShifterAnalog::readButtons(uint64_t* buf){
102  updateAdc();
103 
104  updateReverseState();
105  calculateGear();
106 
107  *buf = 0;
108  // User buttons go first so that switching between sequential and H-pattern
109  // doesn't affect user button assignments.
110  auto numUserButtons{getUserButtons(buf)};
111 
112  if(gear > 0){
113  *buf |= 1 << (gear - 1 + numUserButtons);
114  }
115 
116  return this->btnnum;
117 }

196 ParseStatus ShifterAnalog::command(ParsedCommand* cmd,std::string* reply){
197  ParseStatus result = ParseStatus::OK;
198  // use "shifter_mode!" to print a list of possible modes and choose one
199  if(cmd->cmd == "shifter_mode"){
200  if(cmd->type == CMDtype::set){
201  setMode((ShifterMode)cmd->val);
202  }else if(cmd->type == CMDtype::get){
203  *reply += std::to_string((uint8_t)this->mode);
204  }else{
205  printModes(reply);
206  }
207  }else if (cmd->cmd == "shifter_x_12"){
208  handleGetSet(cmd, reply, X_12);
209  }else if (cmd->cmd == "shifter_x_56"){
210  handleGetSet(cmd, reply, X_56);
211  }else if(cmd->cmd == "shifter_y_135"){
212  handleGetSet(cmd, reply, Y_135);
213  }else if(cmd->cmd == "shifter_y_246"){
214  handleGetSet(cmd, reply, Y_246);
215  }else if (cmd->cmd == "shifter_rev_btn"){
216  handleGetSet(cmd, reply, reverseButtonNum);
217  }else if (cmd->cmd == "shifter_cs_pin"){
218  handleGetSet(cmd, reply, cs_pin_num);
219  }else if (cmd->cmd == "shifter_x_chan"){
220  handleGetSet(cmd, reply, x_chan);
221  }else if (cmd->cmd == "shifter_y_chan"){
222  handleGetSet(cmd, reply, y_chan);
223  }else if (cmd->cmd == "shifter_vals" && cmd->type == CMDtype::get){
224  *reply += std::to_string(x_val) + "," + std::to_string(y_val);
225  }else if (cmd->cmd == "shifter_gear" && cmd->type == CMDtype::get){
226  *reply += std::to_string(gear);
227  }else{
228  result = ParseStatus::NOT_FOUND;
229  }
230  return result;
231 }
