55 void ShifterAnalog::calculateGear() {
56  gear = 0;
57 
58  switch (mode)
59  {
60  case ShifterMode::G29_seq:
61  case ShifterMode::G27_seq:
62  // Sequential mode
63  if(x_val < X_12){
64  gear = 1;
65  }else if(x_val > X_56){
66  gear = 2;
67  }
68  break;
69  case ShifterMode::G27_H:
70  case ShifterMode::G29_H:
71  // Calculate h shifter gears by thresholds
72  if(x_val < X_12){
73  if(y_val > Y_135)
74  gear=1; // 1st gear
75 
76  if(y_val < Y_246)
77  gear=2; // 2nd gear
78 
79  }else if(x_val > X_56){
80  if(y_val > Y_135)
81  gear=5; // 5th gear
82 
83  if(y_val < Y_246)
84  gear=6; // 6th gear
85 
86  }else{
87  if(y_val > Y_135)
88  gear=3; // 3rd gear
89 
90  if(y_val < Y_246)
91  gear=4; // 4th gear
92  }
93 
94  if(gear == 6 && reverseButtonState){
95  gear = 7; // Reverse
96  }
97  }
98 }

117 uint8_t ShifterAnalog::readButtons(uint64_t* buf){
118  updateAdc();
119 
120  updateReverseState();
121  calculateGear();
122 
123  *buf = 0;
124  // User buttons go first so that switching between sequential and H-pattern
125  // doesn't affect user button assignments.
126  auto numUserButtons{getUserButtons(buf)};
127 
128  if(gear > 0){
129  *buf |= 1 << (gear - 1 + numUserButtons);
130  }
131 
132  return this->btnnum;
133 }

214 CommandStatus ShifterAnalog::command(const ParsedCommand& cmd,std::vector<CommandReply>& replies){
215 
216  switch(static_cast<ShifterAnalog_commands>(cmd.cmdId)){
217  case ShifterAnalog_commands::mode:
218  if(cmd.type == CMDtype::set){
219  setMode((ShifterMode)cmd.val);
220  }else if(cmd.type == CMDtype::get){
221  replies.push_back(CommandReply((uint8_t)this->mode));
222  }else if(cmd.type == CMDtype::info){
223  replies.push_back(CommandReply(printModes()));
224  }
225  break;
226 
227  case ShifterAnalog_commands::x12:
228  return handleGetSet(cmd, replies, X_12);
229  case ShifterAnalog_commands::x56:
230  return handleGetSet(cmd, replies, X_56);
231  case ShifterAnalog_commands::y135:
232  return handleGetSet(cmd, replies, Y_135);
233  case ShifterAnalog_commands::y246:
234  return handleGetSet(cmd, replies, Y_246);
235  case ShifterAnalog_commands::revbtn:
236  return handleGetSet(cmd, replies, reverseButtonNum);
237  case ShifterAnalog_commands::cspin:
238  return handleGetSet(cmd, replies, cs_pin_num);
239  case ShifterAnalog_commands::xchan:
240  return handleGetSet(cmd, replies, x_chan);
241  case ShifterAnalog_commands::ychan:
242  return handleGetSet(cmd, replies, y_chan);
243  case ShifterAnalog_commands::vals:
244  if(cmd.type == CMDtype::get){
245  replies.push_back(CommandReply(x_val,y_val));
246  }else{
247  return CommandStatus::ERR;
248  }
249  break;
250  case ShifterAnalog_commands::gear:
251  if(cmd.type == CMDtype::get){
252  replies.push_back(CommandReply(gear));
253  }else{
254  return CommandStatus::ERR;
255  }
256  break;
257 
258  default:
259  return CommandStatus::NOT_FOUND;
260  }
261 
262  return CommandStatus::OK;
263 }
