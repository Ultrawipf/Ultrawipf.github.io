164 bool hidh_open_subtask(uint8_t rhport, uint8_t dev_addr, tusb_desc_interface_t const *p_interface_desc, uint16_t *p_length)
165 {
166  uint8_t const *p_desc = (uint8_t const *) p_interface_desc;
167 
168  //------------- HID descriptor -------------//
169  p_desc += p_desc[DESC_OFFSET_LEN];
170  tusb_hid_descriptor_hid_t const *p_desc_hid = (tusb_hid_descriptor_hid_t const *) p_desc;
171  TU_ASSERT(HID_DESC_TYPE_HID == p_desc_hid->bDescriptorType, TUSB_ERROR_INVALID_PARA);
172 
173  //------------- Endpoint Descriptor -------------//
174  p_desc += p_desc[DESC_OFFSET_LEN];
175  tusb_desc_endpoint_t const * p_endpoint_desc = (tusb_desc_endpoint_t const *) p_desc;
176  TU_ASSERT(TUSB_DESC_ENDPOINT == p_endpoint_desc->bDescriptorType, TUSB_ERROR_INVALID_PARA);
177 
178  if ( HID_SUBCLASS_BOOT == p_interface_desc->bInterfaceSubClass )
179  {
180  #if CFG_TUH_HID_KEYBOARD
181  if ( HID_PROTOCOL_KEYBOARD == p_interface_desc->bInterfaceProtocol)
182  {
183  TU_ASSERT( hidh_interface_open(rhport, dev_addr, p_interface_desc->bInterfaceNumber, p_endpoint_desc, &keyboardh_data[dev_addr-1]) );
184  TU_LOG2_HEX(keyboardh_data[dev_addr-1].ep_in);
185  } else
186  #endif
187 
188  #if CFG_TUH_HID_MOUSE
189  if ( HID_PROTOCOL_MOUSE == p_interface_desc->bInterfaceProtocol)
190  {
191  TU_ASSERT ( hidh_interface_open(rhport, dev_addr, p_interface_desc->bInterfaceNumber, p_endpoint_desc, &mouseh_data[dev_addr-1]) );
192  TU_LOG2_HEX(mouseh_data[dev_addr-1].ep_in);
193  } else
194  #endif
195 
196  {
197  // Not supported protocol
198  return false;
199  }
200  }else
201  {
202  // Not supported subclass
203  return false;
204  }
205 
206  *p_length = sizeof(tusb_desc_interface_t) + sizeof(tusb_hid_descriptor_hid_t) + sizeof(tusb_desc_endpoint_t);
207 
208  return true;
209 }
