33 void MotorPWM::turn(int16_t power){
34  if(!active)
35  return;
36 
37  /*
38  * Generates a classic RC 20ms 1000-2000µs signal
39  * Centered at 1500µs for bidirectional RC ESCs and similiar stuff
40  */
41  if(mode == ModePWM_DRV::RC_PWM){
42  int32_t pval = power;
43 
44  float val = ((pval * 1000)/0x7fff)*tFreq;
45  val = clip((1500*tFreq)-val,1000*tFreq, 2000*tFreq);
46  setPWM(val,ccr_1);
47 
48  /*
49  * Generates a 0-100% PWM signal
50  * and outputs complementary direction signals.
51  * Can be used with cheap halfbridge modules and DC motors
52  */
53  }else if(mode == ModePWM_DRV::PWM_DIR){
54  if(power < 0){
55  setPWM(0,ccr_3);
56  setPWM(0xffff,ccr_4);
57  }else{
58  setPWM(0,ccr_4);
59  setPWM(0xffff,ccr_3);
60  }
61  int32_t val = (uint32_t)((abs(power) * period)/0x7fff);
62  setPWM(val,ccr_1);
63 
64  }else if(mode == ModePWM_DRV::CENTERED_PWM){
65  int32_t pval = 0x7fff+power;
66  int32_t val = (pval * period)/0xffff;
67  setPWM(val,ccr_1);
68 
69  }else if(mode == ModePWM_DRV::PWM_DUAL){
70  int32_t val = (uint32_t)((abs(power) * period)/0x7fff);
71  if(power < 0){
72  setPWM(0,ccr_1);
73  setPWM(val,ccr_2);
74  }else{
75  setPWM(0,ccr_2);
76  setPWM(val,ccr_1);
77  }
78  }
79 }

84 void MotorPWM::setPwmSpeed(SpeedPWM_DRV spd){
85  bool ok = true;
86  switch(spd){
87 
88  case SpeedPWM_DRV::LOW:
89  if(mode == ModePWM_DRV::RC_PWM){
90  period = 40000; //20ms (40000/Sysclock)
91  prescaler = TIM_PWM_FREQ/2000000;
92  }else{
93  period = TIM_PWM_FREQ/3000; // Check if timer can count high enough for very high clock speeds!
94  prescaler = 0;
95  }
96 
97  break;
98  case SpeedPWM_DRV::MID:
99  if(mode == ModePWM_DRV::RC_PWM){
100  period = 30000;//15ms(30000/47)
101  prescaler = TIM_PWM_FREQ/2000000;
102  }else{
103  period = TIM_PWM_FREQ/9000;
104  prescaler = 0;
105  }
106 
107  break;
108  case SpeedPWM_DRV::HIGH:
109  if(mode == ModePWM_DRV::RC_PWM){
110  period = 20000; //10ms (20000/47)
111  prescaler = TIM_PWM_FREQ/2000000;
112  }else{
113  period = TIM_PWM_FREQ/17000;
114  prescaler = 0;
115  }
116  break;
117  case SpeedPWM_DRV::VERYHIGH:
118  if(mode == ModePWM_DRV::RC_PWM){
119  period = 10000; //5ms (20000/23)
120  prescaler = TIM_PWM_FREQ/2000000;
121  }else{
122  period = TIM_PWM_FREQ/24000;
123  prescaler = 0;
124  }
125  break;
126  default:
127  ok = false;
128  }
129 
130  if(ok){
131  this->pwmspeed = spd;
132  tFreq = (float)(TIM_PWM_FREQ/1000000)/(float)(prescaler+1);
133 
134  pwmInitTimer(timer, channel_1,period,prescaler);
135  pwmInitTimer(timer, channel_2,period,prescaler);
136  pwmInitTimer(timer, channel_3,period,prescaler);
137  pwmInitTimer(timer, channel_4,period,prescaler);
138  HAL_TIM_MspPostInit(timer);
139 // setPWM_HAL(0, timer, channel_1, period);
140 // pwmInitTimer(timer, channel_2,period,prescaler);
141 // setPWM_HAL(0, timer, channel_2, period);
142  turn(0);
143  }

174 void MotorPWM::saveFlash(){
175  // 0-3: mode
176  // 4-6: speed
177  uint16_t var = (uint8_t)this->mode & 0xf;
178  var |= ((uint8_t)this->pwmspeed & 0x7) << 4;
179  Flash_Write(ADR_PWM_MODE, var);
180 }
181 void MotorPWM::restoreFlash(){
182  uint16_t var = 0;

184  uint8_t m = var & 0xf;
185  this->setMode(ModePWM_DRV(m));
186 
187  uint8_t s = (var >> 4) & 0x7;
188  this->setPwmSpeed(SpeedPWM_DRV(s));
189  }
190 }
