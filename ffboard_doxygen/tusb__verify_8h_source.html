35 /* This file use an advanced macro technique to mimic the default parameter
36  * as C++ for the sake of code simplicity. Beware of a headache macro
37  * manipulation that you are told to stay away.
38  *
39  *
40  * This contains macros for both VERIFY and ASSERT:
41  *
42  * VERIFY: Used when there is an error condition which is not the
43  * fault of the MCU. For example, bounds checking on data
44  * sent to the micro over USB should use this function.
45  * Another example is checking for buffer overflows, where
46  * returning from the active function causes a NAK.
47  *
48  * ASSERT: Used for error conditions that are caused by MCU firmware
49  * bugs. This is used to discover bugs in the code more
50  * quickly. One example would be adding assertions in library
51  * function calls to confirm a function's (untainted)
52  * parameters are valid.
53  *
54  *
55  * The difference in behaviour is that ASSERT triggers a breakpoint while
56  * verify does not.
57  *
58  * #define TU_VERIFY(cond) if(cond) return false;
59  * #define TU_VERIFY(cond,ret) if(cond) return ret;
60  *
61  * #define TU_VERIFY_HDLR(cond,handler) if(cond) {handler; return false;}
62  * #define TU_VERIFY_HDLR(cond,ret,handler) if(cond) {handler; return ret;}
63  *
64  * #define TU_ASSERT(cond) if(cond) {_MESS_FAILED(); TU_BREAKPOINT(), return false;}
65  * #define TU_ASSERT(cond,ret) if(cond) {_MESS_FAILED(); TU_BREAKPOINT(), return ret;}
66  *
67  *------------------------------------------------------------------*/

77 #if CFG_TUSB_DEBUG
78  #include <stdio.h>
79  #define _MESS_ERR(_err) tu_printf("%s %d: failed, error = %s\r\n", __func__, __LINE__, tusb_strerr[_err])
80  #define _MESS_FAILED() tu_printf("%s %d: assert failed\r\n", __func__, __LINE__)
81 #else
82  #define _MESS_ERR(_err) do {} while (0)
83  #define _MESS_FAILED() do {} while (0)
84 #endif

86 // Halt CPU (breakpoint) when hitting error, only apply for Cortex M3, M4, M7
87 #if defined(__ARM_ARCH_7M__) || defined (__ARM_ARCH_7EM__)
88  #define TU_BREAKPOINT() do \
89  { \
90  volatile uint32_t* ARM_CM_DHCSR = ((volatile uint32_t*) 0xE000EDF0UL); /* Cortex M CoreDebug->DHCSR */ \
91  if ( (*ARM_CM_DHCSR) & 1UL ) __asm("BKPT #0\n"); /* Only halt mcu if debugger is attached */ \
92  } while(0)
93 
94 #elif defined(__riscv)
95  #define TU_BREAKPOINT() do { __asm("ebreak\n"); } while(0)
96 
97 #else
98  #define TU_BREAKPOINT() do {} while (0)
99 #endif

105 // Helper to implement optional parameter for TU_VERIFY Macro family
106 #define GET_3RD_ARG(arg1, arg2, arg3, ...) arg3
107 #define GET_4TH_ARG(arg1, arg2, arg3, arg4, ...) arg4
108 
109 /*------------- Generator for TU_VERIFY and TU_VERIFY_HDLR -------------*/
110 #define TU_VERIFY_DEFINE(_cond, _handler, _ret) do \
111 { \
112  if ( !(_cond) ) { _handler; return _ret; } \
113 } while(0)
114 
115 /*------------------------------------------------------------------*/
116 /* TU_VERIFY
117  * - TU_VERIFY_1ARGS : return false if failed
118  * - TU_VERIFY_2ARGS : return provided value if failed
119  *------------------------------------------------------------------*/
120 #define TU_VERIFY_1ARGS(_cond) TU_VERIFY_DEFINE(_cond, , false)
121 #define TU_VERIFY_2ARGS(_cond, _ret) TU_VERIFY_DEFINE(_cond, , _ret)
122 
123 #define TU_VERIFY(...) GET_3RD_ARG(__VA_ARGS__, TU_VERIFY_2ARGS, TU_VERIFY_1ARGS, UNUSED)(__VA_ARGS__)
124 
125 
126 /*------------------------------------------------------------------*/
127 /* TU_VERIFY WITH HANDLER
128  * - TU_VERIFY_HDLR_2ARGS : execute handler, return false if failed
129  * - TU_VERIFY_HDLR_3ARGS : execute handler, return provided error if failed
130  *------------------------------------------------------------------*/
131 #define TU_VERIFY_HDLR_2ARGS(_cond, _handler) TU_VERIFY_DEFINE(_cond, _handler, false)
132 #define TU_VERIFY_HDLR_3ARGS(_cond, _handler, _ret) TU_VERIFY_DEFINE(_cond, _handler, _ret)
133 
134 #define TU_VERIFY_HDLR(...) GET_4TH_ARG(__VA_ARGS__, TU_VERIFY_HDLR_3ARGS, TU_VERIFY_HDLR_2ARGS,UNUSED)(__VA_ARGS__)
135 
136 /*------------------------------------------------------------------*/
137 /* ASSERT
138  * basically TU_VERIFY with TU_BREAKPOINT() as handler
139  * - 1 arg : return false if failed
140  * - 2 arg : return error if failed
141  *------------------------------------------------------------------*/
142 #define ASSERT_1ARGS(_cond) TU_VERIFY_DEFINE(_cond, _MESS_FAILED(); TU_BREAKPOINT(), false)
143 #define ASSERT_2ARGS(_cond, _ret) TU_VERIFY_DEFINE(_cond, _MESS_FAILED(); TU_BREAKPOINT(), _ret)
144 
145 #ifndef TU_ASSERT
146 #define TU_ASSERT(...) GET_3RD_ARG(__VA_ARGS__, ASSERT_2ARGS, ASSERT_1ARGS,UNUSED)(__VA_ARGS__)
147 #endif
