187  * Calculate soft endstop effect
188  */
189 int16_t NormalizedAxis::updateEndstop(){
190  int8_t clipdir = cliptest<int32_t,int32_t>(metric.current.pos, -0x7fff, 0x7fff);
191  if(clipdir == 0){
192  return 0;
193  }
194  int32_t addtorque = clip<int32_t,int32_t>(abs(metric.current.pos)-0x7fff,-0x7fff,0x7fff);
195  addtorque *= (float)endstop_gain * 0.15f; // Apply endstop gain for stiffness
196  addtorque *= -clipdir;
197 
198  return clip<int32_t,int32_t>(addtorque,-0x7fff,0x7fff);
199 }

205 // pass in ptr to receive the sum of the effects + endstop torque
206 // return true if torque is clipping
207 bool NormalizedAxis::updateTorque(int32_t* totalTorque) {
208 
209  if(abs(effectTorque) >= 0x7fff){
210  pulseClipLed();
211  }
212 
213  // Scale effect torque
214  effectTorque *= torqueScaler;
215 
216  int32_t torque = effectTorque + updateEndstop();
217  torque += axisEffectTorque * torqueScaler; // Updated from effect calculator
218 
219  torque = (invertAxis) ? -torque : torque;
220  metric.current.torque = torque;
221  torque = clip<int32_t, int32_t>(torque, -power, power);
222 
223  bool torqueChanged = torque != metric.previous.torque;
224 
225  if (abs(torque) == power){
226  pulseClipLed();
227  }
228 
229  *totalTorque = torque;
230  return (torqueChanged);
231 }
