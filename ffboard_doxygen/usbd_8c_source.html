382  _usbd_q = osal_queue_create(&_usbd_qdef);
383  TU_ASSERT(_usbd_q);
384 
385  // Get application driver if available
386  if ( usbd_app_driver_get_cb )
387  {
388  _app_driver = usbd_app_driver_get_cb(&_app_driver_count);
389  }
390 
391  // Init class drivers
392  for (uint8_t i = 0; i < TOTAL_DRIVER_COUNT; i++)
393  {
394  usbd_class_driver_t const * driver = get_driver(i);
395  TU_LOG2("%s init\r\n", driver->name);
396  driver->init();
397  }
398 
399  // Init device controller driver
400  dcd_init(TUD_OPT_RHPORT);
401  dcd_int_enable(TUD_OPT_RHPORT);
402 
403  return true;

429 /* USB Device Driver task
430  * This top level thread manages all device controller event and delegates events to class-specific drivers.
431  * This should be called periodically within the mainloop or rtos thread.
432  *
433  @code
434  int main(void)
435  {
436  application_init();
437  tusb_init();
438 
439  while(1) // the mainloop
440  {
441  application_code();
442  tud_task(); // tinyusb device task
443  }
444  }
445  @endcode
446  */
447 void tud_task (void)
448 {
449  // Skip if stack is not initialized
450  if ( !tusb_inited() ) return;
451 
452  // Loop until there is no more events in the queue
453  while (1)
454  {
455  dcd_event_t event;
456 
457  if ( !osal_queue_receive(_usbd_q, &event) ) return;
458 
459 #if CFG_TUSB_DEBUG >= 2
460  if (event.event_id == DCD_EVENT_SETUP_RECEIVED) TU_LOG2("\r\n"); // extra line for setup
461  TU_LOG2("USBD %s ", event.event_id < DCD_EVENT_COUNT ? _usbd_event_str[event.event_id] : "CORRUPTED");
462 #endif
463 
464  switch ( event.event_id )
465  {
466  case DCD_EVENT_BUS_RESET:
467  TU_LOG2("\r\n");
468  usbd_reset(event.rhport);
469  _usbd_dev.speed = event.bus_reset.speed;
470  break;
471 
472  case DCD_EVENT_UNPLUGGED:
473  TU_LOG2("\r\n");
474  usbd_reset(event.rhport);
475 
476  // invoke callback
477  if (tud_umount_cb) tud_umount_cb();
478  break;
479 
480  case DCD_EVENT_SETUP_RECEIVED:
481  TU_LOG2_VAR(&event.setup_received);
482  TU_LOG2("\r\n");
483 
484  // Mark as connected after receiving 1st setup packet.
485  // But it is easier to set it every time instead of wasting time to check then set
486  _usbd_dev.connected = 1;
487 
488  // mark both in & out control as free
489  _usbd_dev.ep_status[0][TUSB_DIR_OUT].busy = false;
490  _usbd_dev.ep_status[0][TUSB_DIR_OUT].claimed = 0;
491  _usbd_dev.ep_status[0][TUSB_DIR_IN ].busy = false;
492  _usbd_dev.ep_status[0][TUSB_DIR_IN ].claimed = 0;
493 
494  // Process control request
495  if ( !process_control_request(event.rhport, &event.setup_received) )
496  {
497  TU_LOG2(" Stall EP0\r\n");
498  // Failed -> stall both control endpoint IN and OUT
499  dcd_edpt_stall(event.rhport, 0);
500  dcd_edpt_stall(event.rhport, 0 | TUSB_DIR_IN_MASK);
501  }
502  break;
503 
504  case DCD_EVENT_XFER_COMPLETE:
505  {
506  // Invoke the class callback associated with the endpoint address
507  uint8_t const ep_addr = event.xfer_complete.ep_addr;
508  uint8_t const epnum = tu_edpt_number(ep_addr);
509  uint8_t const ep_dir = tu_edpt_dir(ep_addr);
510 
511  TU_LOG2("on EP %02X with %u bytes\r\n", ep_addr, (unsigned int) event.xfer_complete.len);
512 
513  _usbd_dev.ep_status[epnum][ep_dir].busy = false;
514  _usbd_dev.ep_status[epnum][ep_dir].claimed = 0;
515 
516  if ( 0 == epnum )
517  {
518  usbd_control_xfer_cb(event.rhport, ep_addr, (xfer_result_t)event.xfer_complete.result, event.xfer_complete.len);
519  }
520  else
521  {
522  usbd_class_driver_t const * driver = get_driver( _usbd_dev.ep2drv[epnum][ep_dir] );
523  TU_ASSERT(driver, );
524 
525  TU_LOG2(" %s xfer callback\r\n", driver->name);
526  driver->xfer_cb(event.rhport, ep_addr, (xfer_result_t)event.xfer_complete.result, event.xfer_complete.len);
527  }
528  }
529  break;
530 
531  case DCD_EVENT_SUSPEND:
532  TU_LOG2("\r\n");
533  if (tud_suspend_cb) tud_suspend_cb(_usbd_dev.remote_wakeup_en);
534  break;
535 
536  case DCD_EVENT_RESUME:
537  TU_LOG2("\r\n");
538  if (tud_resume_cb) tud_resume_cb();
539  break;
540 
541  case DCD_EVENT_SOF:
542  TU_LOG2("\r\n");
543  for ( uint8_t i = 0; i < TOTAL_DRIVER_COUNT; i++ )
544  {
545  usbd_class_driver_t const * driver = get_driver(i);
546  if ( driver->sof ) driver->sof(event.rhport);
547  }
548  break;
549 
550  case USBD_EVENT_FUNC_CALL:
551  TU_LOG2("\r\n");
552  if ( event.func_call.func ) event.func_call.func(event.func_call.param);
553  break;
554 
555  default:
556  TU_BREAKPOINT();
557  break;
558  }
559  }
560 }

566 // Helper to invoke class driver control request handler

574 // This handles the actual request and its response.
575 // return false will cause its caller to stall control endpoint
576 static bool process_control_request(uint8_t rhport, tusb_control_request_t const * p_request)
577 {
578  usbd_control_set_complete_callback(NULL);
579 
580  TU_ASSERT(p_request->bmRequestType_bit.type < TUSB_REQ_TYPE_INVALID);
581 
582  // Vendor request
583  if ( p_request->bmRequestType_bit.type == TUSB_REQ_TYPE_VENDOR )
584  {
585  TU_VERIFY(tud_vendor_control_xfer_cb);
586 
587  usbd_control_set_complete_callback(tud_vendor_control_xfer_cb);
588  return tud_vendor_control_xfer_cb(rhport, CONTROL_STAGE_SETUP, p_request);
589  }
590 
591 #if CFG_TUSB_DEBUG >= 2
592  if (TUSB_REQ_TYPE_STANDARD == p_request->bmRequestType_bit.type && p_request->bRequest <= TUSB_REQ_SYNCH_FRAME)
593  {
594  TU_LOG2(" %s", _tusb_std_request_str[p_request->bRequest]);
595  if (TUSB_REQ_GET_DESCRIPTOR != p_request->bRequest) TU_LOG2("\r\n");
596  }
597 #endif
598 
599  switch ( p_request->bmRequestType_bit.recipient )
600  {
601  //------------- Device Requests e.g in enumeration -------------//
602  case TUSB_REQ_RCPT_DEVICE:
603  if ( TUSB_REQ_TYPE_CLASS == p_request->bmRequestType_bit.type )
604  {
605  uint8_t const itf = tu_u16_low(p_request->wIndex);
606  TU_VERIFY(itf < TU_ARRAY_SIZE(_usbd_dev.itf2drv));
607 
608  usbd_class_driver_t const * driver = get_driver(_usbd_dev.itf2drv[itf]);
609  TU_VERIFY(driver);
610 
611  // forward to class driver: "non-STD request to Interface"
612  return invoke_class_control(rhport, driver, p_request);
613  }
614 
615  if ( TUSB_REQ_TYPE_STANDARD != p_request->bmRequestType_bit.type )
616  {
617  // Non standard request is not supported
618  TU_BREAKPOINT();
619  return false;
620  }
621 
622  switch ( p_request->bRequest )
623  {
624  case TUSB_REQ_SET_ADDRESS:
625  // Depending on mcu, status phase could be sent either before or after changing device address,
626  // or even require stack to not response with status at all
627  // Therefore DCD must take full responsibility to response and include zlp status packet if needed.
628  usbd_control_set_request(p_request); // set request since DCD has no access to tud_control_status() API
629  dcd_set_address(rhport, (uint8_t) p_request->wValue);
630  // skip tud_control_status()
631  _usbd_dev.addressed = 1;
632  break;
633 
634  case TUSB_REQ_GET_CONFIGURATION:
635  {
636  uint8_t cfg_num = _usbd_dev.cfg_num;
637  tud_control_xfer(rhport, p_request, &cfg_num, 1);
638  }
639  break;
640 
641  case TUSB_REQ_SET_CONFIGURATION:
642  {
643  uint8_t const cfg_num = (uint8_t) p_request->wValue;
644 
645  if ( !_usbd_dev.cfg_num && cfg_num ) TU_ASSERT( process_set_config(rhport, cfg_num) );
646  _usbd_dev.cfg_num = cfg_num;
647 
648  tud_control_status(rhport, p_request);
649  }
650  break;
651 
652  case TUSB_REQ_GET_DESCRIPTOR:
653  TU_VERIFY( process_get_descriptor(rhport, p_request) );
654  break;
655 
656  case TUSB_REQ_SET_FEATURE:
657  // Only support remote wakeup for device feature
658  TU_VERIFY(TUSB_REQ_FEATURE_REMOTE_WAKEUP == p_request->wValue);
659 
660  // Host may enable remote wake up before suspending especially HID device
661  _usbd_dev.remote_wakeup_en = true;
662  tud_control_status(rhport, p_request);
663  break;
664 
665  case TUSB_REQ_CLEAR_FEATURE:
666  // Only support remote wakeup for device feature
667  TU_VERIFY(TUSB_REQ_FEATURE_REMOTE_WAKEUP == p_request->wValue);
668 
669  // Host may disable remote wake up after resuming
670  _usbd_dev.remote_wakeup_en = false;
671  tud_control_status(rhport, p_request);
672  break;
673 
674  case TUSB_REQ_GET_STATUS:
675  {
676  // Device status bit mask
677  // - Bit 0: Self Powered
678  // - Bit 1: Remote Wakeup enabled
679  uint16_t status = (_usbd_dev.self_powered ? 1 : 0) | (_usbd_dev.remote_wakeup_en ? 2 : 0);
680  tud_control_xfer(rhport, p_request, &status, 2);
681  }
682  break;
683 
684  // Unknown/Unsupported request
685  default: TU_BREAKPOINT(); return false;
686  }
687  break;
688 
689  //------------- Class/Interface Specific Request -------------//
690  case TUSB_REQ_RCPT_INTERFACE:
691  {
692  uint8_t const itf = tu_u16_low(p_request->wIndex);
693  TU_VERIFY(itf < TU_ARRAY_SIZE(_usbd_dev.itf2drv));
694 
695  usbd_class_driver_t const * driver = get_driver(_usbd_dev.itf2drv[itf]);
696  TU_VERIFY(driver);
697 
698  // all requests to Interface (STD or Class) is forwarded to class driver.
699  // notable requests are: GET HID REPORT DESCRIPTOR, SET_INTERFACE, GET_INTERFACE
700  if ( !invoke_class_control(rhport, driver, p_request) )
701  {
702  // For GET_INTERFACE and SET_INTERFACE, it is mandatory to respond even if the class
703  // driver doesn't use alternate settings or implement this
704  TU_VERIFY(TUSB_REQ_TYPE_STANDARD == p_request->bmRequestType_bit.type);
705 
706  if (TUSB_REQ_GET_INTERFACE == p_request->bRequest)
707  {
708  uint8_t alternate = 0;
709  tud_control_xfer(rhport, p_request, &alternate, 1);
710  }else if (TUSB_REQ_SET_INTERFACE == p_request->bRequest)
711  {
712  tud_control_status(rhport, p_request);
713  } else
714  {
715  return false;
716  }
717  }
718  }
719  break;
720 
721  //------------- Endpoint Request -------------//
722  case TUSB_REQ_RCPT_ENDPOINT:
723  {
724  uint8_t const ep_addr = tu_u16_low(p_request->wIndex);
725  uint8_t const ep_num = tu_edpt_number(ep_addr);
726  uint8_t const ep_dir = tu_edpt_dir(ep_addr);
727 
728  TU_ASSERT(ep_num < TU_ARRAY_SIZE(_usbd_dev.ep2drv) );
729 
730  usbd_class_driver_t const * driver = get_driver(_usbd_dev.ep2drv[ep_num][ep_dir]);
731 
732  if ( TUSB_REQ_TYPE_STANDARD != p_request->bmRequestType_bit.type )
733  {
734  // Forward class request to its driver
735  TU_VERIFY(driver);
736  return invoke_class_control(rhport, driver, p_request);
737  }
738  else
739  {
740  // Handle STD request to endpoint
741  switch ( p_request->bRequest )
742  {
743  case TUSB_REQ_GET_STATUS:
744  {
745  uint16_t status = usbd_edpt_stalled(rhport, ep_addr) ? 0x0001 : 0x0000;
746  tud_control_xfer(rhport, p_request, &status, 2);
747  }
748  break;
749 
750  case TUSB_REQ_CLEAR_FEATURE:
751  case TUSB_REQ_SET_FEATURE:
752  {
753  if ( TUSB_REQ_FEATURE_EDPT_HALT == p_request->wValue )
754  {
755  if ( TUSB_REQ_CLEAR_FEATURE == p_request->bRequest )
756  {
757  usbd_edpt_clear_stall(rhport, ep_addr);
758  }else
759  {
760  usbd_edpt_stall(rhport, ep_addr);
761  }
762  }
763 
764  if (driver)
765  {
766  // Some classes such as USBTMC needs to clear/re-init its buffer when receiving CLEAR_FEATURE request
767  // We will also forward std request targeted endpoint to class drivers as well
768 
769  // STD request must always be ACKed regardless of driver returned value
770  // Also clear complete callback if driver set since it can also stall the request.
771  (void) invoke_class_control(rhport, driver, p_request);
772  usbd_control_set_complete_callback(NULL);
773 
774  // skip ZLP status if driver already did that
775  if ( !_usbd_dev.ep_status[0][TUSB_DIR_IN].busy ) tud_control_status(rhport, p_request);
776  }
777  }
778  break;
779 
780  // Unknown/Unsupported request
781  default: TU_BREAKPOINT(); return false;
782  }
783  }
784  }
785  break;
786 
787  // Unknown recipient
788  default: TU_BREAKPOINT(); return false;
789  }
790 
791  return true;
792 }

794 // Process Set Configure Request
795 // This function parse configuration descriptor & open drivers accordingly
796 static bool process_set_config(uint8_t rhport, uint8_t cfg_num)
797 {
798  tusb_desc_configuration_t const * desc_cfg = (tusb_desc_configuration_t const *) tud_descriptor_configuration_cb(cfg_num-1); // index is cfg_num-1
799  TU_ASSERT(desc_cfg != NULL && desc_cfg->bDescriptorType == TUSB_DESC_CONFIGURATION);
800 
801  // Parse configuration descriptor
802  _usbd_dev.remote_wakeup_support = (desc_cfg->bmAttributes & TUSB_DESC_CONFIG_ATT_REMOTE_WAKEUP) ? 1 : 0;
803  _usbd_dev.self_powered = (desc_cfg->bmAttributes & TUSB_DESC_CONFIG_ATT_SELF_POWERED) ? 1 : 0;
804 
805  // Parse interface descriptor
806  uint8_t const * p_desc = ((uint8_t const*) desc_cfg) + sizeof(tusb_desc_configuration_t);
807  uint8_t const * desc_end = ((uint8_t const*) desc_cfg) + desc_cfg->wTotalLength;
808 
809  while( p_desc < desc_end )
810  {
811  tusb_desc_interface_assoc_t const * desc_itf_assoc = NULL;
812 
813  // Class will always starts with Interface Association (if any) and then Interface descriptor
814  if ( TUSB_DESC_INTERFACE_ASSOCIATION == tu_desc_type(p_desc) )
815  {
816  desc_itf_assoc = (tusb_desc_interface_assoc_t const *) p_desc;
817  p_desc = tu_desc_next(p_desc); // next to Interface
818  }
819 
820  TU_ASSERT( TUSB_DESC_INTERFACE == tu_desc_type(p_desc) );
821 
822  tusb_desc_interface_t const * desc_itf = (tusb_desc_interface_t const*) p_desc;
823  uint16_t const remaining_len = desc_end-p_desc;
824 
825  uint8_t drv_id;
826  for (drv_id = 0; drv_id < TOTAL_DRIVER_COUNT; drv_id++)
827  {
828  usbd_class_driver_t const *driver = get_driver(drv_id);
829  uint16_t const drv_len = driver->open(rhport, desc_itf, remaining_len);
830 
831  if ( drv_len > 0 )
832  {
833  // Open successfully, check if length is correct
834  TU_ASSERT( sizeof(tusb_desc_interface_t) <= drv_len && drv_len <= remaining_len);
835 
836  // Interface number must not be used already
837  TU_ASSERT(DRVID_INVALID == _usbd_dev.itf2drv[desc_itf->bInterfaceNumber]);
838 
839  TU_LOG2(" %s opened\r\n", driver->name);
840  _usbd_dev.itf2drv[desc_itf->bInterfaceNumber] = drv_id;
841 
842  // If IAD exist, assign all interfaces to the same driver
843  if (desc_itf_assoc)
844  {
845  // IAD's first interface number and class should match with opened interface
846  TU_ASSERT(desc_itf_assoc->bFirstInterface == desc_itf->bInterfaceNumber &&
847  desc_itf_assoc->bFunctionClass == desc_itf->bInterfaceClass);
848 
849  for(uint8_t i=1; i<desc_itf_assoc->bInterfaceCount; i++)
850  {
851  _usbd_dev.itf2drv[desc_itf->bInterfaceNumber+i] = drv_id;
852  }
853  }
854 
855  mark_interface_endpoint(_usbd_dev.ep2drv, p_desc, drv_len, drv_id); // TODO refactor
856 
857  p_desc += drv_len; // next interface
858 
859  break;
860  }
861  }
862 
863  // Failed if cannot find supported driver
864  TU_ASSERT(drv_id < TOTAL_DRIVER_COUNT);
865  }
866 
867  // invoke callback
868  if (tud_mount_cb) tud_mount_cb();
869 
870  return true;
871 }

874 static void mark_interface_endpoint(uint8_t ep2drv[][2], uint8_t const* p_desc, uint16_t desc_len, uint8_t driver_id)

892 // return descriptor's buffer and update desc_len
893 static bool process_get_descriptor(uint8_t rhport, tusb_control_request_t const * p_request)
894 {
895  tusb_desc_type_t const desc_type = (tusb_desc_type_t) tu_u16_high(p_request->wValue);
896  uint8_t const desc_index = tu_u16_low( p_request->wValue );
897 
898  switch(desc_type)
899  {
900  case TUSB_DESC_DEVICE:
901  {
902  TU_LOG2(" Device\r\n");
903 
904  uint16_t len = sizeof(tusb_desc_device_t);
905 
906  // Only send up to EP0 Packet Size if not addressed
907  // This only happens with the very first get device descriptor and EP0 size = 8 or 16.
908  if ((CFG_TUD_ENDPOINT0_SIZE < sizeof(tusb_desc_device_t)) && !_usbd_dev.addressed)
909  {
910  len = CFG_TUD_ENDPOINT0_SIZE;
911 
912  // Hack here: we modify the request length to prevent usbd_control response with zlp
913  ((tusb_control_request_t*) p_request)->wLength = CFG_TUD_ENDPOINT0_SIZE;
914  }
915 
916  return tud_control_xfer(rhport, p_request, (void*) tud_descriptor_device_cb(), len);
917  }
918  break;
919 
920  case TUSB_DESC_BOS:
921  {
922  TU_LOG2(" BOS\r\n");
923 
924  // requested by host if USB > 2.0 ( i.e 2.1 or 3.x )
925  if (!tud_descriptor_bos_cb) return false;
926 
927  tusb_desc_bos_t const* desc_bos = (tusb_desc_bos_t const*) tud_descriptor_bos_cb();
928 
929  uint16_t total_len;
930  // Use offsetof to avoid pointer to the odd/misaligned address
931  memcpy(&total_len, (uint8_t*) desc_bos + offsetof(tusb_desc_bos_t, wTotalLength), 2);
932 
933  return tud_control_xfer(rhport, p_request, (void*) desc_bos, total_len);
934  }
935  break;
936 
937  case TUSB_DESC_CONFIGURATION:
938  {
939  TU_LOG2(" Configuration[%u]\r\n", desc_index);
940 
941  tusb_desc_configuration_t const* desc_config = (tusb_desc_configuration_t const*) tud_descriptor_configuration_cb(desc_index);
942  TU_ASSERT(desc_config);
943 
944  uint16_t total_len;
945  // Use offsetof to avoid pointer to the odd/misaligned address
946  memcpy(&total_len, (uint8_t*) desc_config + offsetof(tusb_desc_configuration_t, wTotalLength), 2);
947 
948  return tud_control_xfer(rhport, p_request, (void*) desc_config, total_len);
949  }
950  break;
951 
952  case TUSB_DESC_STRING:
953  {
954  TU_LOG2(" String[%u]\r\n", desc_index);
955 
956  // String Descriptor always uses the desc set from user
957  uint8_t const* desc_str = (uint8_t const*) tud_descriptor_string_cb(desc_index, p_request->wIndex);
958  TU_VERIFY(desc_str);
959 
960  // first byte of descriptor is its size
961  return tud_control_xfer(rhport, p_request, (void*) desc_str, desc_str[0]);
962  }
963  break;
964 
965  case TUSB_DESC_DEVICE_QUALIFIER:
966  TU_LOG2(" Device Qualifier\r\n");
967 
968  // Host sends this request to ask why our device with USB BCD from 2.0
969  // but is running at Full/Low Speed. If not highspeed capable stall this request,
970  // otherwise return the descriptor that could work in highspeed mode
971  if ( tud_descriptor_device_qualifier_cb )
972  {
973  uint8_t const* desc_qualifier = tud_descriptor_device_qualifier_cb();
974  TU_ASSERT(desc_qualifier);
975 
976  // first byte of descriptor is its size
977  return tud_control_xfer(rhport, p_request, (void*) desc_qualifier, desc_qualifier[0]);
978  }else
979  {
980  return false;
981  }
982  break;
983 
984  case TUSB_DESC_OTHER_SPEED_CONFIG:
985  TU_LOG2(" Other Speed Configuration\r\n");
986 
987  // After Device Qualifier descriptor is received host will ask for this descriptor
988  return false; // not supported
989  break;
990 
991  default: return false;
992  }
993 }

995 //--------------------------------------------------------------------+
996 // DCD Event Handler
997 //--------------------------------------------------------------------+
998 void dcd_event_handler(dcd_event_t const * event, bool in_isr)
999 {
1000  switch (event->event_id)
1001  {
1002  case DCD_EVENT_UNPLUGGED:
1003  // UNPLUGGED event can be bouncing, only processing if we are currently connected
1004  if ( _usbd_dev.connected )
1005  {
1006  _usbd_dev.connected = 0;
1007  _usbd_dev.addressed = 0;
1008  _usbd_dev.cfg_num = 0;
1009  _usbd_dev.suspended = 0;
1010  osal_queue_send(_usbd_q, event, in_isr);
1011  }
1012  break;
1013 
1014  case DCD_EVENT_SOF:
1015  return; // skip SOF event for now
1016  break;
1017 
1018  case DCD_EVENT_SUSPEND:
1019  // NOTE: When plugging/unplugging device, the D+/D- state are unstable and
1020  // can accidentally meet the SUSPEND condition ( Bus Idle for 3ms ).
1021  // In addition, some MCUs such as SAMD or boards that haven no VBUS detection cannot distinguish
1022  // suspended vs disconnected. We will skip handling SUSPEND/RESUME event if not currently connected
1023  if ( _usbd_dev.connected )
1024  {
1025  _usbd_dev.suspended = 1;
1026  osal_queue_send(_usbd_q, event, in_isr);
1027  }
1028  break;
1029 
1030  case DCD_EVENT_RESUME:
1031  // skip event if not connected (especially required for SAMD)
1032  if ( _usbd_dev.connected )
1033  {
1034  _usbd_dev.suspended = 0;
1035  osal_queue_send(_usbd_q, event, in_isr);
1036  }
1037  break;
1038 
1039  default:
1040  osal_queue_send(_usbd_q, event, in_isr);
1041  break;
1042  }
1043 }
