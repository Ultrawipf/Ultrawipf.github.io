131 void HID_CommandInterface::hidCmdCallback(HID_CMD_Data_t* data){
132  ParsedCommand cmd;
133  cmd.cmdId = data->cmd;
134  cmd.instance = data->instance;
135  cmd.val = data->data;
136  cmd.adr = data->addr;
137 
138  // Translate type
139  if(data->type == HidCmdType::write){
140  cmd.type = CMDtype::set;
141  }else if(data->type == HidCmdType::request){
142  cmd.type = CMDtype::get;
143  }else if(data->type == HidCmdType::info){
144  cmd.type = CMDtype::info;
145  }else if(data->type == HidCmdType::writeAddr){
146  cmd.type = CMDtype::setat;
147  }else if(data->type == HidCmdType::requestAddr){
148  cmd.type = CMDtype::getat;
149  }
150 
151 
152 
153  if(data->instance != 0xff){
154  cmd.target = CommandHandler::getHandlerFromId(data->clsid,data->instance);
155  if(cmd.target == nullptr || !(cmd.target->isValidCommandId(cmd.cmdId, CMDFLAG_STR_ONLY))){
156  data->type = HidCmdType::notFound;
157  //sendHidCmd(data); // Send back error
158  this->outBuffer.push_back(*data);
159  threadSem.Give();
160  return;
161  }
162  if(cmd.target->isValidCommandId(cmd.cmdId, CMDFLAG_STR_ONLY))
163  commands.push_back(cmd);
164  }else{
165  std::vector<CommandHandler*> handlers = CommandHandler::getHandlersFromId(data->clsid);
166  for(CommandHandler* handler : handlers){
167  ParsedCommand newCmd = cmd;
168  newCmd.target = handler;
169  if(newCmd.target == nullptr || !(cmd.target->isValidCommandId(cmd.cmdId, CMDFLAG_STR_ONLY))){
170  data->type = HidCmdType::notFound;
171  //sendHidCmd(data); // Send back error
172  this->outBuffer.push_back(*data);
173  threadSem.Give();
174  return;
175  }
176  commands.push_back(newCmd);
177  }
178  }
179 
180  if(!commands.empty()){
181  parserReady = true; // Signals that we should execute commands in the thread
182  FFBoardMainCommandThread::wakeUp();
183  }
184 }
