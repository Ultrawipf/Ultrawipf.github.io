210  this->setPidPrecision(pidPrecision);

278 float TMC4671::getTemp(){
279  if(!this->conf.hwconf.temperatureEnabled){
280  return 0;
281  }
282  TMC4671HardwareTypeConf* hwconf = &conf.hwconf;
283 
284  writeReg(0x03, 2);
285  int32_t adcval = ((readReg(0x02)) & 0xffff) - 0x7fff; // Center offset
286  adcval -= hwconf->adcOffset;
287  if(adcval <= 0){
288  return 0.0;
289  }
290  float r = hwconf->thermistor_R2 * (((float)43252 / (float)adcval)); //43252 equivalent ADC count if it was 3.3V and not 2.5V
291 
292  // Beta
293  r = (1.0 / 298.15) + log(r / hwconf->thermistor_R) / hwconf->thermistor_Beta;
294  r = 1.0 / r;
295  r -= 273.15;
296  return r;

304 void TMC4671::Run(){
305  // Main state machine
306  while(1){
307 
308  switch(this->state){
309 
310  case TMC_ControlState::Running:
311  {
312  // Check status, Temps, Everything alright?
313  uint32_t tick = HAL_GetTick();
314  if(tick - lastStatTime > 2000){ // Every 2s
315  lastStatTime = tick;
316  statusCheck();
317  // Get enable input. If tmc does not reply the result will read 0 or 0xffffffff (not possible normally)
318  uint32_t pins = readReg(0x76);
319  bool tmc_en = ((pins >> 15) & 0x01) && pins != 0xffffffff;
320  if(!tmc_en && active){ // Hardware emergency.
321  this->estopTriggered = true;
322  this->emergencyStop();
323  changeState(TMC_ControlState::HardError);
324  }
325 
326  // Temperature sense
327  if(conf.hwconf.temperatureEnabled){
328  float temp = getTemp();
329  if(temp > conf.hwconf.temp_limit){
330  changeState(TMC_ControlState::OverTemp);
331  pulseErrLed();
332  }
333  }
334 
335  }
336  Delay(200);
337  }
338  break;
339 
340  case TMC_ControlState::Init_wait:
341  if(active && hasPower()){
342  if(HAL_GetTick() - initTime > (emergency ? 5000 : 1000)){
343  emergency = false;
344  if(!initialize()){
345  pulseErrLed();
346  }
347  }
348  }
349 
350  break;
351 
352  case TMC_ControlState::ABN_init:
353  ABN_init();
354  break;
355 
356  case TMC_ControlState::AENC_init:
357  AENC_init();
358  break;
359 
360  case TMC_ControlState::HardError:
361 
362  break; // Broken
363 
364  case TMC_ControlState::OverTemp:
365  this->stopMotor();
366  changeState(TMC_ControlState::HardError); // Block
367  break;
368 
369  case TMC_ControlState::EncoderFinished: // Startup sequence done
370  setEncoderIndexFlagEnabled(true); // TODO
371  if(active){
372  startMotor();
373  changeState(TMC_ControlState::Running);
374  }else{
375  stopMotor();
376  laststate = TMC_ControlState::Running; // Go to running when starting again
377  }
378  break;
379 
380  case TMC_ControlState::No_power:
381  if(hasPower() && !emergency){
382  changeState(laststateNopower);
383  setMotionMode(lastMotionMode,true);
384  ErrorHandler::clearError(ErrorCode::undervoltage);
385  enablePin.set();
386  }
387  pulseErrLed();
388  Delay(100); // wait a bit more
389  break;
390 
391  default:
392 
393  break;
394  }
395 
396  // Optional update methods for safety
397 
398  if(!hasPower() && state != TMC_ControlState::No_power){ // low voltage or overvoltage
399  lastMotionMode = curMotionMode;
400  laststateNopower = state;
401  ErrorHandler::addError(lowVoltageError);
402  setMotionMode(MotionMode::stop,true); // Disable tmc
403  changeState(TMC_ControlState::No_power);
404  enablePin.reset();
405  }
406 
407  if(flagCheckInProgress){ // cause some delay until reenabling the status interrupt checking
408  setStatusFlags(0);
409  flagCheckInProgress = false;
410  }
411  Delay(10);
412  } // End while
413 }

566 // Rotates motor to find min and max values of the encoder
567 void TMC4671::calibrateAenc(){
568 
569  // Rotate and measure min/max
570  blinkClipLed(250, 0);
571  PhiE lastphie = getPhiEtype();
572  MotionMode lastmode = getMotionMode();
573  //int32_t pos = getPos();
574  PosSelection possel = this->conf.motconf.pos_sel;
575  setPosSel(PosSelection::PhiE_openloop);
576  setPos(0);
577  // Ramp up flux
578  setFluxTorque(0, 0);
579  writeReg(0x23,0); // set phie openloop 0
580  setPhiEtype(PhiE::openloop);
581  setMotionMode(MotionMode::torque,true);
582 
583  if(this->conf.motconf.motor_type == MotorType::STEPPER || this->conf.motconf.motor_type == MotorType::BLDC)
584  for(int16_t flux = 0; flux <= bangInitPower; flux+=10){
585  setFluxTorque(flux, 0);
586  Delay(10);
587  }
588 
589  uint32_t minVal_0 = 0xffff, minVal_1 = 0xffff, minVal_2 = 0xffff;
590  uint32_t maxVal_0 = 0, maxVal_1 = 0, maxVal_2 = 0;
591  int32_t minpos = -0x8fff/std::max<int32_t>(1,std::min<int32_t>(this->aencconf.cpr/4,20)), maxpos = 0x8fff/std::max<int32_t>(1,std::min<int32_t>(this->aencconf.cpr/4,20));
592  uint32_t speed = std::max<uint32_t>(1,20/std::max<uint32_t>(1,this->aencconf.cpr/10));
593 
594  if(this->conf.motconf.motor_type != MotorType::STEPPER && this->conf.motconf.motor_type != MotorType::BLDC){
595  speed*=10; // dc motors turn at a random speed. reduce the rotation time a bit by increasing openloop speed
596  }
597 
598  runOpenLoop(bangInitPower, 0, speed, 100,true);
599 
600  uint8_t stage = 0;
601  int32_t poles = conf.motconf.pole_pairs;
602  int32_t initialDirPos = 0;
603  while(stage != 3){
604  Delay(2);
605  if(getPos() > maxpos*poles && stage == 0){
606  runOpenLoop(bangInitPower, 0, -speed, 100,true);
607  stage = 1;
608  }else if(getPos() < minpos*poles && stage == 1){
609  // Scale might still be wrong... maxVal-minVal is too high. In theory its 0xffff range and scaler /256. Leave some room to prevent clipping
610  aencconf.AENC0_offset = ((maxVal_0 + minVal_0) / 2);
611  aencconf.AENC0_scale = 0xF6FF00 / (maxVal_0 - minVal_0);
612  if(conf.motconf.enctype == EncoderType_TMC::uvw){
613  aencconf.AENC1_offset = ((maxVal_1 + minVal_1) / 2);
614  aencconf.AENC1_scale = 0xF6FF00 / (maxVal_1 - minVal_1);
615  }
616 
617  aencconf.AENC2_offset = ((maxVal_2 + minVal_2) / 2);
618  aencconf.AENC2_scale = 0xF6FF00 / (maxVal_2 - minVal_2);
619  aencconf.rdir = false;
620  setup_AENC(aencconf);
621  runOpenLoop(0, 0, 0, 1000,true);
622  Delay(250);
623  // Zero aenc
624  writeReg(0x41, 0);
625  initialDirPos = readReg(0x41);
626  runOpenLoop(bangInitPower, 0, speed, 100,true);
627  stage = 2;
628  }else if(getPos() > 0 && stage == 2){
629  stage = 3;
630  runOpenLoop(0, 0, 0, 1000,true);
631  }
632 
633  writeReg(0x03,2);
634  uint32_t aencUX = readReg(0x02)>>16;
635  writeReg(0x03,3);
636  uint32_t aencWY_VN = readReg(0x02) ;
637  uint32_t aencWY = aencWY_VN >> 16;
638  uint32_t aencVN = aencWY_VN & 0xffff;
639 
640  minVal_0 = std::min(minVal_0,aencUX);
641  minVal_1 = std::min(minVal_1,aencVN);
642  minVal_2 = std::min(minVal_2,aencWY);
643

672 bool TMC4671::checkEncoder(){
673  if(this->conf.motconf.motor_type != MotorType::STEPPER && this->conf.motconf.motor_type != MotorType::BLDC){ // If not stepper or bldc return
674  return true;
675  }
676  blinkClipLed(150, 0);
677  uint8_t phiEreg = 0;
678  if(conf.motconf.enctype == EncoderType_TMC::abn){
679  phiEreg = 0x2A;
680  }else if(conf.motconf.enctype == EncoderType_TMC::sincos || conf.motconf.enctype == EncoderType_TMC::uvw){
681  phiEreg = 0x46;
682  }
683 
684  const uint16_t maxcount = 50;
685  const int16_t startAngle = 0;
686  const int16_t targetAngle = 0x3FFF;
687 
688  bool result = true;
689  PhiE lastphie = getPhiEtype();
690  MotionMode lastmode = getMotionMode();
691  setFluxTorque(0, 0);
692  setPhiEtype(PhiE::ext);
693 
694  setPhiE_ext(startAngle);
695  // Ramp up flux
696  for(int16_t flux = 0; flux <= bangInitPower; flux+=20){
697  setFluxTorque(flux, 0);
698  Delay(2);
699  }
700 
701  //Forward
702  int16_t phiE_enc = 0;
703  uint16_t failcount = 0;
704  int16_t revCount = 0;
705  for(int16_t angle = startAngle;angle<targetAngle;angle+=0x00ff){
706  uint16_t c = 0;
707  setPhiE_ext(angle);
708  Delay(10);
709  phiE_enc = (int16_t)(readReg(phiEreg)>>16);
710  int16_t err = abs(phiE_enc - angle);
711  int16_t nErr = abs(phiE_enc + angle);
712  // Wait more
713  while(err > 2500 && nErr > 2500 && c++ < 50){
714  phiE_enc = (int16_t)(readReg(phiEreg)>>16);
715  err = abs(phiE_enc - angle);
716  nErr = abs(angle - phiE_enc);
717  Delay(10);
718  }
719  if(err > nErr){
720  revCount++;
721  }
722  if(c >= maxcount){
723  failcount++;
724  if(failcount > 10){
725  result = false;
726  break;
727  }
728  }
729  }
730 
731  // Backward
732  if(result){ // Only if not already failed
733  for(int16_t angle = targetAngle;angle>startAngle;angle -= 0x00ff){
734  uint16_t c = 0;
735  setPhiE_ext(angle);
736  Delay(10);
737  phiE_enc = (int16_t)(readReg(phiEreg)>>16);
738  int16_t err = abs(phiE_enc - angle);
739  int16_t nErr = abs(phiE_enc + angle);
740  // Wait more
741  while(err > 2500 && nErr > 2500 && c++ < 50){
742  phiE_enc = (int16_t)(readReg(phiEreg)>>16);
743  err = abs(phiE_enc - angle);
744  nErr = abs(angle - phiE_enc);
745  Delay(10);
746  }
747  if(err > nErr){
748  revCount++;
749  }
750  if(c >= maxcount){
751  failcount++;
752  if(failcount > 10){
753  result = false;
754  break;
755  }
756  }
757  }
758  }
759  if(revCount > maxcount){ // Encoder seems reversed
760  // reverse encoder
761  if(this->conf.motconf.enctype == EncoderType_TMC::abn){
762  this->abnconf.rdir = !this->abnconf.rdir;
763  setup_ABN_Enc(abnconf);
764  }else if(this->conf.motconf.enctype == EncoderType_TMC::sincos || this->conf.motconf.enctype == EncoderType_TMC::uvw){
765  this->aencconf.rdir = !this->aencconf.rdir;
766  setup_AENC(aencconf);
767  }
768  }
769 
770  setFluxTorque(0, 0);
771  setPhiE_ext(0);
772  setPhiEtype(lastphie);
773  setMotionMode(lastmode,true);
774 
775  if(result){
776  encstate = ENC_InitState::OK;
777  }
778  blinkClipLed(0, 0);
779  return result;

849 bool TMC4671::calibrateAdcOffset(uint16_t time){
850 
851  uint16_t measuretime_idle = time;
852  uint32_t measurements_idle = 0;

854  uint64_t totalB=0;

865  //enablePin.reset();

868  uint32_t adcraw = readReg(0x02);

877  lastrawB = rawB;
