162 {

164 
165  USB_OTG_GlobalTypeDef * usb_otg = GLOBAL_BASE(rhport);
166 
167  // Determine largest EP size for RX FIFO
168  uint16_t max_epsize = 0;

178 // Setup the control endpoint 0.
179 static void bus_reset(uint8_t rhport)
180 {
181  (void) rhport;
182 
183  USB_OTG_GlobalTypeDef * usb_otg = GLOBAL_BASE(rhport);
184  USB_OTG_DeviceTypeDef * dev = DEVICE_BASE(rhport);
185  USB_OTG_OUTEndpointTypeDef * out_ep = OUT_EP_BASE(rhport);
186  USB_OTG_INEndpointTypeDef * in_ep = IN_EP_BASE(rhport);
187 
188  tu_memclr(xfer_status, sizeof(xfer_status));
189  _out_ep_closed = false;
190 
191  for(uint8_t n = 0; n < EP_MAX; n++) {
192  out_ep[n].DOEPCTL |= USB_OTG_DOEPCTL_SNAK;
193  }
194 
195  dev->DAINTMSK |= (1 << USB_OTG_DAINTMSK_OEPM_Pos) | (1 << USB_OTG_DAINTMSK_IEPM_Pos);
196  dev->DOEPMSK |= USB_OTG_DOEPMSK_STUPM | USB_OTG_DOEPMSK_XFRCM;
197  dev->DIEPMSK |= USB_OTG_DIEPMSK_TOM | USB_OTG_DIEPMSK_XFRCM;
198 
199  // "USB Data FIFOs" section in reference manual
200  // Peripheral FIFO architecture
201  //
202  // The FIFO is split up in a lower part where the RX FIFO is located and an upper part where the TX FIFOs start.
203  // We do this to allow the RX FIFO to grow dynamically which is possible since the free space is located
204  // between the RX and TX FIFOs. This is required by ISO OUT EPs which need a bigger FIFO than the standard
205  // configuration done below.
206  //
207  // Dynamically FIFO sizes are of interest only for ISO EPs since all others are usually not opened and closed.
208  // All EPs other than ISO are opened as soon as the driver starts up i.e. when the host sends a
209  // configure interface command. Hence, all IN EPs other the ISO will be located at the top. IN ISO EPs are usually
210  // opened when the host sends an additional command: setInterface. At this point in time
211  // the ISO EP will be located next to the free space and can change its size. In case more IN EPs change its size
212  // an additional memory
213  //
214  // --------------- 320 or 1024 ( 1280 or 4096 bytes )
215  // | IN FIFO 0 |
216  // --------------- (320 or 1024) - 16
217  // | IN FIFO 1 |
218  // --------------- (320 or 1024) - 16 - x
219  // | . . . . |
220  // --------------- (320 or 1024) - 16 - x - y - ... - z
221  // | IN FIFO MAX |
222  // ---------------
223  // | FREE |
224  // --------------- GRXFSIZ
225  // | OUT FIFO |
226  // | ( Shared ) |
227  // --------------- 0
228  //
229  // According to "FIFO RAM allocation" section in RM, FIFO RAM are allocated as follows (each word 32-bits):
230  // - Each EP IN needs at least max packet size, 16 words is sufficient for EP0 IN
231  //
232  // - All EP OUT shared a unique OUT FIFO which uses
233  // - 13 for setup packets + control words (up to 3 setup packets).
234  // - 1 for global NAK (not required/used here).
235  // - Largest-EPsize / 4 + 1. ( FS: 64 bytes, HS: 512 bytes). Recommended is "2 x (Largest-EPsize/4) + 1"
236  // - 2 for each used OUT endpoint
237  //
238  // Therefore GRXFSIZ = 13 + 1 + 1 + 2 x (Largest-EPsize/4) + 2 x EPOUTnum
239  // - FullSpeed (64 Bytes ): GRXFSIZ = 15 + 2 x 16 + 2 x EP_MAX = 47 + 2 x EP_MAX
240  // - Highspeed (512 bytes): GRXFSIZ = 15 + 2 x 128 + 2 x EP_MAX = 271 + 2 x EP_MAX
241  //
242  // NOTE: Largest-EPsize & EPOUTnum is actual used endpoints in configuration. Since DCD has no knowledge
243  // of the overall picture yet. We will use the worst scenario: largest possible + EP_MAX
244  //
245  // For Isochronous, largest EP size can be 1023/1024 for FS/HS respectively. In addition if multiple ISO
246  // are enabled at least "2 x (Largest-EPsize/4) + 1" are recommended. Maybe provide a macro for application to
247  // overwrite this.
248 
249  usb_otg->GRXFSIZ = calc_rx_ff_size(TUD_OPT_HIGH_SPEED ? 512 : 64);
250 
251  _allocated_fifo_words_tx = 16;
252 
253  // Control IN uses FIFO 0 with 64 bytes ( 16 32-bit word )
254  usb_otg->DIEPTXF0_HNPTXFSIZ = (16 << USB_OTG_TX0FD_Pos) | (EP_FIFO_SIZE/4 - _allocated_fifo_words_tx);
255 
256  // Fixed control EP0 size to 64 bytes
257  in_ep[0].DIEPCTL &= ~(0x03 << USB_OTG_DIEPCTL_MPSIZ_Pos);
258  xfer_status[0][TUSB_DIR_OUT].max_size = xfer_status[0][TUSB_DIR_IN].max_size = 64;
259 
260  out_ep[0].DOEPTSIZ |= (3 << USB_OTG_DOEPTSIZ_STUPCNT_Pos);
261 
262  usb_otg->GINTMSK |= USB_OTG_GINTMSK_OEPINT | USB_OTG_GINTMSK_IEPINT;
263 }

265 // Set turn-around timeout according to link speed
266 extern uint32_t SystemCoreClock;
267 static void set_turnaround(USB_OTG_GlobalTypeDef * usb_otg, tusb_speed_t speed)
268 {
269  usb_otg->GUSBCFG &= ~USB_OTG_GUSBCFG_TRDT;
270 
271  if ( speed == TUSB_SPEED_HIGH )
272  {
273  // Use fixed 0x09 for Highspeed
274  usb_otg->GUSBCFG |= (0x09 << USB_OTG_GUSBCFG_TRDT_Pos);
275  }
276  else
277  {
278  // Turnaround timeout depends on the MCU clock
279  uint32_t turnaround;
280 
281  if ( SystemCoreClock >= 32000000U )
282  turnaround = 0x6U;
283  else if ( SystemCoreClock >= 27500000U )
284  turnaround = 0x7U;
285  else if ( SystemCoreClock >= 24000000U )
286  turnaround = 0x8U;
287  else if ( SystemCoreClock >= 21800000U )
288  turnaround = 0x9U;
289  else if ( SystemCoreClock >= 20000000U )
290  turnaround = 0xAU;
291  else if ( SystemCoreClock >= 18500000U )
292  turnaround = 0xBU;
293  else if ( SystemCoreClock >= 17200000U )
294  turnaround = 0xCU;
295  else if ( SystemCoreClock >= 16000000U )
296  turnaround = 0xDU;
297  else if ( SystemCoreClock >= 15000000U )
298  turnaround = 0xEU;
299  else
300  turnaround = 0xFU;
301 
302  // Fullspeed depends on MCU clocks, but we will use 0x06 for 32+ Mhz
303  usb_otg->GUSBCFG |= (turnaround << USB_OTG_GUSBCFG_TRDT_Pos);
304  }
305 }

376 static void edpt_schedule_packets(uint8_t rhport, uint8_t const epnum, uint8_t const dir, uint16_t const num_packets, uint16_t total_bytes)
377 {
378  (void) rhport;
379 
380  USB_OTG_DeviceTypeDef * dev = DEVICE_BASE(rhport);
381  USB_OTG_OUTEndpointTypeDef * out_ep = OUT_EP_BASE(rhport);
382  USB_OTG_INEndpointTypeDef * in_ep = IN_EP_BASE(rhport);
383 
384  // EP0 is limited to one packet each xfer
385  // We use multiple transaction of xfer->max_size length to get a whole transfer done
386  if(epnum == 0) {
387  xfer_ctl_t * const xfer = XFER_CTL_BASE(epnum, dir);
388  total_bytes = tu_min16(ep0_pending[dir], xfer->max_size);
389  ep0_pending[dir] -= total_bytes;
390  }
391 
392  // IN and OUT endpoint xfers are interrupt-driven, we just schedule them here.
393  if(dir == TUSB_DIR_IN) {
394  // A full IN transfer (multiple packets, possibly) triggers XFRC.
395  in_ep[epnum].DIEPTSIZ = (num_packets << USB_OTG_DIEPTSIZ_PKTCNT_Pos) |
396  ((total_bytes << USB_OTG_DIEPTSIZ_XFRSIZ_Pos) & USB_OTG_DIEPTSIZ_XFRSIZ_Msk);
397 
398  in_ep[epnum].DIEPCTL |= USB_OTG_DIEPCTL_EPENA | USB_OTG_DIEPCTL_CNAK;
399  // For ISO endpoint set correct odd/even bit for next frame.
400  if ((in_ep[epnum].DIEPCTL & USB_OTG_DIEPCTL_EPTYP) == USB_OTG_DIEPCTL_EPTYP_0 && (XFER_CTL_BASE(epnum, dir))->interval == 1)
401  {
402  // Take odd/even bit from frame counter.
403  uint32_t const odd_frame_now = (dev->DSTS & (1u << USB_OTG_DSTS_FNSOF_Pos));
404  in_ep[epnum].DIEPCTL |= (odd_frame_now ? USB_OTG_DIEPCTL_SD0PID_SEVNFRM_Msk : USB_OTG_DIEPCTL_SODDFRM_Msk);
405  }
406  // Enable fifo empty interrupt only if there are something to put in the fifo.
407  if(total_bytes != 0) {
408  dev->DIEPEMPMSK |= (1 << epnum);
409  }
410  } else {
411  // A full OUT transfer (multiple packets, possibly) triggers XFRC.
412  out_ep[epnum].DOEPTSIZ &= ~(USB_OTG_DOEPTSIZ_PKTCNT_Msk | USB_OTG_DOEPTSIZ_XFRSIZ);
413  out_ep[epnum].DOEPTSIZ |= (num_packets << USB_OTG_DOEPTSIZ_PKTCNT_Pos) |
414  ((total_bytes << USB_OTG_DOEPTSIZ_XFRSIZ_Pos) & USB_OTG_DOEPTSIZ_XFRSIZ_Msk);
415 
416  out_ep[epnum].DOEPCTL |= USB_OTG_DOEPCTL_EPENA | USB_OTG_DOEPCTL_CNAK;
417  if ((out_ep[epnum].DOEPCTL & USB_OTG_DOEPCTL_EPTYP) == USB_OTG_DOEPCTL_EPTYP_0 && (XFER_CTL_BASE(epnum, dir))->interval == 1)
418  {
419  // Take odd/even bit from frame counter.
420  uint32_t const odd_frame_now = (dev->DSTS & (1u << USB_OTG_DSTS_FNSOF_Pos));
421  out_ep[epnum].DOEPCTL |= (odd_frame_now ? USB_OTG_DOEPCTL_SD0PID_SEVNFRM_Msk : USB_OTG_DOEPCTL_SODDFRM_Msk);
422  }
423  }

429 void dcd_init (uint8_t rhport)
430 {
431  // Programming model begins in the last section of the chapter on the USB
432  // peripheral in each Reference Manual.
433 
434  USB_OTG_GlobalTypeDef * usb_otg = GLOBAL_BASE(rhport);
435 
436  // No HNP/SRP (no OTG support), program timeout later.
437  if ( rhport == 1 )
438  {
439  // On selected MCUs HS port1 can be used with external PHY via ULPI interface
440 #if CFG_TUSB_RHPORT1_MODE & OPT_MODE_HIGH_SPEED
441  // deactivate internal PHY
442  usb_otg->GCCFG &= ~USB_OTG_GCCFG_PWRDWN;
443 
444  // Init The UTMI Interface
445  usb_otg->GUSBCFG &= ~(USB_OTG_GUSBCFG_TSDPS | USB_OTG_GUSBCFG_ULPIFSLS | USB_OTG_GUSBCFG_PHYSEL);
446 
447  // Select default internal VBUS Indicator and Drive for ULPI
448  usb_otg->GUSBCFG &= ~(USB_OTG_GUSBCFG_ULPIEVBUSD | USB_OTG_GUSBCFG_ULPIEVBUSI);
449 #else
450  usb_otg->GUSBCFG |= USB_OTG_GUSBCFG_PHYSEL;
451 #endif
452 
453 #if defined(USB_HS_PHYC)
454  // Highspeed with embedded UTMI PHYC
455 
456  // Select UTMI Interface
457  usb_otg->GUSBCFG &= ~USB_OTG_GUSBCFG_ULPI_UTMI_SEL;
458  usb_otg->GCCFG |= USB_OTG_GCCFG_PHYHSEN;
459 
460  // Enables control of a High Speed USB PHY
461  USB_HS_PHYCInit();
462 #endif
463  } else
464  {
465  // Enable internal PHY
466  usb_otg->GUSBCFG |= USB_OTG_GUSBCFG_PHYSEL;
467  }
468 
469  // Reset core after selecting PHY
470  // Wait AHB IDLE, reset then wait until it is cleared
471  while ((usb_otg->GRSTCTL & USB_OTG_GRSTCTL_AHBIDL) == 0U) {}
472  usb_otg->GRSTCTL |= USB_OTG_GRSTCTL_CSRST;
473  while ((usb_otg->GRSTCTL & USB_OTG_GRSTCTL_CSRST) == USB_OTG_GRSTCTL_CSRST) {}
474 
475  // Restart PHY clock
476  *((volatile uint32_t *)(RHPORT_REGS_BASE + USB_OTG_PCGCCTL_BASE)) = 0;
477 
478  // Clear all interrupts
479  usb_otg->GINTSTS |= usb_otg->GINTSTS;
480 
481  // Required as part of core initialization.
482  // TODO: How should mode mismatch be handled? It will cause
483  // the core to stop working/require reset.
484  usb_otg->GINTMSK |= USB_OTG_GINTMSK_OTGINT | USB_OTG_GINTMSK_MMISM;
485 
486  USB_OTG_DeviceTypeDef * dev = DEVICE_BASE(rhport);
487 
488  // If USB host misbehaves during status portion of control xfer
489  // (non zero-length packet), send STALL back and discard.
490  dev->DCFG |= USB_OTG_DCFG_NZLSOHSK;
491

556 bool dcd_edpt_open (uint8_t rhport, tusb_desc_endpoint_t const * desc_edpt)
557 {
558  (void) rhport;
559 
560  USB_OTG_GlobalTypeDef * usb_otg = GLOBAL_BASE(rhport);
561  USB_OTG_DeviceTypeDef * dev = DEVICE_BASE(rhport);
562  USB_OTG_OUTEndpointTypeDef * out_ep = OUT_EP_BASE(rhport);
563  USB_OTG_INEndpointTypeDef * in_ep = IN_EP_BASE(rhport);
564 
565  uint8_t const epnum = tu_edpt_number(desc_edpt->bEndpointAddress);
566  uint8_t const dir = tu_edpt_dir(desc_edpt->bEndpointAddress);
567 
568  TU_ASSERT(epnum < EP_MAX);
569 
570  xfer_ctl_t * xfer = XFER_CTL_BASE(epnum, dir);
571  xfer->max_size = desc_edpt->wMaxPacketSize.size;
572  xfer->interval = desc_edpt->bInterval;
573 
574  uint16_t const fifo_size = (desc_edpt->wMaxPacketSize.size + 3) / 4; // Round up to next full word
575 
576  if(dir == TUSB_DIR_OUT)
577  {
578  // Calculate required size of RX FIFO
579  uint16_t const sz = calc_rx_ff_size(4*fifo_size);
580 
581  // If size_rx needs to be extended check if possible and if so enlarge it
582  if (usb_otg->GRXFSIZ < sz)
583  {
584  TU_ASSERT(sz + _allocated_fifo_words_tx <= EP_FIFO_SIZE/4);
585 
586  // Enlarge RX FIFO
587  usb_otg->GRXFSIZ = sz;
588  }
589 
590  out_ep[epnum].DOEPCTL |= (1 << USB_OTG_DOEPCTL_USBAEP_Pos) |
591  (desc_edpt->bmAttributes.xfer << USB_OTG_DOEPCTL_EPTYP_Pos) |
592  (desc_edpt->wMaxPacketSize.size << USB_OTG_DOEPCTL_MPSIZ_Pos);
593 
594  dev->DAINTMSK |= (1 << (USB_OTG_DAINTMSK_OEPM_Pos + epnum));
595  }
596  else
597  {
598  // "USB Data FIFOs" section in reference manual
599  // Peripheral FIFO architecture
600  //
601  // --------------- 320 or 1024 ( 1280 or 4096 bytes )
602  // | IN FIFO 0 |
603  // --------------- (320 or 1024) - 16
604  // | IN FIFO 1 |
605  // --------------- (320 or 1024) - 16 - x
606  // | . . . . |
607  // --------------- (320 or 1024) - 16 - x - y - ... - z
608  // | IN FIFO MAX |
609  // ---------------
610  // | FREE |
611  // --------------- GRXFSIZ
612  // | OUT FIFO |
613  // | ( Shared ) |
614  // --------------- 0
615  //
616  // In FIFO is allocated by following rules:
617  // - IN EP 1 gets FIFO 1, IN EP "n" gets FIFO "n".
618 
619  // Check if free space is available
620  TU_ASSERT(_allocated_fifo_words_tx + fifo_size + usb_otg->GRXFSIZ <= EP_FIFO_SIZE/4);
621 
622  _allocated_fifo_words_tx += fifo_size;
623 
624  // DIEPTXF starts at FIFO #1.
625  // Both TXFD and TXSA are in unit of 32-bit words.
626  usb_otg->DIEPTXF[epnum - 1] = (fifo_size << USB_OTG_DIEPTXF_INEPTXFD_Pos) | (EP_FIFO_SIZE/4 - _allocated_fifo_words_tx);
627 
628  in_ep[epnum].DIEPCTL |= (1 << USB_OTG_DIEPCTL_USBAEP_Pos) |
629  (epnum << USB_OTG_DIEPCTL_TXFNUM_Pos) |
630  (desc_edpt->bmAttributes.xfer << USB_OTG_DIEPCTL_EPTYP_Pos) |
631  (desc_edpt->bmAttributes.xfer != TUSB_XFER_ISOCHRONOUS ? USB_OTG_DOEPCTL_SD0PID_SEVNFRM : 0) |
632  (desc_edpt->wMaxPacketSize.size << USB_OTG_DIEPCTL_MPSIZ_Pos);
633 
634  dev->DAINTMSK |= (1 << (USB_OTG_DAINTMSK_IEPM_Pos + epnum));
635  }
636 
637  return true;
638 }

671 static void dcd_edpt_disable (uint8_t rhport, uint8_t ep_addr, bool stall)
672 {
673  (void) rhport;
674 
675  USB_OTG_GlobalTypeDef * usb_otg = GLOBAL_BASE(rhport);
676  USB_OTG_DeviceTypeDef * dev = DEVICE_BASE(rhport);
677  USB_OTG_OUTEndpointTypeDef * out_ep = OUT_EP_BASE(rhport);
678  USB_OTG_INEndpointTypeDef * in_ep = IN_EP_BASE(rhport);
679 
680  uint8_t const epnum = tu_edpt_number(ep_addr);
681  uint8_t const dir = tu_edpt_dir(ep_addr);
682 
683  if(dir == TUSB_DIR_IN) {
684  // Only disable currently enabled non-control endpoint
685  if ( (epnum == 0) || !(in_ep[epnum].DIEPCTL & USB_OTG_DIEPCTL_EPENA) ){
686  in_ep[epnum].DIEPCTL |= USB_OTG_DIEPCTL_SNAK | (stall ? USB_OTG_DIEPCTL_STALL : 0);
687  } else {
688  // Stop transmitting packets and NAK IN xfers.
689  in_ep[epnum].DIEPCTL |= USB_OTG_DIEPCTL_SNAK;
690  while((in_ep[epnum].DIEPINT & USB_OTG_DIEPINT_INEPNE) == 0);
691 
692  // Disable the endpoint.
693  in_ep[epnum].DIEPCTL |= USB_OTG_DIEPCTL_EPDIS | (stall ? USB_OTG_DIEPCTL_STALL : 0);
694  while((in_ep[epnum].DIEPINT & USB_OTG_DIEPINT_EPDISD_Msk) == 0);
695  in_ep[epnum].DIEPINT = USB_OTG_DIEPINT_EPDISD;
696  }
697 
698  // Flush the FIFO, and wait until we have confirmed it cleared.
699  usb_otg->GRSTCTL |= (epnum << USB_OTG_GRSTCTL_TXFNUM_Pos);
700  usb_otg->GRSTCTL |= USB_OTG_GRSTCTL_TXFFLSH;
701  while((usb_otg->GRSTCTL & USB_OTG_GRSTCTL_TXFFLSH_Msk) != 0);
702  } else {
703  // Only disable currently enabled non-control endpoint
704  if ( (epnum == 0) || !(out_ep[epnum].DOEPCTL & USB_OTG_DOEPCTL_EPENA) ){
705  out_ep[epnum].DOEPCTL |= stall ? USB_OTG_DOEPCTL_STALL : 0;
706  } else {
707  // Asserting GONAK is required to STALL an OUT endpoint.
708  // Simpler to use polling here, we don't use the "B"OUTNAKEFF interrupt
709  // anyway, and it can't be cleared by user code. If this while loop never
710  // finishes, we have bigger problems than just the stack.
711  dev->DCTL |= USB_OTG_DCTL_SGONAK;
712  while((usb_otg->GINTSTS & USB_OTG_GINTSTS_BOUTNAKEFF_Msk) == 0);
713 
714  // Ditto here- disable the endpoint.
715  out_ep[epnum].DOEPCTL |= USB_OTG_DOEPCTL_EPDIS | (stall ? USB_OTG_DOEPCTL_STALL : 0);
716  while((out_ep[epnum].DOEPINT & USB_OTG_DOEPINT_EPDISD_Msk) == 0);
717  out_ep[epnum].DOEPINT = USB_OTG_DOEPINT_EPDISD;
718 
719  // Allow other OUT endpoints to keep receiving.
720  dev->DCTL |= USB_OTG_DCTL_CGONAK;
721  }
722  }

790 // Read a single data packet from receive FIFO
791 static void read_fifo_packet(uint8_t rhport, uint8_t * dst, uint16_t len)
792 {
793  (void) rhport;
794 
795  usb_fifo_t rx_fifo = FIFO_BASE(rhport, 0);
796 
797  // Reading full available 32 bit words from fifo
798  uint16_t full_words = len >> 2;
799  for(uint16_t i = 0; i < full_words; i++) {
800  uint32_t tmp = *rx_fifo;
801  dst[0] = tmp & 0x000000FF;
802  dst[1] = (tmp & 0x0000FF00) >> 8;
803  dst[2] = (tmp & 0x00FF0000) >> 16;
804  dst[3] = (tmp & 0xFF000000) >> 24;
805  dst += 4;
806  }
807 
808  // Read the remaining 1-3 bytes from fifo
809  uint8_t bytes_rem = len & 0x03;
810  if(bytes_rem != 0) {
811  uint32_t tmp = *rx_fifo;
812  dst[0] = tmp & 0x000000FF;
813  if(bytes_rem > 1) {
814  dst[1] = (tmp & 0x0000FF00) >> 8;
815  }
816  if(bytes_rem > 2) {
817  dst[2] = (tmp & 0x00FF0000) >> 16;
818  }
819  }
820 }

822 // Write a single data packet to EPIN FIFO
823 static void write_fifo_packet(uint8_t rhport, uint8_t fifo_num, uint8_t * src, uint16_t len)
824 {
825  (void) rhport;
826 
827  usb_fifo_t tx_fifo = FIFO_BASE(rhport, fifo_num);
828 
829  // Pushing full available 32 bit words to fifo
830  uint16_t full_words = len >> 2;
831  for(uint16_t i = 0; i < full_words; i++){
832  *tx_fifo = (src[3] << 24) | (src[2] << 16) | (src[1] << 8) | src[0];
833  src += 4;
834  }
835 
836  // Write the remaining 1-3 bytes into fifo
837  uint8_t bytes_rem = len & 0x03;
838  if(bytes_rem){
839  uint32_t tmp_word = 0;
840  tmp_word |= src[0];
841  if(bytes_rem > 1){
842  tmp_word |= src[1] << 8;
843  }
844  if(bytes_rem > 2){
845  tmp_word |= src[2] << 16;
846  }
847  *tx_fifo = tmp_word;
848  }

851 static void handle_rxflvl_ints(uint8_t rhport, USB_OTG_OUTEndpointTypeDef * out_ep) {
852  USB_OTG_GlobalTypeDef * usb_otg = GLOBAL_BASE(rhport);
853  usb_fifo_t rx_fifo = FIFO_BASE(rhport, 0);
854 
855  // Pop control word off FIFO
856  uint32_t ctl_word = usb_otg->GRXSTSP;
857  uint8_t pktsts = (ctl_word & USB_OTG_GRXSTSP_PKTSTS_Msk) >> USB_OTG_GRXSTSP_PKTSTS_Pos;
858  uint8_t epnum = (ctl_word & USB_OTG_GRXSTSP_EPNUM_Msk) >> USB_OTG_GRXSTSP_EPNUM_Pos;
859  uint16_t bcnt = (ctl_word & USB_OTG_GRXSTSP_BCNT_Msk) >> USB_OTG_GRXSTSP_BCNT_Pos;
860 
861  switch(pktsts) {
862  case 0x01: // Global OUT NAK (Interrupt)
863  break;
864 
865  case 0x02: // Out packet recvd
866  {
867  xfer_ctl_t * xfer = XFER_CTL_BASE(epnum, TUSB_DIR_OUT);
868 
869  // Read packet off RxFIFO
870  read_fifo_packet(rhport, xfer->buffer, bcnt);
871 
872  // Increment pointer to xfer data
873  xfer->buffer += bcnt;
874 
875  // Truncate transfer length in case of short packet
876  if(bcnt < xfer->max_size) {
877  xfer->total_len -= (out_ep[epnum].DOEPTSIZ & USB_OTG_DOEPTSIZ_XFRSIZ_Msk) >> USB_OTG_DOEPTSIZ_XFRSIZ_Pos;
878  if(epnum == 0) {
879  xfer->total_len -= ep0_pending[TUSB_DIR_OUT];
880  ep0_pending[TUSB_DIR_OUT] = 0;
881  }
882  }
883  }
884  break;
885 
886  case 0x03: // Out packet done (Interrupt)
887  break;
888 
889  case 0x04: // Setup packet done (Interrupt)
890  out_ep[epnum].DOEPTSIZ |= (3 << USB_OTG_DOEPTSIZ_STUPCNT_Pos);
891  break;
892 
893  case 0x06: // Setup packet recvd
894  // We can receive up to three setup packets in succession, but
895  // only the last one is valid.
896  _setup_packet[0] = (* rx_fifo);
897  _setup_packet[1] = (* rx_fifo);
898  break;
899 
900  default: // Invalid
901  TU_BREAKPOINT();
902  break;
903  }
904 }

906 static void handle_epout_ints(uint8_t rhport, USB_OTG_DeviceTypeDef * dev, USB_OTG_OUTEndpointTypeDef * out_ep) {
907  // DAINT for a given EP clears when DOEPINTx is cleared.
908  // OEPINT will be cleared when DAINT's out bits are cleared.
909  for(uint8_t n = 0; n < EP_MAX; n++) {
910  xfer_ctl_t * xfer = XFER_CTL_BASE(n, TUSB_DIR_OUT);
911 
912  if(dev->DAINT & (1 << (USB_OTG_DAINT_OEPINT_Pos + n))) {

914  if(out_ep[n].DOEPINT & USB_OTG_DOEPINT_STUP) {
915  out_ep[n].DOEPINT = USB_OTG_DOEPINT_STUP;
916  dcd_event_setup_received(rhport, (uint8_t*) &_setup_packet[0], true);
917  }
918 
919  // OUT XFER complete
920  if(out_ep[n].DOEPINT & USB_OTG_DOEPINT_XFRC) {
921  out_ep[n].DOEPINT = USB_OTG_DOEPINT_XFRC;
922 
923  // EP0 can only handle one packet
924  if((n == 0) && ep0_pending[TUSB_DIR_OUT]) {
925  // Schedule another packet to be received.
926  edpt_schedule_packets(rhport, n, TUSB_DIR_OUT, 1, ep0_pending[TUSB_DIR_OUT]);
927  } else {
928  dcd_event_xfer_complete(rhport, n, xfer->total_len, XFER_RESULT_SUCCESS, true);
929  }
930  }
931  }
932  }

935 static void handle_epin_ints(uint8_t rhport, USB_OTG_DeviceTypeDef * dev, USB_OTG_INEndpointTypeDef * in_ep) {
936  // DAINT for a given EP clears when DIEPINTx is cleared.
937  // IEPINT will be cleared when DAINT's out bits are cleared.
938  for ( uint8_t n = 0; n < EP_MAX; n++ )
939  {
940  xfer_ctl_t *xfer = XFER_CTL_BASE(n, TUSB_DIR_IN);
941 
942  if ( dev->DAINT & (1 << (USB_OTG_DAINT_IEPINT_Pos + n)) )
943  {
944  // IN XFER complete (entire xfer).
945  if ( in_ep[n].DIEPINT & USB_OTG_DIEPINT_XFRC )
946  {
947  in_ep[n].DIEPINT = USB_OTG_DIEPINT_XFRC;
948 
949  // EP0 can only handle one packet
950  if((n == 0) && ep0_pending[TUSB_DIR_IN]) {
951  // Schedule another packet to be transmitted.
952  edpt_schedule_packets(rhport, n, TUSB_DIR_IN, 1, ep0_pending[TUSB_DIR_IN]);
953  } else {
954  dcd_event_xfer_complete(rhport, n | TUSB_DIR_IN_MASK, xfer->total_len, XFER_RESULT_SUCCESS, true);
955  }
956  }
957 
958  // XFER FIFO empty
959  if ( (in_ep[n].DIEPINT & USB_OTG_DIEPINT_TXFE) && (dev->DIEPEMPMSK & (1 << n)) )
960  {
961  // DIEPINT's TXFE bit is read-only, software cannot clear it.
962  // It will only be cleared by hardware when written bytes is more than
963  // - 64 bytes or
964  // - Half of TX FIFO size (configured by DIEPTXF)
965 
966  uint16_t remaining_packets = (in_ep[n].DIEPTSIZ & USB_OTG_DIEPTSIZ_PKTCNT_Msk) >> USB_OTG_DIEPTSIZ_PKTCNT_Pos;
967 
968  // Process every single packet (only whole packets can be written to fifo)
969  for(uint16_t i = 0; i < remaining_packets; i++){
970  uint16_t remaining_bytes = (in_ep[n].DIEPTSIZ & USB_OTG_DIEPTSIZ_XFRSIZ_Msk) >> USB_OTG_DIEPTSIZ_XFRSIZ_Pos;
971  // Packet can not be larger than ep max size
972  uint16_t packet_size = tu_min16(remaining_bytes, xfer->max_size);
973 
974  // It's only possible to write full packets into FIFO. Therefore DTXFSTS register of current
975  // EP has to be checked if the buffer can take another WHOLE packet
976  if(packet_size > ((in_ep[n].DTXFSTS & USB_OTG_DTXFSTS_INEPTFSAV_Msk) << 2)){
977  break;
978  }
979 
980  // Push packet to Tx-FIFO
981  write_fifo_packet(rhport, n, xfer->buffer, packet_size);
982 
983  // Increment pointer to xfer data
984  xfer->buffer += packet_size;
985  }
986 
987  // Turn off TXFE if all bytes are written.
988  if (((in_ep[n].DIEPTSIZ & USB_OTG_DIEPTSIZ_XFRSIZ_Msk) >> USB_OTG_DIEPTSIZ_XFRSIZ_Pos) == 0)
989  {
990  dev->DIEPEMPMSK &= ~(1 << n);
991  }
992  }
993  }
994  }

997 void dcd_int_handler(uint8_t rhport)
998 {
