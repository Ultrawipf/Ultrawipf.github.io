304 bool msch_xfer_cb(uint8_t dev_addr, uint8_t ep_addr, xfer_result_t event, uint32_t xferred_bytes)
305 {
306  msch_interface_t* p_msc = get_itf(dev_addr);
307  msc_cbw_t const * cbw = &p_msc->cbw;
308  msc_csw_t * csw = &p_msc->csw;
309 
310  switch (p_msc->stage)
311  {
312  case MSC_STAGE_CMD:
313  // Must be Command Block
314  TU_ASSERT(ep_addr == p_msc->ep_out && event == XFER_RESULT_SUCCESS && xferred_bytes == sizeof(msc_cbw_t));
315 
316  if ( cbw->total_bytes && p_msc->buffer )
317  {
318  // Data stage if any
319  p_msc->stage = MSC_STAGE_DATA;
320 
321  uint8_t const ep_data = (cbw->dir & TUSB_DIR_IN_MASK) ? p_msc->ep_in : p_msc->ep_out;
322  TU_ASSERT(usbh_edpt_xfer(dev_addr, ep_data, p_msc->buffer, cbw->total_bytes));
323  }else
324  {
325  // Status stage
326  p_msc->stage = MSC_STAGE_STATUS;
327  TU_ASSERT(usbh_edpt_xfer(dev_addr, p_msc->ep_in, (uint8_t*) &p_msc->csw, sizeof(msc_csw_t)));
328  }
329  break;
330 
331  case MSC_STAGE_DATA:
332  // Status stage
333  p_msc->stage = MSC_STAGE_STATUS;
334  TU_ASSERT(usbh_edpt_xfer(dev_addr, p_msc->ep_in, (uint8_t*) &p_msc->csw, sizeof(msc_csw_t)));
335  break;
336 
337  case MSC_STAGE_STATUS:
338  // SCSI op is complete
339  p_msc->stage = MSC_STAGE_IDLE;
340 
341  if (p_msc->complete_cb) p_msc->complete_cb(dev_addr, cbw, csw);
342  break;
343 
344  // unknown state
345  default: break;
346  }
347 
348  return true;
349 }

360 bool msch_open(uint8_t rhport, uint8_t dev_addr, tusb_desc_interface_t const *itf_desc, uint16_t *p_length)
361 {
362  TU_VERIFY (MSC_SUBCLASS_SCSI == itf_desc->bInterfaceSubClass &&
363  MSC_PROTOCOL_BOT == itf_desc->bInterfaceProtocol);
364 
365  msch_interface_t* p_msc = get_itf(dev_addr);
366 
367  //------------- Open Data Pipe -------------//
368  tusb_desc_endpoint_t const * ep_desc = (tusb_desc_endpoint_t const *) tu_desc_next(itf_desc);
369 
370  for(uint32_t i=0; i<2; i++)
371  {
372  TU_ASSERT(TUSB_DESC_ENDPOINT == ep_desc->bDescriptorType && TUSB_XFER_BULK == ep_desc->bmAttributes.xfer);
373  TU_ASSERT(usbh_edpt_open(rhport, dev_addr, ep_desc));
374 
375  if ( tu_edpt_dir(ep_desc->bEndpointAddress) == TUSB_DIR_IN )
376  {
377  p_msc->ep_in = ep_desc->bEndpointAddress;
378  }else
379  {
380  p_msc->ep_out = ep_desc->bEndpointAddress;
381  }
382 
383  ep_desc = (tusb_desc_endpoint_t const *) tu_desc_next(ep_desc);
384  }
385 
386  p_msc->itf_num = itf_desc->bInterfaceNumber;
387  (*p_length) += sizeof(tusb_desc_interface_t) + 2*sizeof(tusb_desc_endpoint_t);
388 
389  return true;
390 }

419 static bool config_get_maxlun_complete (uint8_t dev_addr, tusb_control_request_t const * request, xfer_result_t result)
420 {
421  (void) request;
422 
423  msch_interface_t* p_msc = get_itf(dev_addr);
424 
425  // STALL means zero
426  p_msc->max_lun = (XFER_RESULT_SUCCESS == result) ? _msch_buffer[0] : 0;
427  p_msc->max_lun++; // MAX LUN is minus 1 by specs
428 
429  // TODO multiple LUN support
430  TU_LOG2("SCSI Test Unit Ready\r\n");
431  uint8_t const lun = 0;
432  tuh_msc_test_unit_ready(dev_addr, lun, config_test_unit_ready_complete);
433 
434  return true;
435 }

437 static bool config_test_unit_ready_complete(uint8_t dev_addr, msc_cbw_t const* cbw, msc_csw_t const* csw)
438 {
439  if (csw->status == 0)
440  {
441  // Unit is ready, read its capacity
442  TU_LOG2("SCSI Read Capacity\r\n");
443  tuh_msc_read_capacity(dev_addr, cbw->lun, (scsi_read_capacity10_resp_t*) _msch_buffer, config_read_capacity_complete);
444  }else
445  {
446  // Note: During enumeration, some device fails Test Unit Ready and require a few retries
447  // with Request Sense to start working !!

449  TU_LOG2("SCSI Request Sense\r\n");
450  TU_ASSERT(tuh_msc_request_sense(dev_addr, cbw->lun, _msch_buffer, config_request_sense_complete));
451  }
452 
453  return true;

463 static bool config_read_capacity_complete(uint8_t dev_addr, msc_cbw_t const* cbw, msc_csw_t const* csw)
464 {
465  TU_ASSERT(csw->status == 0);
466 
467  msch_interface_t* p_msc = get_itf(dev_addr);
468 
469  // Capacity response field: Block size and Last LBA are both Big-Endian
470  scsi_read_capacity10_resp_t* resp = (scsi_read_capacity10_resp_t*) _msch_buffer;
471  p_msc->capacity[cbw->lun].block_count = tu_ntohl(resp->last_lba) + 1;
472  p_msc->capacity[cbw->lun].block_size = tu_ntohl(resp->block_size);
473 
474  // Mark enumeration is complete
475  p_msc->mounted = true;
476  tuh_msc_mount_cb(dev_addr);
477 
478  usbh_driver_set_config_complete(dev_addr, p_msc->itf_num);
479 
480  return true;
481 }
