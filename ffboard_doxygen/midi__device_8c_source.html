88 static void _prep_out_transaction (midid_interface_t* p_midi)
89 {
90  uint8_t const rhport = TUD_OPT_RHPORT;
91  uint16_t available = tu_fifo_remaining(&p_midi->rx_ff);
92 
93  // Prepare for incoming data but only allow what we can store in the ring buffer.
94  // TODO Actually we can still carry out the transfer, keeping count of received bytes
95  // and slowly move it to the FIFO when read().
96  // This pre-check reduces endpoint claiming
97  TU_VERIFY(available >= sizeof(p_midi->epout_buf), );
98 
99  // claim endpoint
100  TU_VERIFY(usbd_edpt_claim(rhport, p_midi->ep_out), );
101 
102  // fifo can be changed before endpoint is claimed
103  available = tu_fifo_remaining(&p_midi->rx_ff);
104 
105  if ( available >= sizeof(p_midi->epout_buf) ) {
106  usbd_edpt_xfer(rhport, p_midi->ep_out, p_midi->epout_buf, sizeof(p_midi->epout_buf));
107  }else
108  {
109  // Release endpoint since we don't make any transfer
110  usbd_edpt_release(rhport, p_midi->ep_out);
111  }
112 }

123 uint32_t tud_midi_n_read(uint8_t itf, uint8_t cable_num, void* buffer, uint32_t bufsize)
124 {
125  (void) cable_num;
126  midid_interface_t* midi = &_midid_itf[itf];
127 
128  // Fill empty buffer
129  if (midi->read_buffer_length == 0) {
130  if (!tud_midi_n_receive(itf, midi->read_buffer)) return 0;
131 
132  uint8_t code_index = midi->read_buffer[0] & 0x0f;
133  // We always copy over the first byte.
134  uint8_t count = 1;
135  // Ignore subsequent bytes based on the code.
136  if (code_index != 0x5 && code_index != 0xf) {
137  count = 2;
138  if (code_index != 0x2 && code_index != 0x6 && code_index != 0xc && code_index != 0xd) {
139  count = 3;
140  }
141  }
142 
143  midi->read_buffer_length = count;
144  }
145 
146  uint32_t n = midi->read_buffer_length - midi->read_target_length;
147  if (bufsize < n) n = bufsize;
148 
149  // Skip the header in the buffer
150  memcpy(buffer, midi->read_buffer + 1 + midi->read_target_length, n);
151  midi->read_target_length += n;
152 
153  if (midi->read_target_length == midi->read_buffer_length) {
154  midi->read_buffer_length = 0;
155  midi->read_target_length = 0;
156  }
157 
158  return n;
159 }

185 static uint32_t write_flush(midid_interface_t* midi)
186 {
187  // No data to send
188  if ( !tu_fifo_count(&midi->tx_ff) ) return 0;
189 
190  uint8_t const rhport = TUD_OPT_RHPORT;
191 
192  // skip if previous transfer not complete
193  TU_VERIFY( usbd_edpt_claim(rhport, midi->ep_in), 0 );
194 
195  uint16_t count = tu_fifo_read_n(&midi->tx_ff, midi->epin_buf, CFG_TUD_MIDI_EP_BUFSIZE);
196  if (count > 0)
197  {
198  TU_ASSERT( usbd_edpt_xfer(rhport, midi->ep_in, midi->epin_buf, count), 0 );
199  return count;
200  }else
201  {
202  // Release endpoint since we don't make any transfer
203  usbd_edpt_release(rhport, midi->ep_in);
204  return 0;
205  }
206 }

208 uint32_t tud_midi_n_write(uint8_t itf, uint8_t cable_num, uint8_t const* buffer, uint32_t bufsize)
209 {
210  midid_interface_t* midi = &_midid_itf[itf];
211  if (midi->itf_num == 0) {
212  return 0;
213  }
214 
215  uint32_t i = 0;
216  while (i < bufsize) {
217  uint8_t data = buffer[i];
218  if (midi->write_buffer_length == 0) {
219  uint8_t msg = data >> 4;
220  midi->write_buffer[1] = data;
221  midi->write_buffer_length = 2;
222  // Check to see if we're still in a SysEx transmit.
223  if (midi->write_buffer[0] == 0x4) {
224  if (data == 0xf7) {
225  midi->write_buffer[0] = 0x5;
226  midi->write_target_length = 2;
227  } else {
228  midi->write_target_length = 4;
229  }
230  } else if ((msg >= 0x8 && msg <= 0xB) || msg == 0xE) {
231  midi->write_buffer[0] = cable_num << 4 | msg;
232  midi->write_target_length = 4;
233  } else if (msg == 0xf) {
234  if (data == 0xf0) {
235  midi->write_buffer[0] = 0x4;
236  midi->write_target_length = 4;
237  } else if (data == 0xf1 || data == 0xf3) {
238  midi->write_buffer[0] = 0x2;
239  midi->write_target_length = 3;
240  } else if (data == 0xf2) {
241  midi->write_buffer[0] = 0x3;
242  midi->write_target_length = 4;
243  } else {
244  midi->write_buffer[0] = 0x5;
245  midi->write_target_length = 2;
246  }
247  } else {
248  // Pack individual bytes if we don't support packing them into words.
249  midi->write_buffer[0] = cable_num << 4 | 0xf;
250  midi->write_buffer[2] = 0;
251  midi->write_buffer[3] = 0;
252  midi->write_buffer_length = 2;
253  midi->write_target_length = 2;
254  }
255  } else {
256  midi->write_buffer[midi->write_buffer_length] = data;
257  midi->write_buffer_length += 1;
258  // See if this byte ends a SysEx.
259  if (midi->write_buffer[0] == 0x4 && data == 0xf7) {
260  midi->write_buffer[0] = 0x4 + (midi->write_buffer_length - 1);
261  midi->write_target_length = midi->write_buffer_length;
262  }
263  }
264 
265  if (midi->write_buffer_length == midi->write_target_length) {
266  uint16_t written = tu_fifo_write_n(&midi->tx_ff, midi->write_buffer, 4);
267  if (written < 4) {
268  TU_ASSERT( written == 0 );
269  break;
270  }
271  midi->write_buffer_length = 0;
272  }
273  i++;
274  }
275 
276  write_flush(midi);
277 
278  return i;
279 }

281 bool tud_midi_n_send (uint8_t itf, uint8_t const packet[4])
282 {
283  midid_interface_t* midi = &_midid_itf[itf];
284  if (midi->itf_num == 0) {
285  return 0;
286  }
287 
288  if (tu_fifo_remaining(&midi->tx_ff) < 4)
289  return false;
290 
291  tu_fifo_write_n(&midi->tx_ff, packet, 4);
292  write_flush(midi);
293 
294  return true;
295 }

332 uint16_t midid_open(uint8_t rhport, tusb_desc_interface_t const * desc_itf, uint16_t max_len)
333 {
334  // 1st Interface is Audio Control v1
335  TU_VERIFY(TUSB_CLASS_AUDIO == desc_itf->bInterfaceClass &&
336  AUDIO_SUBCLASS_CONTROL == desc_itf->bInterfaceSubClass &&
337  AUDIO_FUNC_PROTOCOL_CODE_UNDEF == desc_itf->bInterfaceProtocol, 0);
338 
339  uint16_t drv_len = tu_desc_len(desc_itf);
340  uint8_t const * p_desc = tu_desc_next(desc_itf);
341 
342  // Skip Class Specific descriptors
343  while ( TUSB_DESC_CS_INTERFACE == tu_desc_type(p_desc) && drv_len <= max_len )
344  {
345  drv_len += tu_desc_len(p_desc);
346  p_desc = tu_desc_next(p_desc);
347  }
348 
349  // 2nd Interface is MIDI Streaming
350  TU_VERIFY(TUSB_DESC_INTERFACE == tu_desc_type(p_desc), 0);
351  tusb_desc_interface_t const * desc_midi = (tusb_desc_interface_t const *) p_desc;
352 
353  TU_VERIFY(TUSB_CLASS_AUDIO == desc_midi->bInterfaceClass &&
354  AUDIO_SUBCLASS_MIDI_STREAMING == desc_midi->bInterfaceSubClass &&
355  AUDIO_FUNC_PROTOCOL_CODE_UNDEF == desc_midi->bInterfaceProtocol, 0);
356 
357  // Find available interface
358  midid_interface_t * p_midi = NULL;
359  for(uint8_t i=0; i<CFG_TUD_MIDI; i++)
360  {
361  if ( _midid_itf[i].ep_in == 0 && _midid_itf[i].ep_out == 0 )
362  {
363  p_midi = &_midid_itf[i];
364  break;
365  }
366  }
367 
368  p_midi->itf_num = desc_midi->bInterfaceNumber;
369 
370  // next descriptor
371  drv_len += tu_desc_len(p_desc);
372  p_desc = tu_desc_next(p_desc);
373 
374  // Find and open endpoint descriptors
375  uint8_t found_endpoints = 0;
376  while ( (found_endpoints < desc_midi->bNumEndpoints) && (drv_len <= max_len) )
377  {
378  if ( TUSB_DESC_ENDPOINT == tu_desc_type(p_desc) )
379  {
380  TU_ASSERT(usbd_edpt_open(rhport, (tusb_desc_endpoint_t const *) p_desc), 0);
381  uint8_t ep_addr = ((tusb_desc_endpoint_t const *) p_desc)->bEndpointAddress;
382 
383  if (tu_edpt_dir(ep_addr) == TUSB_DIR_IN)
384  {
385  p_midi->ep_in = ep_addr;
386  } else {
387  p_midi->ep_out = ep_addr;
388  }
389 
390  // Class Specific MIDI Stream endpoint descriptor
391  drv_len += tu_desc_len(p_desc);
392  p_desc = tu_desc_next(p_desc);
393 
394  found_endpoints += 1;
395  }
396 
397  drv_len += tu_desc_len(p_desc);
398  p_desc = tu_desc_next(p_desc);
399  }
400 
401  // Prepare for incoming data
402  _prep_out_transaction(p_midi);
403 
404  return drv_len;
405 }

420 bool midid_xfer_cb(uint8_t rhport, uint8_t ep_addr, xfer_result_t result, uint32_t xferred_bytes)
421 {
422  (void) result;
423  (void) rhport;
424 
425  uint8_t itf;
426  midid_interface_t* p_midi;
427 
428  // Identify which interface to use
429  for (itf = 0; itf < CFG_TUD_MIDI; itf++)
430  {
431  p_midi = &_midid_itf[itf];
432  if ( ( ep_addr == p_midi->ep_out ) || ( ep_addr == p_midi->ep_in ) ) break;
433  }
434  TU_ASSERT(itf < CFG_TUD_MIDI);
435 
436  // receive new data
437  if ( ep_addr == p_midi->ep_out )
438  {
439  tu_fifo_write_n(&p_midi->rx_ff, p_midi->epout_buf, xferred_bytes);
440 
441  // invoke receive callback if available
442  if (tud_midi_rx_cb) tud_midi_rx_cb(itf);
443 
444  // prepare for next
445  // TODO for now ep_out is not used by public API therefore there is no race condition,
446  // and does not need to claim like ep_in
447  _prep_out_transaction(p_midi);
448  }
449  else if ( ep_addr == p_midi->ep_in )
450  {
451  if (0 == write_flush(p_midi))
452  {
453  // If there is no data left, a ZLP should be sent if
454  // xferred_bytes is multiple of EP size and not zero
455  if ( !tu_fifo_count(&p_midi->tx_ff) && xferred_bytes && (0 == (xferred_bytes % CFG_TUD_MIDI_EP_BUFSIZE)) )
456  {
457  if ( usbd_edpt_claim(rhport, p_midi->ep_in) )
458  {
459  usbd_edpt_xfer(rhport, p_midi->ep_in, NULL, 0);
460  }
461  }
462  }
463  }
464 
465  return true;
466 }
