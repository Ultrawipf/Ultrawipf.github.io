57 /*
58 If the metric is less than CP Offset - Dead Band, then the resulting force is given by the following formula:
59  force = Negative Coefficient * (q - (CP Offset – Dead Band))
60 Similarly, if the metric is greater than CP Offset + Dead Band, then the resulting force is given by the
61 following formula:
62  force = Positive Coefficient * (q - (CP Offset + Dead Band))
63 A spring condition uses axis position as the metric.
64 A damper condition uses axis velocity as the metric.
65 An inertia condition uses axis acceleration as the metric.
66 
67 If the number of Condition report blocks is equal to the number of axes for the effect, then the first report
68 block applies to the first axis, the second applies to the second axis, and so on. For example, a two-axis
69 spring condition with CP Offset set to zero in both Condition report blocks would have the same effect as
70 the joystick self-centering spring. When a condition is defined for each axis in this way, the effect must
71 not be rotated.
72 
73 If there is a single Condition report block for an effect with more than one axis, then the direction along
74 which the parameters of the Condition report block are in effect is determined by the direction parameters
75 passed in the Direction field of the Effect report block. For example, a friction condition rotated 45
76 degrees (in polar coordinates) would resist joystick motion in the northeast-southwest direction but would
77 have no effect on joystick motion in the northwest-southeast direction.
78  */

85 void EffectsCalculator::calculateEffects(std::vector<std::unique_ptr<Axis>> &axes)
86 {
87  for (auto &axis : axes) {
88  axis->calculateAxisEffects(isActive());
89  }
90 
91  if(!isActive()){
92  return;
93  }
94 
95  int32_t forceX = 0;
96  int32_t forceY = 0;
97  int32_t forceVector = 0;
98  uint8_t axisCount = axes.size();
99  bool validY = axisCount > 1;
100 #if MAX_AXIS == 3
101  int32_t forceZ = 0;
102  bool validZ = axisCount > 2;
103 #endif
104 
105  for (uint8_t i = 0; i < MAX_EFFECTS; i++)
106  {
107  FFB_Effect *effect = &effects[i];
108 
109  // Effect activated and not infinite
110  if (effect->state != EFFECT_STATE_INACTIVE && effect->duration != FFB_EFFECT_DURATION_INFINITE){
111  // Start delay not yet reached
112  if(HAL_GetTick() < effect->startTime){
113  continue;
114  }
115  // If effect has expired make inactive
116  if (HAL_GetTick() > effect->startTime + effect->duration)
117  {
118  effect->state = EFFECT_STATE_INACTIVE;
119  }
120  }
121 
122  // Filter out inactive effects
123  if (effect->state == EFFECT_STATE_INACTIVE)
124  {
125  continue;
126  }
127 
128 
129  if (effect->conditionsCount == 0) {
130  forceVector = calcNonConditionEffectForce(effect);
131  }
132 
133  if (effect->enableAxis == DIRECTION_ENABLE || (effect->enableAxis & X_AXIS_ENABLE))
134  {
135  forceX += calcComponentForce(effect, forceVector, axes[0]->getMetrics(), 0, axisCount);
136  forceX = clip<int32_t, int32_t>(forceX, -0x7fff, 0x7fff); // Clip
137  }
138  if (validY && ((effect->enableAxis == DIRECTION_ENABLE) || (effect->enableAxis & Y_AXIS_ENABLE)))
139  {
140  forceY += calcComponentForce(effect, forceVector, axes[1]->getMetrics(), 1, axisCount);
141  forceY = clip<int32_t, int32_t>(forceY, -0x7fff, 0x7fff); // Clip
142  }
143 
144  }
145 
146  axes[0]->setEffectTorque(forceX);
147  if (validY)
148  {
149  axes[1]->setEffectTorque(forceY);
150  }
151 }

153 int32_t EffectsCalculator::calcNonConditionEffectForce(FFB_Effect *effect) {
154  int32_t force_vector = 0;
155  switch (effect->type){
156 
157  case FFB_EFFECT_CONSTANT:
158  { // Constant force is just the force
159  force_vector = ((int32_t)effect->magnitude * (int32_t)(1 + effect->gain)) >> 8;
160  // Optional filtering to reduce spikes
161  if (cfFilter_f < calcfrequency / 2)
162  {
163  force_vector = effect->filter[0]->process(force_vector);
164  }
165  break;
166  }
167 
168  case FFB_EFFECT_RAMP:
169  {
170  uint32_t elapsed_time = HAL_GetTick() - effect->startTime;
171  int32_t duration = effect->duration;
172  float force = (int32_t)effect->startLevel + ((int32_t)elapsed_time * (effect->endLevel - effect->startLevel)) / duration;
173  force_vector = (int32_t)(force * (1 + effect->gain)) >> 8;
174  break;
175  }
176 
177  case FFB_EFFECT_SQUARE:
178  {
179  uint32_t elapsed_time = HAL_GetTick() - effect->startTime;
180  int32_t force = ((elapsed_time + effect->phase) % ((uint32_t)effect->period + 2)) < (uint32_t)(effect->period + 2) / 2 ? -effect->magnitude : effect->magnitude;
181  force_vector = force + effect->offset;
182  break;
183  }
184 
185  case FFB_EFFECT_TRIANGLE:
186  {
187  int32_t force = 0;
188  int32_t offset = effect->offset;
189  int32_t magnitude = effect->magnitude;
190  uint32_t elapsed_time = HAL_GetTick() - effect->startTime;
191  uint32_t phase = effect->phase;
192  uint32_t period = effect->period;
193  float periodF = period;
194 
195  int32_t maxMagnitude = offset + magnitude;
196  int32_t minMagnitude = offset - magnitude;
197  uint32_t phasetime = (phase * period) / 35999;
198  uint32_t timeTemp = elapsed_time + phasetime;
199  float remainder = timeTemp % period;
200  float slope = ((maxMagnitude - minMagnitude) * 2) / periodF;
201  if (remainder > (periodF / 2))
202  force = slope * (periodF - remainder);
203  else
204  force = slope * remainder;
205  force += minMagnitude;
206  force_vector = force;
207  break;
208  }
209 
210  case FFB_EFFECT_SAWTOOTHUP:
211  {
212  float offset = effect->offset;
213  float magnitude = effect->magnitude;
214  uint32_t elapsed_time = HAL_GetTick() - effect->startTime;
215  uint32_t phase = effect->phase;
216  uint32_t period = effect->period;
217  float periodF = effect->period;
218 
219  float maxMagnitude = offset + magnitude;
220  float minMagnitude = offset - magnitude;
221  int32_t phasetime = (phase * period) / 35999;
222  uint32_t timeTemp = elapsed_time + phasetime;
223  float remainder = timeTemp % period;
224  float slope = (maxMagnitude - minMagnitude) / periodF;
225  force_vector = (int32_t)(minMagnitude + slope * (period - remainder));
226  break;
227  }
228 
229  case FFB_EFFECT_SAWTOOTHDOWN:
230  {
231  float offset = effect->offset;
232  float magnitude = effect->magnitude;
233  uint32_t elapsed_time = HAL_GetTick() - effect->startTime;
234  float phase = effect->phase;
235  uint32_t period = effect->period;
236  float periodF = effect->period;
237 
238  float maxMagnitude = offset + magnitude;
239  float minMagnitude = offset - magnitude;
240  int32_t phasetime = (phase * period) / 35999;
241  uint32_t timeTemp = elapsed_time + phasetime;
242  float remainder = timeTemp % period;
243  float slope = (maxMagnitude - minMagnitude) / periodF;
244  force_vector = (int32_t)(minMagnitude + slope * (remainder)); // reverse time
245  break;
246  }
247 
248  case FFB_EFFECT_SINE:
249  {
250  uint32_t t = HAL_GetTick() - effect->startTime;
251  float freq = 1.0f / (float)(std::max<uint16_t>(effect->period, 2));
252  float phase = (float)effect->phase / (float)35999; //degrees
253  float sine = sinf(2.0 * (float)M_PI * (t * freq + phase)) * effect->magnitude;
254  force_vector = (int32_t)(effect->offset + sine);
255  break;
256  }
257  default:
258  break;
259  }
260  if(effect->useEnvelope) {
261  force_vector = applyEnvelope(effect, (int32_t)force_vector);
262  }
263  return force_vector;
264 }

282 int32_t EffectsCalculator::calcComponentForce(FFB_Effect *effect, int32_t forceVector, metric_t *metrics, uint8_t axis, uint8_t axisCount)
283 {
284  int32_t result_torque = 0;
285  uint16_t direction;
286  uint8_t con_idx = 0; // condition block index
287 
288  if (effect->enableAxis == DIRECTION_ENABLE)
289  {
290  direction = effect->directionX;
291  if (effect->conditionsCount > 1)
292  {
293  con_idx = axis;
294  }
295  }
296  else
297  {
298  direction = axis == 0 ? effect->directionX : effect->directionY;
299  con_idx = axis;
300  }
301 
302  //bool useForceDirectionForConditionEffect = (effect->enableAxis == DIRECTION_ENABLE && axisCount > 1 && effect->conditionsCount == 1);
303  bool rotateConditionForce = (axisCount > 1 && effect->conditionsCount < axisCount);
304  float angle = ((float)direction * (2*M_PI) / 36000.0);
305  float angle_ratio = axis == 0 ? sin(angle) : -1 * cos(angle);
306  angle_ratio = rotateConditionForce ? angle_ratio : 1.0;
307 
308  switch (effect->type)
309  {
310  case FFB_EFFECT_CONSTANT:
311  case FFB_EFFECT_RAMP:
312  case FFB_EFFECT_SQUARE:
313  case FFB_EFFECT_TRIANGLE:
314  case FFB_EFFECT_SAWTOOTHUP:
315  case FFB_EFFECT_SAWTOOTHDOWN:
316  case FFB_EFFECT_SINE:
317  {
318  result_torque = -forceVector * angle_ratio;
319  break;
320  }
321 
322  case FFB_EFFECT_SPRING:
323  {
324  float metric = metrics->pos;
325  result_torque -= calcConditionEffectForce(effect, metric, gain.spring,
326  con_idx, 0.0004f, angle_ratio);
327  break;
328  }
329 
330  case FFB_EFFECT_FRICTION:
331  {
332 // effect->conditions[con_idx].negativeSaturation = FRICTION_SATURATION;
333 // effect->conditions[con_idx].positiveSaturation = FRICTION_SATURATION;
334  float metric = effect->filter[con_idx]->process(metrics->speed) * .25;
335  result_torque -= calcConditionEffectForce(effect, metric, gain.friction,
336  con_idx, .08f, angle_ratio);
337  break;
338  }
339  case FFB_EFFECT_DAMPER:
340  {
341  float metric = effect->filter[con_idx]->process(metrics->speed) * .0625f;
342  result_torque -= calcConditionEffectForce(effect, metric, gain.damper,
343  con_idx, 0.6f, angle_ratio);
344  break;
345  }
346 
347  case FFB_EFFECT_INERTIA:
348  {
349 // effect->conditions[con_idx].negativeSaturation = INERTIA_SATURATION;
350 // effect->conditions[con_idx].positiveSaturation = INERTIA_SATURATION;
351  float metric = effect->filter[con_idx]->process(metrics->accel*4);
352  result_torque -= calcConditionEffectForce(effect, metric, gain.inertia,
353  con_idx, 0.5f, angle_ratio);
354  break;
355  }
356 
357  default:
358  // Unsupported effect
359  break;
360  }
361  return (result_torque * (global_gain+1)) >> 8; // Apply global gain
362 }

364 int32_t EffectsCalculator::calcConditionEffectForce(FFB_Effect *effect, float metric, uint8_t gain,
365  uint8_t idx, float scale, float angle_ratio)
366 {
367  int16_t offset = effect->conditions[idx].cpOffset;
368  int16_t deadBand = effect->conditions[idx].deadBand;
369  int32_t force = 0;
370  if (abs(metric - offset) > deadBand)
371  {
372  if (metric < offset)
373  { // Deadband side
374  force = clip<int32_t, int32_t>(((float)effect->conditions[idx].negativeCoefficient *
375  scale * (float)(metric - (offset - deadBand))),
376  -effect->conditions[idx].negativeSaturation,
377  effect->conditions[idx].positiveSaturation);
378  }
379  else
380  {
381  force = clip<int32_t, int32_t>(((float)effect->conditions[idx].positiveCoefficient *
382  scale * (float)(metric - (offset + deadBand))),
383  -effect->conditions[idx].negativeSaturation,
384  effect->conditions[idx].positiveSaturation);
385  }
386  }
387  force = ((gain+1) * force) >> 8;
388  return force * angle_ratio;
389 }

392 int32_t EffectsCalculator::applyEnvelope(FFB_Effect *effect, int32_t value)
393 {
394  int32_t magnitude = (effect->magnitude);
395  int32_t attackLevel = (effect->attackLevel);
396  int32_t fadeLevel = (effect->fadeLevel);
397  int32_t newValue = magnitude;
398  uint32_t elapsed_time = HAL_GetTick() - effect->startTime;
399  if (elapsed_time < effect->attackTime)
400  {
401  newValue = (magnitude - attackLevel) * elapsed_time;
402  newValue /= (int32_t)effect->attackTime;
403  newValue += attackLevel;
404  }
405  if (effect->duration != FFB_EFFECT_DURATION_INFINITE &&
406  elapsed_time > (effect->duration - effect->fadeTime))
407  {
408  newValue = (magnitude - fadeLevel) * (effect->duration - elapsed_time);
409  newValue /= (int32_t)effect->fadeTime;
410  newValue += fadeLevel;
411  }
412 
413  newValue *= value;
414  newValue /= 0x7fff; // 16 bit
415  return newValue;
416 }
