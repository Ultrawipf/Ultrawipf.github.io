84 void MidiMain::play(){
85  // Take only first channel for now and last note...
86  uint8_t chan = 0;
87  if(notes[chan].empty()){
88  if(active[chan]){
89  active[chan] = false;
90  // stop driver
91  drv->setUdUq(0, 0);
92  }
93  }else{
94  if(!active[chan]){
95  drv->setUdUq(this->power,0);
96  active[chan] = true;
97  }
98 
99  MidiNote* note = &notes[chan].back();
100  float freq = noteToFreq[note->note];
101  // Speed up period counter instead of changing frequency to prevent phase jumps
102  float time = periodf;//(abs((uint16_t)this->timer_update->Instance->CNT - lastSystick)%period) / 1000000.0;
103  //lastSystick = this->timer_update->Instance->CNT;
104  note->counter += time * note->pitchbend;
105 
106  float volume = note->volume / 127.0f;
107  float p = (note->counter*freq);
108  float sine = sinf(M_PI*p) * volume;
109  int16_t val = (sine * movementrange); // 180° phase range
110 
111  drv->setPhiE_ext(val); // Wobble motor
112  }
113 }
