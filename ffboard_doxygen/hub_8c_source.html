130 bool hub_open(uint8_t rhport, uint8_t dev_addr, tusb_desc_interface_t const *itf_desc, uint16_t *p_length)
131 {
132  // not support multiple TT yet
133  if ( itf_desc->bInterfaceProtocol > 1 ) return false;
134 
135  //------------- Open Interrupt Status Pipe -------------//
136  tusb_desc_endpoint_t const *ep_desc;
137  ep_desc = (tusb_desc_endpoint_t const *) tu_desc_next(itf_desc);
138 
139  TU_ASSERT(TUSB_DESC_ENDPOINT == ep_desc->bDescriptorType);
140  TU_ASSERT(TUSB_XFER_INTERRUPT == ep_desc->bmAttributes.xfer);
141 
142  TU_ASSERT(usbh_edpt_open(rhport, dev_addr, ep_desc));
143 
144  hub_data[dev_addr-1].itf_num = itf_desc->bInterfaceNumber;
145  hub_data[dev_addr-1].ep_in = ep_desc->bEndpointAddress;
146 
147  (*p_length) = sizeof(tusb_desc_interface_t) + sizeof(tusb_desc_endpoint_t);
148 
149  return true;
150 }

155 static bool config_get_hub_desc_complete (uint8_t dev_addr, tusb_control_request_t const * request, xfer_result_t result)
156 {
157  (void) request;
158  TU_ASSERT(XFER_RESULT_SUCCESS == result);
159 
160  usbh_hub_t* p_hub = &hub_data[dev_addr-1];
161 
162  // only use number of ports in hub descriptor
163  descriptor_hub_desc_t const* desc_hub = (descriptor_hub_desc_t const*) _hub_buffer;
164  p_hub->port_count = desc_hub->bNbrPorts;
165 
166  // May need to GET_STATUS
167 
168  // Ports must be powered on to be able to detect connection
169  tusb_control_request_t const new_request =
170  {
171  .bmRequestType_bit =
172  {
173  .recipient = TUSB_REQ_RCPT_OTHER,
174  .type = TUSB_REQ_TYPE_CLASS,
175  .direction = TUSB_DIR_OUT
176  },
177  .bRequest = HUB_REQUEST_SET_FEATURE,
178  .wValue = HUB_FEATURE_PORT_POWER,
179  .wIndex = 1, // starting with port 1
180  .wLength = 0
181  };
182 
183  TU_ASSERT( tuh_control_xfer(dev_addr, &new_request, NULL, config_port_power_complete) );
184 
185  return true;

188 static bool config_port_power_complete (uint8_t dev_addr, tusb_control_request_t const * request, xfer_result_t result)
189 {
190  TU_ASSERT(XFER_RESULT_SUCCESS == result);
191  usbh_hub_t* p_hub = &hub_data[dev_addr-1];
192 
193  if (request->wIndex == p_hub->port_count)
194  {
195  // All ports are power -> queue notification status endpoint and
196  // complete the SET CONFIGURATION
197  TU_ASSERT( usbh_edpt_xfer(dev_addr, p_hub->ep_in, &p_hub->status_change, 1) );
198 
199  usbh_driver_set_config_complete(dev_addr, p_hub->itf_num);
200  }else
201  {
202  tusb_control_request_t new_request = *request;
203  new_request.wIndex++; // power next port
204 
205  TU_ASSERT( tuh_control_xfer(dev_addr, &new_request, NULL, config_port_power_complete) );
206  }
207 
208  return true;
209 }

295 static bool connection_get_status_complete (uint8_t dev_addr, tusb_control_request_t const * request, xfer_result_t result)

297  TU_ASSERT(result == XFER_RESULT_SUCCESS);

299  uint8_t const port_num = (uint8_t) request->wIndex;
300 
301  // Connection change

304  // Port is powered and enabled
305  //TU_VERIFY(port_status.status_current.port_power && port_status.status_current.port_enable, );
306 
307  // Acknowledge Port Connection Change
308  hub_port_clear_feature(dev_addr, port_num, HUB_FEATURE_PORT_CONNECTION_CHANGE, connection_clear_conn_change_complete);
309  }else
310  {
311  // Other changes are: Enable, Suspend, Over Current, Reset, L1 state
312  // TODO clear change
313 
314  // prepare for next hub status
315  // TODO continue with status_change, or maybe we can do it again with status
316  hub_status_pipe_queue(dev_addr);
317  }
318 
319  return true;

324 bool hub_xfer_cb(uint8_t dev_addr, uint8_t ep_addr, xfer_result_t result, uint32_t xferred_bytes)
325 {
326  (void) xferred_bytes; // TODO can be more than 1 for hub with lots of ports
327  (void) ep_addr;
328  TU_ASSERT( result == XFER_RESULT_SUCCESS);
329 
330  usbh_hub_t * p_hub = &hub_data[dev_addr-1];
331 
332  TU_LOG2("Port Status Change = 0x%02X\r\n", p_hub->status_change);
333  for (uint8_t port=1; port <= p_hub->port_count; port++)
334  {
335  // TODO HUB ignore bit0 hub_status_change
336  if ( tu_bit_test(p_hub->status_change, port) )
337  {
338  hub_port_get_status(dev_addr, port, &p_hub->port_status, connection_get_status_complete);
339  break;
340  }
341  }
342 
343  // NOTE: next status transfer is queued by usbh.c after handling this request
344 
345  return true;
346 }
