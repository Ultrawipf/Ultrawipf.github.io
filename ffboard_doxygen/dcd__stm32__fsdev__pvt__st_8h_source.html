185 static inline void pcd_set_ep_cnt_rx_reg(__O uint16_t * pdwReg, size_t wCount) {
186  uint32_t wNBlocks;
187  if(wCount > 62u)
188  {
189  wNBlocks = wCount >> 5u;
190  if((wCount & 0x1fU) == 0u)
191  {
192  wNBlocks--;
193  }
194  wNBlocks = wNBlocks << 10u;
195  wNBlocks |= 0x8000u; // Mark block size as 32byte
196  *pdwReg = (uint16_t)wNBlocks;
197  }
198  else
199  {
200  wNBlocks = wCount >> 1u;
201  if((wCount & 0x1U) != 0u)
202  {
203  wNBlocks++;
204  }
205  *pdwReg = (uint16_t)((wNBlocks) << 10u);
206  }
207 }

271 static inline void pcd_set_ep_tx_status(USB_TypeDef * USBx, uint32_t bEpNum, uint32_t wState)
272 {
273  uint32_t regVal = pcd_get_endpoint(USBx, bEpNum);
274  regVal &= USB_EPTX_DTOGMASK;
275 
276  /* toggle first bit ? */

281  /* toggle second bit ? */
