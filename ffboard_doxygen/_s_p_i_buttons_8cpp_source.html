85  this->conf = config;
86

89  if(newPin != nullptr){
90  this->spiConfig.cs = *newPin;
91 
92  }
93  spiPort.reserveCsPin(this->spiConfig.cs);
94  // Setup presets
95  if(conf.mode == SPI_BtnMode::TM){
96  this->spiConfig.cspol = true;
97  this->conf.cutRight = true;
98  this->spiConfig.peripheral.CLKPolarity = SPI_POLARITY_LOW;
99 
100  }else if(conf.mode == SPI_BtnMode::PISOSR){
101  this->spiConfig.cspol = false;
102  this->conf.cutRight = false;
103  this->spiConfig.peripheral.CLKPhase = SPI_PHASE_1EDGE;
104  this->spiConfig.peripheral.CLKPolarity = SPI_POLARITY_LOW;
105  }
106  spiPort.takeSemaphore();
107  spiPort.configurePort(&this->spiConfig.peripheral);
108  spiPort.giveSemaphore();
109 
110  mask = pow(2,config.numButtons)-1;
111  offset = 8 - (config.numButtons % 8);
112 
113  // Thrustmaster uses extra bits for IDs
114  if(config.mode == SPI_BtnMode::TM){
115  bytes = 1+((config.numButtons+2)/8);
116  }else{
117  bytes = 1+((config.numButtons-1)/8);
118  }

236 static ButtonSourceConfig decodeIntToConf(uint16_t config_int, uint16_t config_int_2){
237  ButtonSourceConfig c;
238  c.numButtons = config_int & 0x3F;
239  c.invert = (config_int >> 6) & 0x1;
240  c.cutRight = (config_int >> 7) & 0x1;
241  c.mode = SPI_BtnMode(config_int >> 8);
242  c.cs_num = static_cast<uint8_t>(config_int_2 & 0xF); // Leaving space for other use.
243  return c;
244 }
245 static std::tuple<uint16_t, uint16_t> encodeConfToInt(ButtonSourceConfig* c){
246  uint16_t val = c->numButtons & 0x3F;
247  val |= c->invert << 6;

249  val |= (uint8_t)c->mode << 8;
250  return { val, c->cs_num & 0xF };
