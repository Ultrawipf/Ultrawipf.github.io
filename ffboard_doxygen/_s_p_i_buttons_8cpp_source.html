92 void SPI_Buttons::setConfig(ButtonSourceConfig config){
93  config.numButtons = std::min<uint8_t>(this->maxButtons, config.numButtons);
94  this->conf = config;
95 
96  spiPort.freeCsPin(this->spiConfig.cs);
97  OutputPin* newPin = spiPort.getCsPin(config.cs_num-1); // TODO update internal pin number if requested pin is blocked
98  if(newPin != nullptr){
99  this->spiConfig.cs = *newPin;
100 
101  }
102  spiPort.reserveCsPin(this->spiConfig.cs);
103  // Setup presets
104  if(conf.mode == SPI_BtnMode::TM){
105  this->spiConfig.cspol = true;
106  this->conf.cutRight = true;
107  this->spiConfig.peripheral.CLKPolarity = SPI_POLARITY_LOW;
108 
109  }else if(conf.mode == SPI_BtnMode::PISOSR){
110  this->spiConfig.cspol = false;
111  this->conf.cutRight = false;
112  this->spiConfig.peripheral.CLKPhase = SPI_PHASE_1EDGE;
113  this->spiConfig.peripheral.CLKPolarity = SPI_POLARITY_LOW;
114  }
115  spiPort.takeSemaphore();
116  spiPort.configurePort(&this->spiConfig.peripheral);
117  spiPort.giveSemaphore();
118 
119  mask = pow(2,config.numButtons)-1;
120  offset = 8 - (config.numButtons % 8);
121 
122  // Thrustmaster uses extra bits for IDs
123  if(config.mode == SPI_BtnMode::TM){
124  bytes = 1+((config.numButtons+2)/8);
125  }else{
126  bytes = 1+((config.numButtons-1)/8);
127  }
128 
129  this->btnnum = config.numButtons;
130 }

203  this->setConfig(*c);
204  }else if(cmd.type == CMDtype::get){
205  replies.push_back(CommandReply(this->getConfig()->invert ? 1 : 0));
206  }else{
207  return CommandStatus::ERR;

209  break;
210  case SPIButtons_commands::btncut:
211  if(cmd.type == CMDtype::set){
212  ButtonSourceConfig* c = this->getConfig();
213  c->cutRight = cmd.val != 0;
214  this->setConfig(*c);
215  }else if(cmd.type == CMDtype::get){
216  replies.push_back(CommandReply(this->getConfig()->cutRight ? 1 : 0));
217  }else{
218  return CommandStatus::ERR;
219  }
220  break;
221  case SPIButtons_commands::mode:
222  if(cmd.type == CMDtype::set){
223  setMode((SPI_BtnMode)cmd.val);
224  }else if(cmd.type == CMDtype::get){
225  replies.push_back(CommandReply((uint8_t)this->conf.mode));

235  setConfig(this->conf);

237  break;
238

241  }

246 static ButtonSourceConfig decodeIntToConf(uint16_t config_int, uint16_t config_int_2){
247  ButtonSourceConfig c;
248  c.numButtons = config_int & 0x3F;
249  c.invert = (config_int >> 6) & 0x1;
250  c.cutRight = (config_int >> 7) & 0x1;
251  c.mode = SPI_BtnMode(config_int >> 8);
252  c.cs_num = static_cast<uint8_t>(config_int_2 & 0xF); // Leaving space for other use.
253  return c;
254 }
255 static std::tuple<uint16_t, uint16_t> encodeConfToInt(ButtonSourceConfig* c){
256  uint16_t val = c->numButtons & 0x3F;
257  val |= c->invert << 6;
258  val |= c->cutRight << 7;
259  val |= (uint8_t)c->mode << 8;
260  return { val, c->cs_num & 0xF };
261 }
