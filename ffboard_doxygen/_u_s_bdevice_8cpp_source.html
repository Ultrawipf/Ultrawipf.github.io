61 uint16_t* USBdevice::getUsbStringDesc(uint8_t index,uint16_t langid){
62  (void) langid;
63  uint16_t chr_count = 0;
64  if (index == 0) // Language
65  {
66  _desc_str[1] = string_desc->langId;
67  chr_count = 1;
68  }else{
69  std::string field;
70  if(index == desc_device->iSerialNumber){
71  field = getUsbSerial();
72  }else if(index == desc_device->iProduct){
73  field = string_desc->product;
74  }else if(index == desc_device->iManufacturer){
75  field = string_desc->manufacturer;
76  }else if(index > 3 && ((index - 4) < (int)string_desc->interfaces.size())){
77  field = string_desc->interfaces[index-4];
78  }else{
79  return NULL;
80  }
81  chr_count = std::min<uint8_t>(USB_STRING_DESC_BUF_SIZE-1,field.length());
82  // Convert ASCII string into UTF-16
83  for(uint8_t i=0; i < chr_count; i++)
84  {
85  _desc_str[1+i] = field[i];
86  }
87  }
88 
89  // first byte is length (including header), second byte is string type
90  _desc_str[0] = (TUSB_DESC_STRING << 8 ) | (2*chr_count + 2);
91
