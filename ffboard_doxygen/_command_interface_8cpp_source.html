91 void StringCommandInterface::formatReply(std::string& reply,const std::vector<CommandResult>& results,const bool formatWriteAsRead){
92  //uint16_t lastId = 0xFFFF;
93  for(const CommandResult& result : results){ // All commands processed this batch
94  if(formatWriteAsRead && !(result.originalCommand.type == CMDtype::set || result.originalCommand.type == CMDtype::setat)){
95  return; // Ignore commands that should be formatted as read if they are not set commands
96  }
97 
98  reply += "["; // Start marker
99  reply += StringCommandInterface::formatOriginalCommandFromResult(result.originalCommand,result.commandHandler, formatWriteAsRead);
100  reply += "|"; // Separator
101 
102  if(formatWriteAsRead){
103  std::string tstr;
104  StringCommandInterface::generateReplyFromCmd(tstr,result.originalCommand);
105  reply += tstr;
106  }else{
107  if(result.type == CommandStatus::NOT_FOUND){
108  reply += "NOT_FOUND";
109  //ErrorHandler::addError(CommandInterface::cmdNotFoundError);
110  }else if(result.type == CommandStatus::OK || result.type == CommandStatus::BROADCAST){
111  if(result.reply.empty()){
112  reply += "OK";
113  }else{
114  uint16_t repliesRemaining = result.reply.size();
115  for(const CommandReply& cmdReply : result.reply){ // For all entries of this command. Normally just one
116  std::string tstr;
117  //TODO Long replies fail HERE while copying reply strings
118  StringCommandInterface::generateReplyValueString(tstr,cmdReply);
119  reply += tstr;
120  if(--repliesRemaining > 0){
121  reply += "\n"; // Separate replies with newlines
122  }
123  }
124  }
125  }else if(result.type == CommandStatus::ERR){
126  reply += "ERR";
127  }else{
128  reply += "None";
129  }
130  }
131 
132  reply += "]\n"; // end marker
133  }

139 std::string StringCommandInterface::formatOriginalCommandFromResult(const ParsedCommand& originalCommand,CommandHandler* commandHandler,const bool formatWriteAsRead){
140

142  std::string cmdstring = commandHandler->getCommandHandlerInfo()->clsname;

150  if(!cmdDef){
