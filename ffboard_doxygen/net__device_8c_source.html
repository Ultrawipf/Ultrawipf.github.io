138 uint16_t netd_open(uint8_t rhport, tusb_desc_interface_t const * itf_desc, uint16_t max_len)
139 {
140  bool const is_rndis = (TUD_RNDIS_ITF_CLASS == itf_desc->bInterfaceClass &&
141  TUD_RNDIS_ITF_SUBCLASS == itf_desc->bInterfaceSubClass &&
142  TUD_RNDIS_ITF_PROTOCOL == itf_desc->bInterfaceProtocol);
143 
144  bool const is_ecm = (TUSB_CLASS_CDC == itf_desc->bInterfaceClass &&
145  CDC_COMM_SUBCLASS_ETHERNET_CONTROL_MODEL == itf_desc->bInterfaceSubClass &&
146  0x00 == itf_desc->bInterfaceProtocol);
147 
148  TU_VERIFY(is_rndis || is_ecm, 0);
149 
150  // confirm interface hasn't already been allocated
151  TU_ASSERT(0 == _netd_itf.ep_notif, 0);
152 
153  // sanity check the descriptor
154  _netd_itf.ecm_mode = is_ecm;
155 
156  //------------- Management Interface -------------//
157  _netd_itf.itf_num = itf_desc->bInterfaceNumber;
158 
159  uint16_t drv_len = sizeof(tusb_desc_interface_t);
160  uint8_t const * p_desc = tu_desc_next( itf_desc );
161 
162  // Communication Functional Descriptors
163  while ( TUSB_DESC_CS_INTERFACE == tu_desc_type(p_desc) && drv_len <= max_len )
164  {
165  drv_len += tu_desc_len(p_desc);
166  p_desc = tu_desc_next(p_desc);
167  }
168 
169  // notification endpoint (if any)
170  if ( TUSB_DESC_ENDPOINT == tu_desc_type(p_desc) )
171  {
172  TU_ASSERT( usbd_edpt_open(rhport, (tusb_desc_endpoint_t const *) p_desc), 0 );
173 
174  _netd_itf.ep_notif = ((tusb_desc_endpoint_t const *) p_desc)->bEndpointAddress;
175 
176  drv_len += tu_desc_len(p_desc);
177  p_desc = tu_desc_next(p_desc);
178  }
179 
180  //------------- Data Interface -------------//
181  // - RNDIS Data followed immediately by a pair of endpoints
182  // - CDC-ECM data interface has 2 alternate settings
183  // - 0 : zero endpoints for inactive (default)
184  // - 1 : IN & OUT endpoints for active networking
185  TU_ASSERT(TUSB_DESC_INTERFACE == tu_desc_type(p_desc), 0);
186 
187  do
188  {
189  tusb_desc_interface_t const * data_itf_desc = (tusb_desc_interface_t const *) p_desc;
190  TU_ASSERT(TUSB_CLASS_CDC_DATA == data_itf_desc->bInterfaceClass, 0);
191 
192  drv_len += tu_desc_len(p_desc);
193  p_desc = tu_desc_next(p_desc);
194  }while( _netd_itf.ecm_mode && (TUSB_DESC_INTERFACE == tu_desc_type(p_desc)) && (drv_len <= max_len) );
195 
196  // Pair of endpoints
197  TU_ASSERT(TUSB_DESC_ENDPOINT == tu_desc_type(p_desc), 0);
198 
199  if ( _netd_itf.ecm_mode )
200  {
201  // ECM by default is in-active, save the endpoint attribute
202  // to open later when received setInterface
203  _netd_itf.ecm_desc_epdata = p_desc;
204  }else
205  {
206  // Open endpoint pair for RNDIS
207  TU_ASSERT( usbd_open_edpt_pair(rhport, p_desc, 2, TUSB_XFER_BULK, &_netd_itf.ep_out, &_netd_itf.ep_in), 0 );
208 
209  tud_network_init_cb();
210 
211  // we are ready to transmit a packet
212  can_xmit = true;
213 
214  // prepare for incoming packets
215  tud_network_recv_renew();
216  }
217 
218  drv_len += 2*sizeof(tusb_desc_endpoint_t);
219 
220  return drv_len;
221 }

230 // Invoked when a control transfer occurred on an interface of this class
231 // Driver response accordingly to the request and the transfer stage (setup/data/ack)
232 // return false to stall control endpoint (e.g unsupported request)
233 bool netd_control_xfer_cb (uint8_t rhport, uint8_t stage, tusb_control_request_t const * request)
234 {
235  if ( stage == CONTROL_STAGE_SETUP )
236  {
237  switch ( request->bmRequestType_bit.type )
238  {
239  case TUSB_REQ_TYPE_STANDARD:
240  switch ( request->bRequest )
241  {
242  case TUSB_REQ_GET_INTERFACE:
243  {
244  uint8_t const req_itfnum = (uint8_t) request->wIndex;
245  TU_VERIFY(_netd_itf.itf_num+1 == req_itfnum);
246 
247  tud_control_xfer(rhport, request, &_netd_itf.itf_data_alt, 1);
248  }
249  break;
250 
251  case TUSB_REQ_SET_INTERFACE:
252  {
253  uint8_t const req_itfnum = (uint8_t) request->wIndex;
254  uint8_t const req_alt = (uint8_t) request->wValue;
255 
256  // Only valid for Data Interface with Alternate is either 0 or 1
257  TU_VERIFY(_netd_itf.itf_num+1 == req_itfnum && req_alt < 2);
258 
259  // ACM-ECM only: qequest to enable/disable network activities
260  TU_VERIFY(_netd_itf.ecm_mode);
261 
262  _netd_itf.itf_data_alt = req_alt;
263 
264  if ( _netd_itf.itf_data_alt )
265  {
266  // TODO since we don't actually close endpoint
267  // hack here to not re-open it
268  if ( _netd_itf.ep_in == 0 && _netd_itf.ep_out == 0 )
269  {
270  TU_ASSERT(_netd_itf.ecm_desc_epdata);
271  TU_ASSERT( usbd_open_edpt_pair(rhport, _netd_itf.ecm_desc_epdata, 2, TUSB_XFER_BULK, &_netd_itf.ep_out, &_netd_itf.ep_in) );
272 
273  // TODO should be merge with RNDIS's after endpoint opened
274  // Also should have opposite callback for application to disable network !!
275  tud_network_init_cb();
276  can_xmit = true; // we are ready to transmit a packet
277  tud_network_recv_renew(); // prepare for incoming packets
278  }
279  }else
280  {
281  // TODO close the endpoint pair
282  // For now pretend that we did, this should have no harm since host won't try to
283  // communicate with the endpoints again
284  // _netd_itf.ep_in = _netd_itf.ep_out = 0
285  }
286 
287  tud_control_status(rhport, request);
288  }
289  break;
290 
291  // unsupported request
292  default: return false;
293  }
294  break;
295 
296  case TUSB_REQ_TYPE_CLASS:
297  TU_VERIFY (_netd_itf.itf_num == request->wIndex);
298 
299  if (_netd_itf.ecm_mode)
300  {
301  /* the only required CDC-ECM Management Element Request is SetEthernetPacketFilter */
302  if (0x43 /* SET_ETHERNET_PACKET_FILTER */ == request->bRequest)
303  {
304  tud_control_xfer(rhport, request, NULL, 0);
305  ecm_report(true);
306  }
307  }
308  else
309  {
310  if (request->bmRequestType_bit.direction == TUSB_DIR_IN)
311  {
312  rndis_generic_msg_t *rndis_msg = (rndis_generic_msg_t *) ((void*) notify.rndis_buf);
313  uint32_t msglen = tu_le32toh(rndis_msg->MessageLength);
314  TU_ASSERT(msglen <= sizeof(notify.rndis_buf));
315  tud_control_xfer(rhport, request, notify.rndis_buf, msglen);
316  }
317  else
318  {
319  tud_control_xfer(rhport, request, notify.rndis_buf, sizeof(notify.rndis_buf));
320  }
321  }
322  break;
323 
324  // unsupported request
325  default: return false;
326  }
327  }
328  else if ( stage == CONTROL_STAGE_DATA )
329  {
330  // Handle RNDIS class control OUT only
331  if (request->bmRequestType_bit.type == TUSB_REQ_TYPE_CLASS &&
332  request->bmRequestType_bit.direction == TUSB_DIR_OUT &&
333  _netd_itf.itf_num == request->wIndex)
334  {
335  if ( !_netd_itf.ecm_mode )
336  {
337  rndis_class_set_handler(notify.rndis_buf, request->wLength);
338  }
339  }
340  }
341 
342  return true;
343 }

345 static void handle_incoming_packet(uint32_t len)
346 {
347  uint8_t *pnt = received;
348  uint32_t size = 0;
349 
350  if (_netd_itf.ecm_mode)
351  {
352  size = len;
353  }
354  else
355  {
356  rndis_data_packet_t *r = (rndis_data_packet_t *) ((void*) pnt);
357  if (len >= sizeof(rndis_data_packet_t))
358  if ( (r->MessageType == REMOTE_NDIS_PACKET_MSG) && (r->MessageLength <= len))
359  if ( (r->DataOffset + offsetof(rndis_data_packet_t, DataOffset) + r->DataLength) <= len)
360  {
361  pnt = &received[r->DataOffset + offsetof(rndis_data_packet_t, DataOffset)];
362  size = r->DataLength;
363  }
364  }
365 
366  if (!tud_network_recv_cb(pnt, size))
367  {
368  /* if a buffer was never handled by user code, we must renew on the user's behalf */
369  tud_network_recv_renew();
370  }
371 }

373 bool netd_xfer_cb(uint8_t rhport, uint8_t ep_addr, xfer_result_t result, uint32_t xferred_bytes)
374 {
375  (void) rhport;
376  (void) result;
377 
378  /* new packet received */
379  if ( ep_addr == _netd_itf.ep_out )
380  {
381  handle_incoming_packet(xferred_bytes);
382  }
383 
384  /* data transmission finished */
385  if ( ep_addr == _netd_itf.ep_in )
386  {
387  /* TinyUSB requires the class driver to implement ZLP (since ZLP usage is class-specific) */
388 
389  if ( xferred_bytes && (0 == (xferred_bytes % CFG_TUD_NET_ENDPOINT_SIZE)) )
390  {
391  do_in_xfer(NULL, 0); /* a ZLP is needed */
392  }
393  else
394  {
395  /* we're finally finished */
396  can_xmit = true;
397  }
398  }
399 
400  if ( _netd_itf.ecm_mode && (ep_addr == _netd_itf.ep_notif) )
401  {
402  if (sizeof(notify.ecm_buf.header) == xferred_bytes) ecm_report(false);
403  }
404 
405  return true;
406 }

413 void tud_network_xmit(void *ref, uint16_t arg)
414 {
415  uint8_t *data;
416  uint16_t len;
417 
418  if (!can_xmit)
419  return;
420 
421  len = (_netd_itf.ecm_mode) ? 0 : CFG_TUD_NET_PACKET_PREFIX_LEN;
422  data = transmitted + len;
423 
424  len += tud_network_xmit_cb(data, ref, arg);
425 
426  if (!_netd_itf.ecm_mode)
427  {
428  rndis_data_packet_t *hdr = (rndis_data_packet_t *) ((void*) transmitted);
429  memset(hdr, 0, sizeof(rndis_data_packet_t));
430  hdr->MessageType = REMOTE_NDIS_PACKET_MSG;
431  hdr->MessageLength = len;
432  hdr->DataOffset = sizeof(rndis_data_packet_t) - offsetof(rndis_data_packet_t, DataOffset);
433  hdr->DataLength = len - sizeof(rndis_data_packet_t);
434  }
435 
436  do_in_xfer(transmitted, len);
437 }
