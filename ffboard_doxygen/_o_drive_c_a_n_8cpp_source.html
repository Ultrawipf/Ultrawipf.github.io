77  if(motorId == 0){

92 void ODriveCAN::saveFlash(){
93  uint16_t canIds = 0x3040;
94  Flash_Read(ADR_ODRIVE_CANID, &canIds); // Read again
95  if(motorId == 0){
96  canIds &= ~0x3F; // reset bits
97  canIds |= nodeId & 0x3f;
98  }else if(motorId == 1){
99  canIds &= ~0xFC0; // reset bits
100  nodeId = (canIds & 0x3f) << 6;
101  }
102  canIds &= ~0x7000; // reset bits
103  canIds |= (this->baudrate & 0x7) << 12;
104  Flash_Write(ADR_ODRIVE_CANID,canIds);
105 
106  uint16_t settings1 = ((int32_t)(maxTorque*100) & 0xfff);
107  Flash_Write(ADR_ODRIVE_SETTING1, settings1);
108 }

110 void ODriveCAN::Run(){
111  while(true){
112  this->Delay(500);
113 
114  switch(state){
115  case ODriveLocalState::WAIT_READY:
116  if(this->odriveCurrentState == ODriveState::AXIS_STATE_IDLE){
117  this->state = ODriveLocalState::WAIT_CALIBRATION;
118  this->setState(ODriveState::AXIS_STATE_FULL_CALIBRATION_SEQUENCE);
119  }
120 
121  break;
122 
123  // Calibration in progress. wait until its finished to enter torque mode
124  case ODriveLocalState::WAIT_CALIBRATION:
125  if(odriveCurrentState == ODriveState::AXIS_STATE_IDLE){
126  setState(ODriveState::AXIS_STATE_CLOSED_LOOP_CONTROL);
127  state = ODriveLocalState::START_RUNNING;
128 
129  }else if(odriveCurrentState == ODriveState::AXIS_STATE_CLOSED_LOOP_CONTROL){
130  state = ODriveLocalState::START_RUNNING;
131  }
132  break;
133 
134  case ODriveLocalState::START_RUNNING:
135  state = ODriveLocalState::RUNNING;
136  // Odrive is active,
137  // enable torque mode
138  if(odriveCurrentState == ODriveState::AXIS_STATE_CLOSED_LOOP_CONTROL){
139  this->setPos(0);
140  setMode(ODriveControlMode::CONTROL_MODE_TORQUE_CONTROL, ODriveInputMode::INPUT_MODE_PASSTHROUGH);
141  }
142 
143  break;
144  default:
145 
146  break;
147  }
148 
149  // If odrive is currently performing any calibration task wait until finished
150  if(odriveCurrentState == ODriveState::AXIS_STATE_FULL_CALIBRATION_SEQUENCE || odriveCurrentState == ODriveState::AXIS_STATE_MOTOR_CALIBRATION || odriveCurrentState == ODriveState::AXIS_STATE_ENCODER_OFFSET_CALIBRATION || odriveCurrentState == ODriveState::AXIS_STATE_ENCODER_HALL_PHASE_CALIBRATION || odriveCurrentState == ODriveState::AXIS_STATE_ENCODER_INDEX_SEARCH){
151  if(state != ODriveLocalState::WAIT_CALIBRATION)
152  state = ODriveLocalState::WAIT_CALIBRATION;
153  // If closed loop mode is on assume its ready and already calibrated
154  }else if(odriveCurrentState == ODriveState::AXIS_STATE_CLOSED_LOOP_CONTROL){
155  if(state != ODriveLocalState::START_RUNNING && state != ODriveLocalState::RUNNING)
156  state = ODriveLocalState::RUNNING;
157  }
158 
159  if(HAL_GetTick() - lastVoltageUpdate > 1000){
160  requestMsg(0x17); // Update voltage
161  }
162  }
163 }

250 ParseStatus ODriveCAN::command(ParsedCommand* cmd,std::string* reply){
251  // Prefix set but not our prefix
252  if(cmd->prefix != this->getInfo().unique && cmd->prefix != '\0'){
253  return ParseStatus::NOT_FOUND;
254  }
255  ParseStatus status = ParseStatus::OK;
256  if(cmd->cmd == "odriveCanId"){
257  handleGetSet(cmd, reply, this->nodeId);
258 
259  }else if(cmd->cmd == "odriveMaxTorque"){
260  if(cmd->type == CMDtype::get){
261  int32_t val = maxTorque*100;
262  *reply += std::to_string(val);
263  }else if(cmd->type == CMDtype::set){
264  maxTorque = (float)clip(cmd->val, 0, 0xfff) / 100.0;
265  }
266  }else if(cmd->cmd == "odriveVbus"){
267  if(cmd->type == CMDtype::get){
268  //requestMsg(0x17); // Update voltage for next time
269  int32_t val = lastVoltage*1000;
270  *reply += std::to_string(val);
271  }
272 
273  }else if(cmd->cmd == "odriveErrors"){
274  if(cmd->type == CMDtype::get){
275  *reply += std::to_string(errors);
276  }
277 
278  }else if(cmd->cmd == "odriveAnticogging"){
279  if(cmd->type == CMDtype::set && cmd->val == 1){
280  this->startAnticogging();
281  }else{
282  *reply+="=1 to start calibration sequence";
283  }
284  }else if(cmd->cmd == "odriveState"){
285  if(cmd->type == CMDtype::get){
286  *reply += std::to_string((uint32_t)odriveCurrentState);
287  }
288 
289  }else if(cmd->cmd == "odriveCanSpd"){
290  if(cmd->type == CMDtype::get){
291  *reply += std::to_string(baudrate);
292  }else if(cmd->type == CMDtype::set){
293  setCanRate(cmd->val);
294  }
295  }else{
296  status = ParseStatus::NOT_FOUND;
297  }
298  return status;
299 }

302 void ODriveCAN::canRxPendCallback(CAN_HandleTypeDef *hcan,uint8_t* rxBuf,CAN_RxHeaderTypeDef* rxHeader,uint32_t fifo){
303 
304  uint16_t node = (rxHeader->StdId >> 5) & 0x3F;
305  if(node != this->nodeId){
306  return;
307  }
308  uint64_t msg = *reinterpret_cast<uint64_t*>(rxBuf);
309  uint8_t cmd = rxHeader->StdId & 0x1F;
310 
311  switch(cmd){
312  case 1:
313  {
314  // TODO error handling
315  errors = msg & 0xffffffff;
316  odriveCurrentState = (ODriveState)( (msg >> 32) & 0xffffffff);
317  if(waitReady){
318  waitReady = false;
319  state = ODriveLocalState::WAIT_READY;
320  }
321  break;
322  }
323  case 0x09: // encoder pos float
324  {
325  uint64_t tp = msg & 0xffffffff;
326  memcpy(&lastPos,&tp,sizeof(float));
327

329  memcpy(&lastSpeed,&ts,sizeof(float));

331  }
332
