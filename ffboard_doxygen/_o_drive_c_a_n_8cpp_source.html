105 void ODriveCAN::saveFlash(){
106  uint16_t setting1addr = ADR_ODRIVE_SETTING1_M0;
107 
108  uint16_t canIds = 0x3040;
109  Flash_Read(ADR_ODRIVE_CANID, &canIds); // Read again
110  if(motorId == 0){
111  canIds &= ~0x3F; // reset bits
112  canIds |= nodeId & 0x3f;
113  }else if(motorId == 1){
114  setting1addr = ADR_ODRIVE_SETTING1_M1;
115  canIds &= ~0xFC0; // reset bits
116  canIds |= (nodeId & 0x3f) << 6;
117  }
118  canIds &= ~0x7000; // reset bits

120  Flash_Write(ADR_ODRIVE_CANID,canIds);

126 void ODriveCAN::Run(){
127  while(true){
128  this->Delay(500);
129 
130  switch(state){
131  case ODriveLocalState::WAIT_READY:
132  if(this->odriveCurrentState == ODriveState::AXIS_STATE_IDLE){
133  this->state = ODriveLocalState::WAIT_CALIBRATION; // Wait
134  this->setState(ODriveState::AXIS_STATE_FULL_CALIBRATION_SEQUENCE);
135  }
136 
137  break;
138 
139  // Calibration in progress. wait until its finished to enter torque mode
140  case ODriveLocalState::WAIT_CALIBRATION_DONE:
141  if(odriveCurrentState == ODriveState::AXIS_STATE_IDLE){
142  setState(ODriveState::AXIS_STATE_CLOSED_LOOP_CONTROL);
143  state = ODriveLocalState::START_RUNNING;
144 
145  }else if(odriveCurrentState == ODriveState::AXIS_STATE_CLOSED_LOOP_CONTROL){
146  state = ODriveLocalState::START_RUNNING;
147  }
148  break;
149 
150  case ODriveLocalState::START_RUNNING:
151  state = ODriveLocalState::RUNNING;
152  // Odrive is active,
153  // enable torque mode
154  if(odriveCurrentState == ODriveState::AXIS_STATE_CLOSED_LOOP_CONTROL){
155  this->setPos(0);
156  setMode(ODriveControlMode::CONTROL_MODE_TORQUE_CONTROL, ODriveInputMode::INPUT_MODE_PASSTHROUGH);
157  }
158 
159  break;
160  default:
161 
162  break;
163  }
164 
165  // If odrive is currently performing any calibration task wait until finished
166  if(odriveCurrentState == ODriveState::AXIS_STATE_FULL_CALIBRATION_SEQUENCE || odriveCurrentState == ODriveState::AXIS_STATE_MOTOR_CALIBRATION || odriveCurrentState == ODriveState::AXIS_STATE_ENCODER_OFFSET_CALIBRATION || odriveCurrentState == ODriveState::AXIS_STATE_ENCODER_HALL_PHASE_CALIBRATION || odriveCurrentState == ODriveState::AXIS_STATE_ENCODER_INDEX_SEARCH){
167  if(state != ODriveLocalState::WAIT_CALIBRATION_DONE)
168  state = ODriveLocalState::WAIT_CALIBRATION_DONE;
169  // If closed loop mode is on assume its ready and already calibrated
170  }else if(odriveCurrentState == ODriveState::AXIS_STATE_CLOSED_LOOP_CONTROL){
171  if(state != ODriveLocalState::START_RUNNING && state != ODriveLocalState::RUNNING)
172  state = ODriveLocalState::RUNNING;
173  }
174 
175  if(HAL_GetTick() - lastVoltageUpdate > 1000){
176  requestMsg(0x17); // Update voltage
177  }
178 
179  if(HAL_GetTick() - lastCanMessage > 2000){ // Timeout
180  odriveCurrentState = ODriveState::AXIS_STATE_UNDEFINED;
181  state = ODriveLocalState::IDLE;
182  waitReady = true;
183  connected = false;
184  }else{
185  connected = true;
186  }
187 
188  }
189 }

348 void ODriveCAN::canRxPendCallback(CAN_HandleTypeDef *hcan,uint8_t* rxBuf,CAN_RxHeaderTypeDef* rxHeader,uint32_t fifo){
349 
350  uint16_t node = (rxHeader->StdId >> 5) & 0x3F;
351  if(node != this->nodeId){
352  return;
353  }
354  uint64_t msg = *reinterpret_cast<uint64_t*>(rxBuf);
355  uint8_t cmd = rxHeader->StdId & 0x1F;
356 
357  lastCanMessage = HAL_GetTick();
358 
359  switch(cmd){
360  case 1:
361  {
362  // TODO error handling
363  errors = (msg & 0xffffffff);
364  odriveCurrentState = (ODriveState)( (msg >> 32) & 0xff);
365  odriveMotorFlags = (msg >> 40) & 0xff;
366  odriveEncoderFlags = ((msg >> 48) & 0xff);
367  odriveControllerFlags = (msg >> 56) & 0xff;
368 
369  if(waitReady){
370  waitReady = false;
371  state = ODriveLocalState::WAIT_READY;
372  }
373  break;
374  }
375  case 0x09: // encoder pos float
376  {
377  uint64_t tp = msg & 0xffffffff;
378  memcpy(&lastPos,&tp,sizeof(float));
379 
380  uint64_t ts = (msg >> 32) & 0xffffffff;
381  memcpy(&lastSpeed,&ts,sizeof(float));
382  break;
383  }
384 
385  case 0x17: // voltage
386  {
387  lastVoltageUpdate = HAL_GetTick();
388  uint64_t t = msg & 0xffffffff;
389  memcpy(&lastVoltage,&t,sizeof(float));
390 
391  break;
392  }

394 
395
