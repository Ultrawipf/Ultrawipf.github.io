145 
146  // Not a gvret binary command
147 
148  if(*Buf == 0xE7){
149  gvretMode = true;
150  }
151 
152  if(gvretMode == false || (*Buf != 0xE7 && *Buf != 0xF1)){ // Not gvret command
153  FFBoardMain::cdcRcv(Buf, Len);
154  return;
155  }
156 
157 
158  /*
159  * Data format in buf:
160  * 0xE1,0xE1,0xF1,cmd,data...
161  * data = buf+2
162  */
163 
164  // Find start marker
165  uint8_t pos = 0;
166  for(uint8_t i = 0; i < *Len-pos; i++){
167  pos = i+1;
168  if((Buf[i]) == 0xF1){
169  break;
170  }
171  }
172 
173  std::vector<char> reply;
174  while(pos < *Len){
175 
176  uint8_t datalength = *Len;
177  uint8_t* data = (uint8_t*)(Buf+pos+1);
178  uint8_t cmd = *(Buf+pos);
179 
180  // find next F1 or end
181  for(uint8_t i = 0; i < *Len-pos; i++){
182  datalength = i;
183  if((data[i]) == 0xF1){
184  break;
185  }
186  }
187  pos = pos+datalength+2; // Move offset for next loop
188 
189 
190  switch(cmd){
191 
192  case(0):
193  { // send can
194  if(*Len < 8)
195  break;
196  uint32_t id = *data | (*(data+1) << 8) | (*(data+2) << 16) | (*(data+3) << 24); // = dat[2] | dat[3] >> 8 | dat[3] >> 16 | dat[4] >> 21
197  uint64_t msg = 0;
198  uint8_t bus = data[4];
199  uint8_t msgLen = data[5];
200  if(bus == 0){
201  memcpy(&msg,data+6,std::min<int>(msgLen, 8)); // Copy variable length buffer

203  }
