171 void CanBridge::cdcRcv(char* Buf, uint32_t *Len){
172 
173  // Not a gvret binary command
174 
175  if(*Buf == 0xE7){
176  gvretMode = true;
177  }
178 
179  if(gvretMode == false || (*Buf != 0xE7 && *Buf != 0xF1)){ // Not gvret command
180  FFBoardMain::cdcRcv(Buf, Len);
181  return;
182  }
183 
184 
185  /*
186  * Data format in buf:
187  * 0xE1,0xE1,0xF1,cmd,data...
188  * data = buf+2
189  */
190 
191  // Find start marker
192  uint8_t pos = 0;
193  for(uint8_t i = 0; i < *Len-pos; i++){
194  pos = i+1;
195  if((Buf[i]) == 0xF1){
196  break;
197  }
198  }
199 
200  std::vector<char> reply;
201  while(pos < *Len){
202 
203  uint8_t datalength = *Len;
204  uint8_t* data = (uint8_t*)(Buf+pos+1);
205  uint8_t cmd = *(Buf+pos);
206 
207  // find next F1 or end
208  for(uint8_t i = 0; i < *Len-pos; i++){
209  datalength = i;
210  if((data[i]) == 0xF1){
211  break;
212  }
213  }
214  pos = pos+datalength+2; // Move offset for next loop
215 
216 
217  switch(cmd){
218 
219  case(0):
220  { // send can
221  if(*Len < 8)
222  break;
223  uint32_t id = *data | (*(data+1) << 8) | (*(data+2) << 16) | (*(data+3) << 24); // = dat[2] | dat[3] >> 8 | dat[3] >> 16 | dat[4] >> 21
224  uint64_t msg = 0;
225  uint8_t bus = data[4];
226  uint8_t msgLen = data[5];
227  if(bus == 0){
228  memcpy(&msg,data+6,std::min<int>(msgLen, 8)); // Copy variable length buffer
229  sendMessage(id, msg,msgLen);
230  }
231 
232  break;
233  }

238  reply.insert(reply.end(),t.begin(),t.end());
239 
240  break;
241  }

243 
244  break;
245 
246  case(3): // get analog in
247 
248  break;
249 
250  case(4): // set digital out
251 
252  break;
253 
254  case(5): // set can config
255  {
256  uint32_t speed = *data | (*(data+1) << 8) | (*(data+2) << 16) | (*(data+3) << 24);
