49 uint16_t CDCcomm::cdcSend(std::string* reply,uint8_t itf){
50  if(!tud_ready() || reply->empty()){
51  return 0;
52  }
53  cdcSems[itf].Take();
54  uint32_t bufferRemaining = tud_cdc_n_write_available(itf);
55  uint32_t cdc_sent = tud_cdc_n_write(itf,reply->c_str(), std::min<uint16_t>(reply->length(),bufferRemaining));
56  if(!usb_busy_retry) // We did not retransmit so flush now. otherwise TUD will flush if we were in the callback before
57  tud_cdc_n_write_flush(itf);
58 
59  // If we can't write the whole reply copy remainder to send later
60  if(cdc_sent < reply->length()){
61  remainingStrs[itf] = reply->substr(cdc_sent);
62  usb_busy_retry = true;
63  remainingStrs[itf].shrink_to_fit();
64  }else{
65  usb_busy_retry = false;
66  remainingStrs[itf].clear();
67  }
68 
69  return cdc_sent;
70 }
