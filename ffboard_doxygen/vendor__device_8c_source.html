80 //--------------------------------------------------------------------+
81 // Read API
82 //--------------------------------------------------------------------+
83 static void _prep_out_transaction (vendord_interface_t* p_itf)
84 {
85  // skip if previous transfer not complete
86  if ( usbd_edpt_busy(TUD_OPT_RHPORT, p_itf->ep_out) ) return;
87 
88  // Prepare for incoming data but only allow what we can store in the ring buffer.
89  uint16_t max_read = tu_fifo_remaining(&p_itf->rx_ff);
90  if ( max_read >= CFG_TUD_VENDOR_EPSIZE )
91  {
92  usbd_edpt_xfer(TUD_OPT_RHPORT, p_itf->ep_out, p_itf->epout_buf, CFG_TUD_VENDOR_EPSIZE);
93  }
94 }

96 uint32_t tud_vendor_n_read (uint8_t itf, void* buffer, uint32_t bufsize)
97 {
98  vendord_interface_t* p_itf = &_vendord_itf[itf];
99  uint32_t num_read = tu_fifo_read_n(&p_itf->rx_ff, buffer, bufsize);

104 //--------------------------------------------------------------------+
105 // Write API
106 //--------------------------------------------------------------------+
107 static bool maybe_transmit(vendord_interface_t* p_itf)
108 {
109  // skip if previous transfer not complete
110  TU_VERIFY( !usbd_edpt_busy(TUD_OPT_RHPORT, p_itf->ep_in) );
111 
112  uint16_t count = tu_fifo_read_n(&p_itf->tx_ff, p_itf->epin_buf, CFG_TUD_VENDOR_EPSIZE);
113  if (count > 0)
114  {
115  TU_ASSERT( usbd_edpt_xfer(TUD_OPT_RHPORT, p_itf->ep_in, p_itf->epin_buf, count) );
116  }
117  return true;
118 }

136 void vendord_init(void)
137 {
138  tu_memclr(_vendord_itf, sizeof(_vendord_itf));
139 
140  for(uint8_t i=0; i<CFG_TUD_VENDOR; i++)
141  {
142  vendord_interface_t* p_itf = &_vendord_itf[i];
143 
144  // config fifo
145  tu_fifo_config(&p_itf->rx_ff, p_itf->rx_ff_buf, CFG_TUD_VENDOR_RX_BUFSIZE, 1, false);
146  tu_fifo_config(&p_itf->tx_ff, p_itf->tx_ff_buf, CFG_TUD_VENDOR_TX_BUFSIZE, 1, false);
147 
148 #if CFG_FIFO_MUTEX
149  tu_fifo_config_mutex(&p_itf->rx_ff, osal_mutex_create(&p_itf->rx_ff_mutex));
150  tu_fifo_config_mutex(&p_itf->tx_ff, osal_mutex_create(&p_itf->tx_ff_mutex));
151 #endif
152  }
153 }

169 uint16_t vendord_open(uint8_t rhport, tusb_desc_interface_t const * itf_desc, uint16_t max_len)
170 {
171  TU_VERIFY(TUSB_CLASS_VENDOR_SPECIFIC == itf_desc->bInterfaceClass, 0);
172 
173  uint16_t const drv_len = sizeof(tusb_desc_interface_t) + itf_desc->bNumEndpoints*sizeof(tusb_desc_endpoint_t);
174  TU_VERIFY(max_len >= drv_len, 0);
175 
176  // Find available interface
177  vendord_interface_t* p_vendor = NULL;
178  for(uint8_t i=0; i<CFG_TUD_VENDOR; i++)
179  {
180  if ( _vendord_itf[i].ep_in == 0 && _vendord_itf[i].ep_out == 0 )
181  {
182  p_vendor = &_vendord_itf[i];
183  break;
184  }
185  }
186  TU_VERIFY(p_vendor, 0);
187 
188  // Open endpoint pair with usbd helper
189  TU_ASSERT(usbd_open_edpt_pair(rhport, tu_desc_next(itf_desc), 2, TUSB_XFER_BULK, &p_vendor->ep_out, &p_vendor->ep_in), 0);
190 
191  p_vendor->itf_num = itf_desc->bInterfaceNumber;
192 
193  // Prepare for incoming data
194  if ( !usbd_edpt_xfer(rhport, p_vendor->ep_out, p_vendor->epout_buf, sizeof(p_vendor->epout_buf)) )
195  {
196  TU_LOG1_FAILED();
197  TU_BREAKPOINT();
198  }
199 
200  return drv_len;
201 }

203 bool vendord_xfer_cb(uint8_t rhport, uint8_t ep_addr, xfer_result_t result, uint32_t xferred_bytes)
204 {
205  (void) rhport;
206  (void) result;
207 
208  uint8_t itf = 0;
209  vendord_interface_t* p_itf = _vendord_itf;
210 
211  for ( ; ; itf++, p_itf++)
212  {
213  if (itf >= TU_ARRAY_SIZE(_vendord_itf)) return false;
214 
215  if ( ( ep_addr == p_itf->ep_out ) || ( ep_addr == p_itf->ep_in ) ) break;
216  }
217 
218  if ( ep_addr == p_itf->ep_out )
219  {
220  // Receive new data
221  tu_fifo_write_n(&p_itf->rx_ff, p_itf->epout_buf, xferred_bytes);
222 
223  // Invoked callback if any
224  if (tud_vendor_rx_cb) tud_vendor_rx_cb(itf);
225 
226  _prep_out_transaction(p_itf);
227  }
228  else if ( ep_addr == p_itf->ep_in )
229  {
230  // Send complete, try to send more if possible
231  maybe_transmit(p_itf);
232  }
233 
234  return true;
235 }
