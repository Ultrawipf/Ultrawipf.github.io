221  * Executes parsed commands and calls other command handlers.
222  * Not global so it can be overridden by main classes to change behaviour or suppress outputs.
223  */
224 void FFBoardMainCommandThread::executeCommands(std::vector<ParsedCommand> commands){
225  if(clearReply)
226  this->cmd_reply.clear();
227  for(ParsedCommand cmd : commands){
228  ParseStatus status = ParseStatus::NOT_FOUND;
229  if(cmd.cmd.empty())
230  continue; // Empty command
231 
232  this->cmd_reply+= ">" + cmd.rawcmd + ":"; // Start marker. Echo command
233 
234  std::string reply; // Reply of current command
235  status = executeSysCommand(&cmd,&reply);
236  if(status == ParseStatus::NOT_FOUND || status == ParseStatus::OK_CONTINUE){ // Not a system command
237  // Call all command handlers
238 
239  // Prevent other threads from running while we call all command handlers
240  // TODO monitor if this causes problems or is really needed.
241  //cpp_freertos::CriticalSection::SuspendScheduler();
242  for(CommandHandler* handler : CommandHandler::cmdHandlers){
243  if(handler->hasCommands()){
244  ParseStatus newstatus = handler->command(&cmd,&reply);
245  // If last class did not have commands but a previous one asked to continue keep continue status
246  if(!(status == ParseStatus::OK_CONTINUE && newstatus == ParseStatus::NOT_FOUND)){
247  status = newstatus;
248  }
249  if(status == ParseStatus::ERR || status == ParseStatus::OK || status == ParseStatus::NO_REPLY)
250  break; // Stop after this class if finished flag is returned
251  }
252  }
253  // Resume scheduler
254  //cpp_freertos::CriticalSection::ResumeScheduler();
255  }
256  if(status == ParseStatus::NO_REPLY){
257  cmd_reply.clear();
258  continue; // don't send reply. Just continue
259  }
260 
261  if(reply.empty() && status == ParseStatus::OK){
262  reply = "OK";
263  }
264  // Append newline if reply is not empty
265  if(!reply.empty() && reply.back()!='\n'){
266  reply+='\n';
267  }
268  // Errors
269  if(status == ParseStatus::NOT_FOUND){ //No class reported success. Show error
270  Error err = cmdNotFoundError;
271  reply = "Err. invalid";
272  err.info = cmd.rawcmd + " not found";
273  ErrorHandler::addError(err);
274 
275  }else if(status == ParseStatus::ERR){ //Error reported in command
276  reply = "Err. exec error";
277  Error err = cmdExecError;
278  err.info = "Error executing" + cmd.rawcmd;
279  ErrorHandler::addError(err);
280  }
281  this->cmd_reply+=reply;
282  }
283 
284  if(this->cmd_reply.length()>0){ // Check if cdc busy
285  this->main->parserDone(&cmd_reply, this);
286  }
287 }
