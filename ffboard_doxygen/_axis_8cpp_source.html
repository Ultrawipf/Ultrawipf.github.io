169 void Axis::prepareForUpdate(){
170  if (drv == nullptr){
171  pulseErrLed();
172  return;
173  }
174 
175  if (!drv->motorReady()) return;
176 
177  float angle = getEncAngle(this->drv->getEncoder());
178 
179  // Scale encoder value to set rotation range
180  // Update a change of range only when new range is within valid range
181  // if degree change, compute the SpeedScaler, it depends on degreesOfRotation
182  if (nextDegreesOfRotation != degreesOfRotation && abs(scaleEncValue(angle, nextDegreesOfRotation)) < 0x7fff){
183  degreesOfRotation = nextDegreesOfRotation;
184 
185 // speedScalerNormalized = getNormalizedSpeedScaler(maxSpeedDegS, degreesOfRotation);
186 // accelScalerNormalized = getNormalizedAccelScaler(maxAccelDegSS, degreesOfRotation);
187  }
188 
189 
190  // scaledEnc now gets inverted if necessary in updateMetrics
191  int32_t scaledEnc = scaleEncValue(angle, degreesOfRotation);
192 
193  if (abs(scaledEnc) > 0xffff){
194  // We are way off. Shut down
195  drv->stopMotor();
196  pulseErrLed();
197  }
198 
199  this->updateMetrics(angle);
200

485 int16_t Axis::updateEndstop(){
486  int8_t clipdir = cliptest<int32_t,int32_t>(metric.current.pos, -0x7fff, 0x7fff);
487  if(clipdir == 0){
488  return 0;
489  }
490  int32_t addtorque = clip<int32_t,int32_t>(abs(metric.current.pos)-0x7fff,-0x7fff,0x7fff);
491  addtorque *= (float)endstop_gain * 1.25 * torqueScaler; // Apply endstop gain for stiffness. 0.15f
492  addtorque *= -clipdir;
493 
494  return clip<int32_t,int32_t>(addtorque,-0x7fff,0x7fff);

501 // pass in ptr to receive the sum of the effects + endstop torque
502 // return true if torque is clipping
503 bool Axis::updateTorque(int32_t* totalTorque) {
504 
505  if(abs(effectTorque) >= 0x7fff){
506  pulseClipLed();
507  }
508 
509  // Scale effect torque
510  effectTorque *= torqueScaler;
511 
512  int32_t torque = effectTorque + updateEndstop();
513  torque += axisEffectTorque * torqueScaler; // Updated from effect calculator
514 
515  // TODO speed and accel limiters
516  if(maxSpeedDegS > 0){
517 
518  float torqueSign = torque > 0 ? 1 : -1; // Used to prevent metrics against the force to go into the limiter
519  // Speed. Mostly tuned...
520  spdlimiterAvg.addValue(metric.current.speedInstant);
521  float speedreducer = (float)((spdlimiterAvg.getAverage()*torqueSign) - (float)maxSpeedDegS) * getSpeedScalerNormalized();
522  spdlimitreducerI = clip<float,int32_t>( spdlimitreducerI + ((speedreducer * 0.015) * torqueScaler),0,power);
523 
524  // Accel limit. Not really useful. Maybe replace with torque slew rate limit?
525 // float accreducer = (float)((metric.current.accel*torqueSign) - (float)maxAccelDegSS) * getAccelScalerNormalized();
526 // acclimitreducerI = clip<float,int32_t>( acclimitreducerI + ((accreducer * 0.02) * torqueScaler),0,power);
527 
528 
529  // Only reduce torque. Don't invert it to prevent oscillation
530  float torqueReduction = spdlimitreducerI + speedreducer * 0.025;// accreducer * 0.025 + acclimitreducerI
531  if(torque > 0){
532  torqueReduction = clip<float,int32_t>(torqueReduction,0,torque);
533  }else{
534  torqueReduction = clip<float,int32_t>(-torqueReduction,torque,0);
535  }
536 
537  torque -= torqueReduction;
538  }
539  // Torque slew rate limiter
540  if(maxTorqueRateMS > 0){
541  torque = clip<int32_t,int32_t>(torque, metric.previous.torque - maxTorqueRateMS,metric.previous.torque + maxTorqueRateMS);
542  }
543  if(torque - metric.previous.torque)
544 
545  // Torque calculated. Now sending to driver
546  torque = (invertAxis) ? -torque : torque;
547  metric.current.torque = torque;
548  torque = clip<int32_t, int32_t>(torque, -power, power);
549 
550  bool torqueChanged = torque != metric.previous.torque;
551 
552  if (abs(torque) == power){
553  pulseClipLed();
554  }
555 
556  *totalTorque = torque;
557  return (torqueChanged);
558 }

572 CommandStatus Axis::command(const ParsedCommand& cmd,std::vector<CommandReply>& replies){
573 
574  switch(static_cast<Axis_commands>(cmd.cmdId)){
575 
576  case Axis_commands::power:
577  if (cmd.type == CMDtype::get)
578  {
579  replies.push_back(CommandReply(this->power));
580  }
581  else if (cmd.type == CMDtype::set)
582  {
583  setPower(cmd.val);
584  }
585  break;
586 
587  case Axis_commands::degrees:
588  handleGetSetFunc(cmd, replies, degreesOfRotation, &Axis::setDegrees,this);
589 // if (cmd.type == CMDtype::get)
590 // {
591 // replies.push_back(CommandReply(degreesOfRotation));
592 // }
593 // else if (cmd.type == CMDtype::set)
594 // {
595 // setDegrees(cmd.val);
596 // }
597  break;
598 
599  case Axis_commands::esgain:
600  handleGetSet(cmd, replies, this->endstop_gain);
601  break;
602 
603  case Axis_commands::zeroenc:
604  this->setPos(0);
605  break;
606 
607  case Axis_commands::invert:
608  if (cmd.type == CMDtype::get)
609  {
610  replies.push_back(CommandReply(invertAxis ? 1 : 0));
611  }
612  else if (cmd.type == CMDtype::set)
613  {
614  invertAxis = cmd.val >= 1 ? true : false;
615  resetMetrics(-metric.current.posDegrees);
616  }
617  break;
618 
619  case Axis_commands::idlespring:
620  if (cmd.type == CMDtype::get)
621  {
622  replies.push_back(CommandReply(idlespringstrength));
623  }
624  else if (cmd.type == CMDtype::set)
625  {
626  setIdleSpringStrength(cmd.val);
627  }
628  break;
629 
630  case Axis_commands::axisdamper:
631  if (cmd.type == CMDtype::get)
632  {
633  replies.push_back(CommandReply(damperIntensity));
634  }
635  else if (cmd.type == CMDtype::set)
636  {
637  setDamperStrength(cmd.val);
638  }
639  break;
640 
641  case Axis_commands::enctype:
642  if(cmd.type == CMDtype::info){
643  enc_chooser.replyAvailableClasses(replies);
644  }else if(cmd.type == CMDtype::get){
645  replies.push_back(CommandReply(this->getEncType()));
646  }else if(cmd.type == CMDtype::set){
647  this->setEncType(cmd.val);
648  }
649  break;
650 
651  case Axis_commands::drvtype:
652  if(cmd.type == CMDtype::info){
653  drv_chooser.replyAvailableClasses(replies);
654  }else if(cmd.type == CMDtype::get){
655  replies.push_back(CommandReply(this->getDrvType()));
656  }else if(cmd.type == CMDtype::set){
657  this->setDrvType(cmd.val);
658  }
659  break;
660 
661  case Axis_commands::pos:
662  if (cmd.type == CMDtype::get)
663  {
664  replies.push_back(CommandReply(this->drv->getEncoder()->getPos()));
665  }
666  else if (cmd.type == CMDtype::set && this->drv->getEncoder() != nullptr)
667  {
668  this->drv->getEncoder()->setPos(cmd.val);
669  }
670  else
671  {
672  return CommandStatus::ERR;
673  }
674  break;
675 
676  case Axis_commands::maxspeed:
677  handleGetSet(cmd, replies, this->maxSpeedDegS);
678  break;
679 
680  case Axis_commands::maxtorquerate:
681  handleGetSet(cmd, replies, this->maxTorqueRateMS);
682  break;
683 
684  case Axis_commands::fxratio:
685  if(cmd.type == CMDtype::get){
686  replies.push_back(CommandReply(this->fx_ratio_i));
687  }else if(cmd.type == CMDtype::set){
688  setFxRatio(cmd.val);
689  }
690  break;
691 
692  default:
693  return CommandStatus::NOT_FOUND;
694  }
695  return CommandStatus::OK;
696 }
