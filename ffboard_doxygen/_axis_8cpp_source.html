92 void Axis::prepareForUpdate(){
93  if (drv == nullptr){
94  pulseErrLed();
95  return;
96  }
97 
98  if (!drv->motorReady()) return;
99 
100  // Scale encoder value to set rotation range
101  // Update a change of range only when new range is within valid range
102  if (nextDegreesOfRotation != degreesOfRotation && abs(getEncValue(drv->getEncoder(), nextDegreesOfRotation)) < 0x7fff){
103  degreesOfRotation = nextDegreesOfRotation;
104  }
105  // scaledEnc now gets inverted if necessary in updateMetrics
106  int32_t scaledEnc = getEncValue(drv->getEncoder(), degreesOfRotation);
107 
108  if (abs(scaledEnc) > 0xffff){
109  // We are way off. Shut down
110  drv->stopMotor();
111  pulseErrLed();
112  }
113 
114  this->updateMetrics(scaledEnc);
115 
116 }

221 int32_t Axis::getEncValue(Encoder *enc, uint16_t degrees){
222  if (enc == nullptr || degrees == 0){
223  return 0x7fff; // Return center if no encoder present
224  }
225  float angle = 360.0 * enc->getPos_f();
226  int32_t val = (0xffff / (float)degrees) * angle;
227  if (isInverted()){
228  val= -val;
229  }
230  return val;
