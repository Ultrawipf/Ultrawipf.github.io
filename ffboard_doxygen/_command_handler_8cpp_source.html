172 CommandStatus CommandHandler::internalCommand(const ParsedCommand& cmd,std::vector<CommandReply>& replies,CommandInterface* interface){
173  CommandStatus result = CommandStatus::OK;
174  switch(static_cast<CommandHandlerCommands>(cmd.cmdId)){
175 
176  case CommandHandlerCommands::id:
177  replies.push_back(CommandReply(this->getInfo().id));
178  break;
179 
180  case CommandHandlerCommands::name:
181  replies.push_back(CommandReply(this->getInfo().name));
182  break;
183 
184  case CommandHandlerCommands::help:
185  if(cmd.type == CMDtype::info){
186  replies.push_back(CommandReply(this->getCsvHelpstring()));
187  }else{
188  replies.push_back(CommandReply(this->getCommandsHelpstring()));
189  }
190  break;
191 
192  case CommandHandlerCommands::cmdhandleruid:
193  replies.push_back(CommandReply(this->getCommandHandlerID()));
194  break;
195 
196  case CommandHandlerCommands::instance:
197  replies.push_back(CommandReply(this->getCommandHandlerInfo()->instance));
198  break;
199 
200  case CommandHandlerCommands::selectionid:
201  replies.push_back(CommandReply(this->getSelectionID()));
202  break;
203 
204  default:
205  result = CommandStatus::NOT_FOUND;
206  break;
207  }
208  return result;
209 }
