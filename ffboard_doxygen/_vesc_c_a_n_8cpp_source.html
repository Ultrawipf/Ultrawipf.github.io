113 void VescCAN::restoreFlash() {
114  uint16_t dataFlash = 0;
115 
116  if (Flash_Read(flashAddrs.canId, &dataFlash)) {
117  this->OFFB_can_Id = dataFlash & 0xFF;
118  this->VESC_can_Id = (dataFlash >> 8) & 0xFF;
119  }
120 
121  if (Flash_Read(flashAddrs.data, &dataFlash)) {
122 
123  uint8_t canspd = dataFlash & 0b111;
124  this->baudrate = canspd;
125 
126  this->useEncoder = (dataFlash >> 3) & 0x1;
127  }
128 
129  if (Flash_Read(flashAddrs.offset, &dataFlash)) {
130  this->posOffset = (float) ((int16_t) dataFlash / 10000.0);
131 
132  this->posOffset = this->posOffset - (int) this->posOffset; // Remove the multi-turn value
133  bool moreThanHalfTurn = fabs(this->posOffset) > 0.5 ? 1 : 0;
134  if (moreThanHalfTurn) {
135  // if delta is neg, turn is CCW, decrement multi turn pos... else increment it
136  this->posOffset += (this->posOffset > 0) ? -1.0 : 1.0;
137  }
138  }

142 void VescCAN::saveFlash() {
143  uint16_t dataFlash = 0;
144 
145  // save the OpenFFBoard Can Id in 0..7 and the Vesc_can_id in the 8..15
146  dataFlash = (this->VESC_can_Id << 8) | this->OFFB_can_Id;
147  Flash_Write(flashAddrs.canId, dataFlash);
148 
149  dataFlash = 0;
150  dataFlash |= (this->baudrate & 0b111); // set the baudrate in 0..2 bit
151  dataFlash |= (this->useEncoder & 0x1) << 3; // set the encoder use in bit 3
152  Flash_Write(flashAddrs.data, dataFlash);
153 
154  saveFlashOffset();
155 }

213 CommandStatus VescCAN::command(const ParsedCommand &cmd,
214  std::vector<CommandReply> &replies) {
215 
216  switch (static_cast<VescCAN_commands>(cmd.cmdId)) {
217 
218  case VescCAN_commands::offbcanid:
219  return handleGetSet(cmd, replies, this->OFFB_can_Id);
220  case VescCAN_commands::vesccanid:
221  return handleGetSet(cmd, replies, this->VESC_can_Id);
222  case VescCAN_commands::errorflags:
223  if (cmd.type == CMDtype::get)
224  replies.push_back(CommandReply(vescErrorFlag));
225  break;
226  case VescCAN_commands::canspd:
227  if (cmd.type == CMDtype::get) {
228  replies.push_back(CommandReply(this->baudrate));
229  } else if (cmd.type == CMDtype::set) {
230  this->setCanRate(cmd.val);
231  }
232  break;
233  case VescCAN_commands::vescstate:
234  if (cmd.type == CMDtype::get)
235  replies.push_back(CommandReply((uint32_t) this->state));
236  break;
237  case VescCAN_commands::encrate:
238  if (cmd.type == CMDtype::get)
239  replies.push_back(CommandReply((uint32_t) this->encRate));
240  break;
241  case VescCAN_commands::pos:
242  if (cmd.type == CMDtype::get) {
243  replies.push_back(
244  CommandReply(
245  (int32_t) ((lastPos - posOffset) * 1000000000)));
246  }
247  break;
248  case VescCAN_commands::torque:
249  if (cmd.type == CMDtype::get) {
250  replies.push_back(CommandReply((int32_t) (lastTorque * 10000)));
251  }
252  break;
253  case VescCAN_commands::forcePosRead:
254  this->askPositionEncoder();
255  break;
256  case VescCAN_commands::useEncoder:
257  if (cmd.type == CMDtype::get) {
258  replies.push_back(CommandReply(useEncoder ? 1 : 0));
259  } else if (cmd.type == CMDtype::set) {
260  useEncoder = cmd.val != 0;
261  }
262  break;
263  case VescCAN_commands::offset: {
264  if (cmd.type == CMDtype::get) {
265  replies.push_back(CommandReply((int32_t) (posOffset * 10000)));
266  } else if (cmd.type == CMDtype::set) {
267  posOffset = (float) cmd.val / 10000.0;
268  this->saveFlashOffset(); // Really save to flash?
269  }
270  }
271  break;
272  case VescCAN_commands::voltage:
273  if (cmd.type == CMDtype::get)
274  replies.push_back(CommandReply(voltage * 1000));
275  break;
276 
277  default:
278  return CommandStatus::NOT_FOUND;
279  }
280 
281  return CommandStatus::OK;
282 }

284 /*
285  * state = VESC_UNKOWN => ping until the state go to PONG
286  * = VESC_PONG => call a status message
287  * = VESC_ERROR => call a status message
288  *
289  * if error => state = VESC_ERROR
290  * else=> state = VESC_RESPOND
291  *
292  * if RespondVescTime > 100 => VESC_NOT_RESPOND
293  *
294  */
295 void VescCAN::Run() {
296 
297  while (true) {
298  Delay(500);
299  // if status is UNKNOW, ask firmware info and wait next 500ms
300  if (state == VescState::VESC_STATE_UNKNOWN) {
301  this->getFirmwareInfo();
302  // the return of getFirmwareInfo change the state to COMPATIBLE/INCOMPATIBLE
303  // skip the follow control if we don't change from unknown state
304  continue;
305  }
306 
307  // if vesc is compatible, ready or in error => check the status
308  if (state >= VescState::VESC_STATE_COMPATIBLE) {
309  this->askGetValue();
310  // the return of askGetValue put the state in READY or ERROR is vesc respond to status
311  }
312 
313  // test is a ready/error vesc is always respond
314  uint32_t period = HAL_GetTick() - lastVescResponse;
315  if ((state == VescState::VESC_STATE_READY
316  || state == VescState::VESC_STATE_ERROR) && period > 1000) {
317  state = VescState::VESC_STATE_UNKNOWN;
318  }
319 
320  // when motor is active we call vesc each 500ms, to disable the vesc watchdog
321  // (if vesc not received message in 1sec, it cut off motor)
322  if (lastTorque != 0.0 && activeMotor
323  && state == VescState::VESC_STATE_READY) {
324  this->setTorque(lastTorque);
325  }
326 
327  // compute encoderRate each second
328  period = HAL_GetTick() - encStartPeriod;
329  if (period > 1000) {
330  encRate = encCount / (period / 1000.0);
331  encCount = 0;
332  encStartPeriod = HAL_GetTick();
333  }
334 
335  }
336 }

417 /*
418  * Compute the new position (laspos) value from data received by vesc:
419  * newPos is a 0-360° angle position
420  * if delta with previous and new position is > 180° then we count 1 turn
421  * at 1kz, more than 180° mean 30000rpm, not admissible in simracing
422  */
423 void VescCAN::decodeEncoderPosition(const float newPos) {
424  float delta = newPos - prevPos360; // Compute the delta position
425 
426  bool moreThanHalfTurn = fabs(delta) > 180.0 ? 1 : 0;
427  if (moreThanHalfTurn) {
428  // if delta is neg, turn is CCW, decrement multi turn pos... else increment it
429  mtPos += (delta > 0) ? -1.0 : 1.0;
430  }
431 
432  lastPos = (newPos / 360.0) + mtPos; // normalize the position
433  prevPos360 = newPos;
434 
435  encCount++;
436 }

451 void VescCAN::decode_buffer(uint8_t *buffer, uint8_t len) {
452 
453  if (!len) {
454  return;
455  }
456 
457  VescCmd command = (VescCmd) (buffer[0] & 0xFF);
458  buffer++;
459  len--;
460 
461  switch (command) {
462 
463  case VescCmd::COMM_FW_VERSION: {
464  int32_t ind = 0;
465  uint8_t fw_major = buffer[ind++];
466  uint8_t fw_min = buffer[ind++];
467 
468  std::string test((char*) buffer + ind);
469  ind += test.length() + 1;
470 
471  uint8_t uuid[12] = { 0 };
472  memcpy(buffer + ind, uuid, 12);
473  ind += 12;
474 
475  //bool isPaired = buffer[ind++];
476  uint8_t isTestFw = buffer[ind++];
477 
478  //uint8_t hwType = buffer[ind++]; // Comment because unused
479  //uint8_t customConfig = buffer[ind++]; // Comment because unused
480 
481  // bool hasPhaseFilters = buffer[ind++]; // Comment because unused
482 
483  bool compatible = false;
484  if (!isTestFw)
485  compatible = ((fw_major << 8) | fw_min) >= (FW_MIN_RELEASE >> 8);
486  else
487  compatible = ((fw_major << 16) | (fw_min << 8) | isTestFw)
488  >= FW_MIN_RELEASE;
489 
490  if (compatible) {
491  this->state = VescState::VESC_STATE_COMPATIBLE;
492  } else {
493  this->state = VescState::VESC_STATE_INCOMPATIBLE;
494  }
495 
496  break;
497  }
498 
499  case VescCmd::COMM_GET_VALUES_SELECTIVE: {
500  int32_t ind = 0;
501  uint32_t mask = this->buffer_get_uint32(buffer, &ind);
502 
503  if (mask & ((uint32_t) 1 << 8)) {
504  voltage = this->buffer_get_float16(buffer, 1e1, &ind);
505  }
506  if (mask & ((uint32_t) 1 << 15)) {
507  vescErrorFlag = buffer[ind++];
508 
509  // if the vesc respond but with an error, we put the vesc in error flag
510  if (vescErrorFlag) {
511  state = VescState::VESC_STATE_ERROR;
512  } else {
513  state = VescState::VESC_STATE_READY;
514  }
515 
516  }
517  if (mask & ((uint32_t) 1 << 16)) {
518  float pos = this->buffer_get_float32(buffer, 1e6, &ind);
519  this->decodeEncoderPosition(pos);
520  }
521  break;
522 
523  }
524 
525  default:
526  break;
527 
528  }

532 /*
533  * Process the received message on the CAN bus.
534  * Message have several struct, depends on message type, check the comm_can.c and
535  * command.c in the vesc source project to identify the struct format.
536  *
537  * Quick tips :
538  * Msg struct for CAN_PACKET_PROCESS_RX_BUFFER and CAN_PACKET_PROCESS_SHORT_BUFFER :
539  * uint8_t[0] => can_emitter_id (vesc id)
540  * uint8_t[1] => send bool
541  * uint8_t[2] => COMM_PACKET_ID command
542  * uint8_t[3] -> uint8_t[7] => DATA (5 bytes)
543  *
544  * Msg struct for CAN_PACKET_POLL_ROTOR_POS
545  * uint8_t[0]..uint8_t[3] : uint32 f_pos / 10000
546  *
547  */
548 void VescCAN::canRxPendCallback(CAN_HandleTypeDef *hcan, uint8_t *rxBuf,
549  CAN_RxHeaderTypeDef *rxHeader, uint32_t fifo) {
550 
551  // we record the last time respond to check if vesc is OK
552  lastVescResponse = HAL_GetTick();
553 
554  // Check if message is for the openFFBoard
555  uint16_t destCanID = rxHeader->ExtId & 0xFF;
556 
557  // Extract the command encoded in the ExtId
558  VescCANMsg cmd = (VescCANMsg) (rxHeader->ExtId >> 8);
559 
560  bool messageIsForThisVescAxis = destCanID == this->OFFB_can_Id;
561  messageIsForThisVescAxis |= (cmd == VescCANMsg::CAN_PACKET_POLL_ROTOR_POS) && // if it's a encoder position message
562  (this->VESC_can_Id == 0xFF); // and the vescCanId is not the default one
563  messageIsForThisVescAxis |= (cmd == VescCANMsg::CAN_PACKET_POLL_ROTOR_POS)
564  && // if it's a encoder position message
565  (this->VESC_can_Id != 0xFF) && // and the vescCanId is not the default one
566  (destCanID == this->VESC_can_Id); // we check that emiterId is the vescId for this axis
567 
568  if (!messageIsForThisVescAxis)
569  return;
570 
571  // Process the CAN message received
572  switch (cmd) {
573 
574  case VescCANMsg::CAN_PACKET_FILL_RX_BUFFER: {
575  memcpy(buffer_rx + rxBuf[0], rxBuf + 1, rxHeader->DLC - 1);
576  break;
577  }
578 
579  case VescCANMsg::CAN_PACKET_FILL_RX_BUFFER_LONG: {
580  uint32_t rxbuf_ind = (unsigned int) rxBuf[0] << 8;
581  rxbuf_ind |= rxBuf[1];
582  if (rxbuf_ind < BUFFER_RX_SIZE) {
583  memcpy(buffer_rx + rxbuf_ind, rxBuf + 2, rxHeader->DLC - 2);
584  }
585  break;
586  }
587 
588  case VescCANMsg::CAN_PACKET_PROCESS_RX_BUFFER: {
589 
590  // remove 2 first byte to exclude "can emitter id" and the "send bool" flag from buffer
591  int32_t index = 2;
592 
593  uint32_t rxbuf_length = (uint32_t) rxBuf[index++] << 8;
594  rxbuf_length |= (uint32_t) rxBuf[index++];
595 
596  if (rxbuf_length > BUFFER_RX_SIZE) {
597  break;
598  }
599 
600  uint8_t crc_high = rxBuf[index++];
601  uint8_t crc_low = rxBuf[index++];
602 
603  if (crc16(buffer_rx, rxbuf_length)
604  == ((uint8_t) crc_high << 8 | (uint8_t) crc_low)) {
605 
606  this->decode_buffer(buffer_rx, rxbuf_length);
607 
608  }
609 
610  break;
611  }
612  case VescCANMsg::CAN_PACKET_PROCESS_SHORT_BUFFER: {
613 
614  // remove 2 first byte to exclude "can emitter id" and the "send bool" flag from buffer
615  this->decode_buffer(rxBuf + 2, rxHeader->DLC - 2);
616 
617  break;
618  }
619 
620  case VescCANMsg::CAN_PACKET_PONG:
621  state = VescState::VESC_STATE_PONG;
622  break;
623 
624  case VescCANMsg::CAN_PACKET_POLL_ROTOR_POS: { // decode encoder data
625  int32_t index = 0;
626  float pos = this->buffer_get_int32(rxBuf, &index) / 100000.0; // extract the 0-360 float position
627  this->decodeEncoderPosition(pos);
628  break;
629  }
630 
631  default:
632  break;
633  }
634

642 void VescCAN::buffer_append_int32(uint8_t *buffer, int32_t number,
643  int32_t *index) {
644  buffer[(*index)++] = number >> 24;
645  buffer[(*index)++] = number >> 16;
