84 void VescCAN::restoreFlash() {
85  uint16_t canIds = 0x0D40;
86  if (Flash_Read(ADR_VESC_CANDATA, &canIds)) {
87  nodeId = canIds & 0xFF;
88 
89  uint8_t canspd = (canIds >> 8) & 0x7;
90  this->setCanRate(canspd);
91 
92  useEncoder = (canIds >> 11) & 0x1;
93  }
94 
95  uint16_t offset = 0;
96  if(Flash_Read(ADR_VESC_OFFSET, &offset)){
97  posOffset = (float) ((int16_t)offset / 10000.0);
98 
99  posOffset = posOffset - (int)posOffset; // Remove the multi-turn value
100  bool moreThanHalfTurn = fabs(posOffset) > 0.5 ? 1 : 0;
101  if (moreThanHalfTurn) {
102  // if delta is neg, turn is CCW, decrement multi turn pos... else increment it
103  posOffset += (posOffset > 0) ? -1.0 : 1.0;
104  }
105  }

109 void VescCAN::saveFlash() {
110  uint16_t canIds = 0x0D40;
111  Flash_Read(ADR_VESC_CANDATA, &canIds); // Read again
112  canIds &= ~0xFF; // reset bits for nodeId
113  canIds |= nodeId & 0xFF; // set the nodeId
114 
115  canIds &= ~0x700; // reset and set the baudrate
116  canIds |= (this->baudrate & 0x7) << 8;
117 
118  canIds &= ~0x800; // reset and set the useEncoder
119  canIds |= (useEncoder & 0x1) << 11;
120  Flash_Write(ADR_VESC_CANDATA, canIds);
121 
122  saveFlashOffset();
123 }

165 ParseStatus VescCAN::command(ParsedCommand *cmd, std::string *reply) {
166  // Prefix set but not our prefix
167  if (cmd->prefix != this->getInfo().unique && cmd->prefix != '\0') {
168  return ParseStatus::NOT_FOUND;
169  }
170  ParseStatus status = ParseStatus::OK;
171  if (cmd->cmd == "vescCanId") {
172  handleGetSet(cmd, reply, this->nodeId);
173  } else if (cmd->cmd == "vescErrorFlag") {
174  if (cmd->type == CMDtype::get) {
175  *reply += std::to_string(vescErrorFlag);
176  }
177  } else if (cmd->cmd == "vescState") {
178  if (cmd->type == CMDtype::get) {
179  *reply += std::to_string((uint32_t) state);
180  }
181  } else if (cmd->cmd == "vescCanSpd") {
182  if (cmd->type == CMDtype::get) {
183  *reply += std::to_string(baudrate);
184  } else if (cmd->type == CMDtype::set) {
185  this->setCanRate(cmd->val);
186  }
187  } else if (cmd->cmd == "vescEncRate") {
188  if (cmd->type == CMDtype::get) {
189  *reply += std::to_string((uint32_t) encRate);
190  }
191  } else if (cmd->cmd == "vescPos") {
192  if (cmd->type == CMDtype::get) {
193  *reply += std::to_string((int32_t) ((lastPos - posOffset) * 1000000000));
194  }
195  } else if (cmd->cmd == "vescTorque") {
196  if (cmd->type == CMDtype::get) {
197  *reply += std::to_string((int32_t) (lastTorque * 10000));
198  }
199  } else if (cmd->cmd == "vescPosReadForce") {
200  if(cmd->type == CMDtype::set){
201  this->askPositionEncoder();
202  }
203  } else if (cmd->cmd == "vescUseEncoder") {
204  if (cmd->type == CMDtype::get) {
205  *reply += std::to_string((int32_t) useEncoder);
206  } else if(cmd->type == CMDtype::set){
207  useEncoder = cmd->val;
208  }
209  } else if (cmd->cmd == "vescOffset") {
210  if (cmd->type == CMDtype::get) {
211  *reply += std::to_string((int32_t) (posOffset * 10000));
212  } else if(cmd->type == CMDtype::set){
213  posOffset = 0;
214  this->saveFlashOffset();
215  }
216  } else if (cmd->cmd == "vescVoltage") {
217  if (cmd->type == CMDtype::get) {
218  *reply += std::to_string((int32_t) (voltage * 100));
219  }
220  } else {
221  status = ParseStatus::NOT_FOUND;
222  }
223  return status;
224 }

226 /*
227  * state = VESC_UNKOWN => ping until the state go to PONG
228  * = VESC_PONG => call a status message
229  * = VESC_ERROR => call a status message
230  *
231  * if error => state = VESC_ERROR
232  * else=> state = VESC_RESPOND
233  *
234  * if RespondVescTime > 100 => VESC_NOT_RESPOND
235  *
236  */
237 void VescCAN::Run() {
238 
239  while (true) {
240  Delay(500);
241 
242  // if status is UNKNOW, ask firmware info and wait next 500ms
243  if (state == VescState::VESC_STATE_UNKNOWN) {
244  this->getFirmwareInfo();
245  // the return of getFirmwareInfo change the state to COMPATIBLE/INCOMPATIBLE
246  // skip the follow control if we don't change from unknown state
247  continue;
248  }
249 
250  // if vesc is compatible, ready or in error => check the status
251  if (state >= VescState::VESC_STATE_COMPATIBLE) {
252  this->askGetValue();
253  // the return of askGetValue put the state in READY or ERROR is vesc respond to status
254  }
255 
256  // test is a ready/error vesc is always respond
257  uint32_t period = HAL_GetTick() - lastVescResponse;
258  if ( (state == VescState::VESC_STATE_READY || state == VescState::VESC_STATE_ERROR)
259  && period > 1000) {
260  state = VescState::VESC_STATE_UNKNOWN;
261  }
262 
263 
264  // when motor is active we call vesc each 500ms, to disable the vesc watchdog
265  // (if vesc not received message in 1sec, it cut off motor)
266  if (lastTorque != 0.0 && activeMotor && state == VescState::VESC_STATE_READY) {
267  this->setTorque(lastTorque);
268  }
269 
270  // compute encoderRate each second
271  period = HAL_GetTick() - encStartPeriod;
272  if (period > 1000) {
273  encRate = encCount / (period / 1000.0);
274  encCount = 0;
275  encStartPeriod = HAL_GetTick();
276  }
277 
278  }
279 }

359 /*
360  * Compute the new position (laspos) value from data received by vesc:
361  * newPos is a 0-360° angle position
362  * if delta with previous and new position is > 180° then we count 1 turn
363  * at 1kz, more than 180° mean 30000rpm, not admissible in simracing
364  */
365 void VescCAN::decodeEncoderPosition(const float newPos) {
366  float delta = newPos - prevPos360; // Compute the delta position
367 
368  bool moreThanHalfTurn = fabs(delta) > 180.0 ? 1 : 0;
369  if (moreThanHalfTurn) {
370  // if delta is neg, turn is CCW, decrement multi turn pos... else increment it
371  mtPos += (delta > 0) ? -1.0 : 1.0;
372  }
373 
374  lastPos = (newPos / 360.0) + mtPos; // normalize the position
375  prevPos360 = newPos;
376 
377  encCount++;
378 }

393 void VescCAN::decode_buffer(uint8_t *buffer, uint8_t len) {
394 
395  if (!len) {
396  return;
397  }
398 
399  VescCmd command = (VescCmd)(buffer[0] & 0xFF);
400  buffer++;
401  len--;
402 
403  switch (command) {
404 
405  case VescCmd::COMM_FW_VERSION : {
406  int32_t ind = 0;
407  uint8_t fw_major = buffer[ind++];
408  uint8_t fw_min = buffer[ind++];
409 
410  std::string test ((char *)buffer + ind);
411  ind += test.length() + 1;
412 
413  uint8_t uuid[12];
414  memcpy(buffer + ind, uuid, 12);
415  ind += 12;
416 
417  bool isPaired = buffer[ind++];
418  uint8_t isTestFw = buffer[ind++];
419 
420  //uint8_t hwType = buffer[ind++]; // Comment because unused
421  //uint8_t customConfig = buffer[ind++]; // Comment because unused
422 
423  // bool hasPhaseFilters = buffer[ind++]; // Comment because unused
424 
425 
426  bool compatible = false;
427  if (!isTestFw)
428  compatible = ((fw_major << 8) | fw_min) >= (FW_MIN_RELEASE >> 8);
429  else
430  compatible = ((fw_major << 16) | (fw_min << 8) | isTestFw) >= FW_MIN_RELEASE;
431 
432  if (compatible) {
433  this->state = VescState::VESC_STATE_COMPATIBLE;
434  } else {
435  this->state = VescState::VESC_STATE_INCOMPATIBLE;
436  }
437 
438  break;
439  }
440 
441  case VescCmd::COMM_GET_VALUES_SELECTIVE : {
442  int32_t ind = 0;
443  uint32_t mask = this->buffer_get_uint32(buffer, &ind);
444 
445  if (mask & ((uint32_t)1 << 8)) {
446  voltage = this->buffer_get_float16(buffer, 1e1, &ind);
447  }
448  if (mask & ((uint32_t)1 << 15)) {
449  vescErrorFlag = buffer[ind++];
450 
451  // if the vesc respond but with an error, we put the vesc in error flag
452  if (vescErrorFlag) {
453  state = VescState::VESC_STATE_ERROR;
454  } else {
455  state = VescState::VESC_STATE_READY;
456  }
457 
458  }
459  if (mask & ((uint32_t)1 << 16)) {
460  float pos = this->buffer_get_float32(buffer, 1e6, &ind);
461  this->decodeEncoderPosition(pos);
462  }
463  break;
464 
465  }
466 
467  default:
468  break;
469 
470  }
471

474 /*
475  * Process the received message on the CAN bus.
476  * Message have several struct, depends on message type, check the comm_can.c and
477  * command.c in the vesc source project to identify the struct format.
478  *
479  * Quick tips :
480  * Msg struct for CAN_PACKET_PROCESS_RX_BUFFER and CAN_PACKET_PROCESS_SHORT_BUFFER :
481  * uint8_t[0] => can_emitter_id (vesc id)
482  * uint8_t[1] => send bool
483  * uint8_t[2] => COMM_PACKET_ID command
484  * uint8_t[3] -> uint8_t[7] => DATA (5 bytes)
485  *
486  * Msg struct for CAN_PACKET_POLL_ROTOR_POS
487  * uint8_t[0]..uint8_t[3] : uint32 f_pos / 10000
488  *
489  */
490 void VescCAN::canRxPendCallback(CAN_HandleTypeDef *hcan, uint8_t *rxBuf,
491  CAN_RxHeaderTypeDef *rxHeader, uint32_t fifo) {
492 
493  // we record the last time respond to check if vesc is OK
494  lastVescResponse = HAL_GetTick();
495 
496  // Check if message is for the openFFBoard
497  uint16_t node = rxHeader->ExtId & 0xFF;
498 
499  // Extract the command encoded in the ExtId
500  VescCANMsg cmd = (VescCANMsg) (rxHeader->ExtId >> 8);
501 
502  // if msg is not for this node && mesg not a poll result, bypass
503  if ((node != this->nodeId) && (cmd != VescCANMsg::CAN_PACKET_POLL_ROTOR_POS )) {
504  return;
505  }
506 
507  // Process the CAN message received
508  switch (cmd) {
509 
510  case VescCANMsg::CAN_PACKET_FILL_RX_BUFFER: {
511  memcpy(buffer_rx + rxBuf[0], rxBuf + 1, rxHeader->DLC - 1);
512  break;
513  }
514 
515  case VescCANMsg::CAN_PACKET_FILL_RX_BUFFER_LONG: {
516  uint32_t rxbuf_ind = (unsigned int)rxBuf[0] << 8;
517  rxbuf_ind |= rxBuf[1];
518  if (rxbuf_ind < BUFFER_RX_SIZE) {
519  memcpy(buffer_rx + rxbuf_ind, rxBuf + 2, rxHeader->DLC - 2);
520  }
521  break;
522  }
523 
524  case VescCANMsg::CAN_PACKET_PROCESS_RX_BUFFER: {
525 
526  // remove 2 first byte to exclude "can emitter id" and the "send bool" flag from buffer
527  int32_t index = 2;
528 
529  uint32_t rxbuf_length = (uint32_t)rxBuf[index++] << 8;
530  rxbuf_length |= (uint32_t)rxBuf[index++];
531 
532  if (rxbuf_length > BUFFER_RX_SIZE) {
533  break;
534  }
535 
536  uint8_t crc_high = rxBuf[index++];
537  uint8_t crc_low = rxBuf[index++];
538 
539  if (crc16(buffer_rx, rxbuf_length) == ((uint8_t) crc_high << 8 | (uint8_t) crc_low)) {
540 
541  this->decode_buffer(buffer_rx, rxbuf_length);
542 
543  }
544 
545  break;
546  }
547  case VescCANMsg::CAN_PACKET_PROCESS_SHORT_BUFFER: {
548 
549  // remove 2 first byte to exclude "can emitter id" and the "send bool" flag from buffer
550  this->decode_buffer(rxBuf + 2, rxHeader->DLC - 2);
551 
552  break;
553  }
554 
555  case VescCANMsg::CAN_PACKET_PONG:
556  state = VescState::VESC_STATE_PONG;
557  break;
558 
559  case VescCANMsg::CAN_PACKET_POLL_ROTOR_POS : { // decode encoder data
560  int32_t index = 0;
561  float pos = this->buffer_get_int32(rxBuf, &index) / 100000.0; // extract the 0-360 float position
562  this->decodeEncoderPosition(pos);
563  break;
564  }
565 
566  default:
567  break;
568  }
569 
570 }
