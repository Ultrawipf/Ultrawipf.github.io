64 void Biquad::calcBiquad(void) {
65  z1 = 0.0;
66  z2 = 0.0;
67  float norm;
68  float V = pow(10, fabs(peakGain) / 20.0);
69  float K = tan(M_PI * Fc);
70  switch (this->type) {
71  case BiquadType::lowpass:
72  norm = 1 / (1 + K / Q + K * K);
73  a0 = K * K * norm;
74  a1 = 2 * a0;
75  a2 = a0;
76  b1 = 2 * (K * K - 1) * norm;
77  b2 = (1 - K / Q + K * K) * norm;
78  break;
79 
80  case BiquadType::highpass:
81  norm = 1 / (1 + K / Q + K * K);
82  a0 = 1 * norm;
83  a1 = -2 * a0;
84  a2 = a0;
85  b1 = 2 * (K * K - 1) * norm;
86  b2 = (1 - K / Q + K * K) * norm;
87  break;
88 
89  case BiquadType::bandpass:
90  norm = 1 / (1 + K / Q + K * K);
91  a0 = K / Q * norm;
92  a1 = 0;
93  a2 = -a0;
94  b1 = 2 * (K * K - 1) * norm;
95  b2 = (1 - K / Q + K * K) * norm;
96  break;
97 
98  case BiquadType::notch:
99  norm = 1 / (1 + K / Q + K * K);
100  a0 = (1 + K * K) * norm;
101  a1 = 2 * (K * K - 1) * norm;
102  a2 = a0;
103  b1 = a1;
104  b2 = (1 - K / Q + K * K) * norm;
105  break;
106 
107  case BiquadType::peak:
108  if (peakGain >= 0) { // boost
109  norm = 1 / (1 + 1/Q * K + K * K);
110  a0 = (1 + V/Q * K + K * K) * norm;
111  a1 = 2 * (K * K - 1) * norm;
112  a2 = (1 - V/Q * K + K * K) * norm;
113  b1 = a1;
114  b2 = (1 - 1/Q * K + K * K) * norm;
115  }
116  else { // cut
117  norm = 1 / (1 + V/Q * K + K * K);
118  a0 = (1 + 1/Q * K + K * K) * norm;
119  a1 = 2 * (K * K - 1) * norm;
120  a2 = (1 - 1/Q * K + K * K) * norm;
121  b1 = a1;
122  b2 = (1 - V/Q * K + K * K) * norm;
123  }
124  break;
125  case BiquadType::lowshelf:
126  if (peakGain >= 0) { // boost
127  norm = 1 / (1 + sqrt(2) * K + K * K);
128  a0 = (1 + sqrt(2*V) * K + V * K * K) * norm;
129  a1 = 2 * (V * K * K - 1) * norm;
130  a2 = (1 - sqrt(2*V) * K + V * K * K) * norm;
131  b1 = 2 * (K * K - 1) * norm;
132  b2 = (1 - sqrt(2) * K + K * K) * norm;
133  }
134  else { // cut
135  norm = 1 / (1 + sqrt(2*V) * K + V * K * K);
136  a0 = (1 + sqrt(2) * K + K * K) * norm;
137  a1 = 2 * (K * K - 1) * norm;
138  a2 = (1 - sqrt(2) * K + K * K) * norm;
139  b1 = 2 * (V * K * K - 1) * norm;
140  b2 = (1 - sqrt(2*V) * K + V * K * K) * norm;
141  }
142  break;
143  case BiquadType::highshelf:
144  if (peakGain >= 0) { // boost
145  norm = 1 / (1 + sqrt(2) * K + K * K);
146  a0 = (V + sqrt(2*V) * K + K * K) * norm;
147  a1 = 2 * (K * K - V) * norm;
148  a2 = (V - sqrt(2*V) * K + K * K) * norm;
149  b1 = 2 * (K * K - 1) * norm;
150  b2 = (1 - sqrt(2) * K + K * K) * norm;
151  }
152  else { // cut
153  norm = 1 / (V + sqrt(2*V) * K + K * K);
154  a0 = (1 + sqrt(2) * K + K * K) * norm;
155  a1 = 2 * (K * K - 1) * norm;
156  a2 = (1 - sqrt(2) * K + K * K) * norm;
157  b1 = 2 * (K * K - V) * norm;
158  b2 = (V - sqrt(2*V) * K + K * K) * norm;
159  }
160  break;
161  }
