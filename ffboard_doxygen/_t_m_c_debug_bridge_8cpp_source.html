95 void TMCDebugBridge::cdcRcv(char* Buf, uint32_t *Len){
96  std::vector<uint8_t> dat;
97  dat.assign(Buf,Buf+*Len);
98  if(*Len == 9 && *Buf == 1){
99  uint8_t crc = checksum(&dat,*Len-1);
100 
101  //uint8_t id = dat[0];
102  uint8_t cmd = dat[1];
103  uint8_t addr = dat[2];
104 
105  if(dat[dat.size()-1] != crc){
106  //fail
107  return;
108  }
109 
110  if(cmd == 148){ // read reg
111  uint8_t buf[5];
112  uint8_t rpl[4] = {2,1,0x64,0x94};
113  tmcReadRegRaw(addr, buf);
114  std::vector<uint8_t> rx_data;
115 
116  rx_data.insert(rx_data.begin(), &rpl[0],&rpl[4]);
117  rx_data.insert(rx_data.end(),buf+1, buf+5);
118  rx_data.push_back(checksum(&rx_data,8));
119  sendCdc((char*)rx_data.data(), 9);
120 
121  }else if(cmd == 146){ // write
122 
123  uint32_t ndat;
124  memcpy(&ndat,Buf+4,4);
125  drv->writeReg(addr,__REV(ndat));
126  std::vector<uint8_t> repl({2,1,0x64,0x92,dat[3],dat[4],dat[5],dat[6]});
127  repl.push_back(checksum(&repl,8));
128  sendCdc((char*)repl.data(), 9);
129  }else if(cmd == 143){
130  std::vector<uint8_t> repl(8,0);
131  if(addr == 3){
132  repl.assign({2,1,40,0x8f,0,0,0,2});
133  }else if(addr == 4){
134  repl.assign({2,1,40,0x8f,2,6,2,2});
135  }
136  repl.push_back(checksum(&repl,8));
137  sendCdc((char*)repl.data(), 9);
138  }else if(cmd == 10){ // Get global parameter.
139  std::vector<uint8_t> repl({2,1,64,0x0A,0,0,0,0});
140  if(addr == 5){
141  repl[7] = 2; // hwid
142  }else if(addr == 2){
143  repl[7] = 1; // active
144  }
145  repl.push_back(checksum(&repl,8));
146  sendCdc((char*)repl.data(), 9);
147  }else if(cmd == 0x0F){ // Get input
148  std::vector<uint8_t> repl({2,1,64,0x0F,0,0,0,0});
149  if(addr == 5){ // Voltage
150  uint16_t v = getIntV()/100;
151  repl[7] = v & 0xff;
152  repl[6] = (v>>8) & 0xff;
153  }
154  repl.push_back(checksum(&repl,8));
155  sendCdc((char*)repl.data(), 9);
156  }else if(cmd == 0x88){
157  char version[9] = {2,'0','0','1','5','V','3','0','7'}; // Version string
158  if(addr == 0){
159  sendCdc(version, 9);
160  }else if(addr == 1){ // Version binary
161  // module version high
162  uint32_t tmpVal = (uint8_t) version[1] - '0';
163  std::vector<uint8_t> repl = {2,0x01,0x64,0x88,0x00,0,0,0};
164  tmpVal *= 10;
165  tmpVal += (uint8_t) version[2] - '0';
166  repl[4] = tmpVal;
167 
168  // module version low
169  tmpVal = (uint8_t) version[3] - '0';
170  tmpVal *= 10;
171  tmpVal += (uint8_t) version[4] - '0';
172  repl[5] = tmpVal;
173 
174  // fw version high
175  repl[6] = (uint8_t) version[6] - '0';
176 
177  // fw version low
178  tmpVal = (uint8_t) version[7] - '0';
179  tmpVal *= 10;
180  tmpVal += (uint8_t) version[8] - '0';
181  repl[7] = tmpVal;
182  repl.push_back(checksum(&repl,8));
183  sendCdc((char*)repl.data(), 9);
184  }
185 
186  }
187  }else{
