103 uint16_t btd_open(uint8_t rhport, tusb_desc_interface_t const *itf_desc, uint16_t max_len)
104 {
105  tusb_desc_endpoint_t const *desc_ep;
106  uint16_t drv_len = 0;
107  // Size of single alternative of ISO interface
108  const uint16_t iso_alt_itf_size = sizeof(tusb_desc_interface_t) + 2 * sizeof(tusb_desc_endpoint_t);
109  // Size of hci interface
110  const uint16_t hci_itf_size = sizeof(tusb_desc_interface_t) + 3 * sizeof(tusb_desc_endpoint_t);
111  // Ensure this is BT Primary Controller
112  TU_VERIFY(TUSB_CLASS_WIRELESS_CONTROLLER == itf_desc->bInterfaceClass &&
113  TUD_BT_APP_SUBCLASS == itf_desc->bInterfaceSubClass &&
114  TUD_BT_PROTOCOL_PRIMARY_CONTROLLER == itf_desc->bInterfaceProtocol, 0);
115 
116  // Distinguish interface by number of endpoints, as both interface have same class, subclass and protocol
117  if (itf_desc->bNumEndpoints == 3 && max_len >= hci_itf_size)
118  {
119  _btd_itf.itf_num = itf_desc->bInterfaceNumber;
120 
121  desc_ep = (tusb_desc_endpoint_t const *) tu_desc_next(itf_desc);
122 
123  TU_ASSERT(TUSB_DESC_ENDPOINT == desc_ep->bDescriptorType && TUSB_XFER_INTERRUPT == desc_ep->bmAttributes.xfer, 0);
124  TU_ASSERT(usbd_edpt_open(rhport, desc_ep), 0);
125  _btd_itf.ep_ev = desc_ep->bEndpointAddress;
126 
127  // Open endpoint pair
128  TU_ASSERT(usbd_open_edpt_pair(rhport, tu_desc_next(desc_ep), 2, TUSB_XFER_BULK, &_btd_itf.ep_acl_out,
129  &_btd_itf.ep_acl_in), 0);
130 
131  // Prepare for incoming data from host
132  TU_ASSERT(usbd_edpt_xfer(rhport, _btd_itf.ep_acl_out, _btd_itf.epout_buf, CFG_TUD_BTH_DATA_EPSIZE), 0);
133 
134  drv_len = hci_itf_size;
135  }
136  else if (itf_desc->bNumEndpoints == 2 && max_len >= iso_alt_itf_size)
137  {
138  uint8_t dir;
139 
140  desc_ep = (tusb_desc_endpoint_t const *)tu_desc_next(itf_desc);
141  TU_ASSERT(itf_desc->bAlternateSetting < CFG_TUD_BTH_ISO_ALT_COUNT, 0);
142  TU_ASSERT(desc_ep->bDescriptorType == TUSB_DESC_ENDPOINT, 0);
143  dir = tu_edpt_dir(desc_ep->bEndpointAddress);
144  _btd_itf.ep_voice[dir] = desc_ep->bEndpointAddress;
145  // Store endpoint size for alternative
146  _btd_itf.ep_voice_size[dir][itf_desc->bAlternateSetting] = (uint8_t)desc_ep->wMaxPacketSize.size;
147 
148  desc_ep = (tusb_desc_endpoint_t const *)tu_desc_next(desc_ep);
149  TU_ASSERT(desc_ep->bDescriptorType == TUSB_DESC_ENDPOINT, 0);
150  dir = tu_edpt_dir(desc_ep->bEndpointAddress);
151  _btd_itf.ep_voice[dir] = desc_ep->bEndpointAddress;
152  // Store endpoint size for alternative
153  _btd_itf.ep_voice_size[dir][itf_desc->bAlternateSetting] = (uint8_t)desc_ep->wMaxPacketSize.size;
154  drv_len += iso_alt_itf_size;
155 
156  for (int i = 1; i < CFG_TUD_BTH_ISO_ALT_COUNT && drv_len + iso_alt_itf_size <= max_len; ++i) {
157  // Make sure rest of alternatives matches
158  itf_desc = (tusb_desc_interface_t const *)tu_desc_next(desc_ep);
159  if (itf_desc->bDescriptorType != TUSB_DESC_INTERFACE ||
160  TUSB_CLASS_WIRELESS_CONTROLLER != itf_desc->bInterfaceClass ||
161  TUD_BT_APP_SUBCLASS != itf_desc->bInterfaceSubClass ||
162  TUD_BT_PROTOCOL_PRIMARY_CONTROLLER != itf_desc->bInterfaceProtocol)
163  {
164  // Not an Iso interface instance
165  break;
166  }
167  TU_ASSERT(itf_desc->bAlternateSetting < CFG_TUD_BTH_ISO_ALT_COUNT, 0);
168 
169  desc_ep = (tusb_desc_endpoint_t const *)tu_desc_next(itf_desc);
170  dir = tu_edpt_dir(desc_ep->bEndpointAddress);
171  // Verify that alternative endpoint are same as first ones
172  TU_ASSERT(desc_ep->bDescriptorType == TUSB_DESC_ENDPOINT &&
173  _btd_itf.ep_voice[dir] == desc_ep->bEndpointAddress, 0);
174  _btd_itf.ep_voice_size[dir][itf_desc->bAlternateSetting] = (uint8_t)desc_ep->wMaxPacketSize.size;
175 
176  desc_ep = (tusb_desc_endpoint_t const *)tu_desc_next(desc_ep);
177  dir = tu_edpt_dir(desc_ep->bEndpointAddress);
178  // Verify that alternative endpoint are same as first ones
179  TU_ASSERT(desc_ep->bDescriptorType == TUSB_DESC_ENDPOINT &&
180  _btd_itf.ep_voice[dir] == desc_ep->bEndpointAddress, 0);
181  _btd_itf.ep_voice_size[dir][itf_desc->bAlternateSetting] = (uint8_t)desc_ep->wMaxPacketSize.size;
182  drv_len += iso_alt_itf_size;
183  }
184  }
185 
186  return drv_len;
187 }

189 // Invoked when a control transfer occurred on an interface of this class
190 // Driver response accordingly to the request and the transfer stage (setup/data/ack)
191 // return false to stall control endpoint (e.g unsupported request)
192 bool btd_control_xfer_cb(uint8_t rhport, uint8_t stage, tusb_control_request_t const *request)
193 {
194  (void)rhport;
195 
196  if ( stage == CONTROL_STAGE_SETUP )
197  {
198  if (request->bmRequestType_bit.type == TUSB_REQ_TYPE_CLASS &&
199  request->bmRequestType_bit.recipient == TUSB_REQ_RCPT_DEVICE)
200  {
201  // HCI command packet addressing for single function Primary Controllers
202  TU_VERIFY(request->bRequest == 0 && request->wValue == 0 && request->wIndex == 0);
203  }
204  else if (request->bmRequestType_bit.recipient == TUSB_REQ_RCPT_INTERFACE)
205  {
206  if (request->bRequest == TUSB_REQ_SET_INTERFACE && _btd_itf.itf_num + 1 == request->wIndex)
207  {
208  // TODO: Set interface it would involve changing size of endpoint size
209  }
210  else
211  {
212  // HCI command packet for Primary Controller function in a composite device
213  TU_VERIFY(request->bRequest == 0 && request->wValue == 0 && request->wIndex == _btd_itf.itf_num);
214  }
215  }
216  else return false;
217 
218  return tud_control_xfer(rhport, request, &_btd_itf.hci_cmd, request->wLength);
219  }
220  else if ( stage == CONTROL_STAGE_DATA )
221  {
222  // Handle class request only
223  TU_VERIFY(request->bmRequestType_bit.type == TUSB_REQ_TYPE_CLASS);
224 
225  if (tud_bt_hci_cmd_cb) tud_bt_hci_cmd_cb(&_btd_itf.hci_cmd, request->wLength);
226  }
227 
228  return true;
229 }

231 bool btd_xfer_cb(uint8_t rhport, uint8_t ep_addr, xfer_result_t result, uint32_t xferred_bytes)
232 {
233  (void)result;
234 
235  // received new data from host
236  if (ep_addr == _btd_itf.ep_acl_out)
237  {
238  if (tud_bt_acl_data_received_cb) tud_bt_acl_data_received_cb(_btd_itf.epout_buf, xferred_bytes);
239 
240  // prepare for next data
241  TU_ASSERT(usbd_edpt_xfer(rhport, _btd_itf.ep_acl_out, _btd_itf.epout_buf, CFG_TUD_BTH_DATA_EPSIZE));
242  }
243  else if (ep_addr == _btd_itf.ep_ev)
244  {
245  if (tud_bt_event_sent_cb) tud_bt_event_sent_cb((uint16_t)xferred_bytes);
246  }
247  else if (ep_addr == _btd_itf.ep_acl_in)
248  {
249  if (tud_bt_acl_data_sent_cb) tud_bt_acl_data_sent_cb((uint16_t)xferred_bytes);
250  }
251 
252  return true;
253 }
