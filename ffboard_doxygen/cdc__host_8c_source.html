65 bool tuh_cdc_is_busy(uint8_t dev_addr, cdc_pipeid_t pipeid)
66 {
67  if ( !tuh_cdc_mounted(dev_addr) ) return false;
68 
69  cdch_data_t const * p_cdc = get_itf(dev_addr);
70 
71  switch (pipeid)
72  {
73  case CDC_PIPE_NOTIFICATION:
74  return hcd_edpt_busy(dev_addr, p_cdc->ep_notif );
75 
76  case CDC_PIPE_DATA_IN:
77  return hcd_edpt_busy(dev_addr, p_cdc->ep_in );
78 
79  case CDC_PIPE_DATA_OUT:
80  return hcd_edpt_busy(dev_addr, p_cdc->ep_out );
81 
82  default:
83  return false;
84  }
85 }

97 bool tuh_cdc_send(uint8_t dev_addr, void const * p_data, uint32_t length, bool is_notify)
98 {
99  TU_VERIFY( tuh_cdc_mounted(dev_addr) );
100  TU_VERIFY( p_data != NULL && length, TUSB_ERROR_INVALID_PARA);
101 
102  uint8_t const ep_out = cdch_data[dev_addr-1].ep_out;
103  if ( hcd_edpt_busy(dev_addr, ep_out) ) return false;
104 
105  return hcd_pipe_xfer(dev_addr, ep_out, (void *) p_data, length, is_notify);
106 }

119 bool tuh_cdc_set_control_line_state(uint8_t dev_addr, bool dtr, bool rts, tuh_control_complete_cb_t complete_cb)
120 {
121  cdch_data_t const * p_cdc = get_itf(dev_addr);

123  {
124  .bmRequestType_bit =
125  {
126  .recipient = TUSB_REQ_RCPT_INTERFACE,
127  .type = TUSB_REQ_TYPE_CLASS,
128  .direction = TUSB_DIR_OUT
129  },
130  .bRequest = CDC_REQUEST_SET_CONTROL_LINE_STATE,
131  .wValue = (rts ? 2 : 0) | (dtr ? 1 : 0),
132  .wIndex = p_cdc->itf_num,
133  .wLength = 0

135

148 bool cdch_open(uint8_t rhport, uint8_t dev_addr, tusb_desc_interface_t const *itf_desc, uint16_t *p_length)
149 {
150  // Only support ACM
151  TU_VERIFY( CDC_COMM_SUBCLASS_ABSTRACT_CONTROL_MODEL == itf_desc->bInterfaceSubClass);
152 
153  // Only support AT commands, no protocol and vendor specific commands.
154  TU_VERIFY(tu_within(CDC_COMM_PROTOCOL_NONE, itf_desc->bInterfaceProtocol, CDC_COMM_PROTOCOL_ATCOMMAND_CDMA) ||
155  0xff == itf_desc->bInterfaceProtocol);
156 
157  uint8_t const * p_desc;
158  cdch_data_t * p_cdc;
159 
160  p_desc = tu_desc_next(itf_desc);
161  p_cdc = get_itf(dev_addr);
162 
163  p_cdc->itf_num = itf_desc->bInterfaceNumber;
164  p_cdc->itf_protocol = itf_desc->bInterfaceProtocol; // TODO 0xff is consider as rndis candidate, other is virtual Com
165 
166  //------------- Communication Interface -------------//
167  (*p_length) = sizeof(tusb_desc_interface_t);
168 
169  // Communication Functional Descriptors
170  while( TUSB_DESC_CS_INTERFACE == p_desc[DESC_OFFSET_TYPE] )
171  {
172  if ( CDC_FUNC_DESC_ABSTRACT_CONTROL_MANAGEMENT == cdc_functional_desc_typeof(p_desc) )
173  {
174  // save ACM bmCapabilities
175  p_cdc->acm_capability = ((cdc_desc_func_acm_t const *) p_desc)->bmCapabilities;
176  }
177 
178  (*p_length) += p_desc[DESC_OFFSET_LEN];
179  p_desc = tu_desc_next(p_desc);
180  }
181 
182  if ( TUSB_DESC_ENDPOINT == p_desc[DESC_OFFSET_TYPE])
183  {
184  // notification endpoint
185  tusb_desc_endpoint_t const * ep_desc = (tusb_desc_endpoint_t const *) p_desc;
186 
187  TU_ASSERT( usbh_edpt_open(rhport, dev_addr, ep_desc) );
188  p_cdc->ep_notif = ep_desc->bEndpointAddress;
189 
190  (*p_length) += p_desc[DESC_OFFSET_LEN];
191  p_desc = tu_desc_next(p_desc);
192  }
193 
194  //------------- Data Interface (if any) -------------//
195  if ( (TUSB_DESC_INTERFACE == p_desc[DESC_OFFSET_TYPE]) &&
196  (TUSB_CLASS_CDC_DATA == ((tusb_desc_interface_t const *) p_desc)->bInterfaceClass) )
197  {
198  (*p_length) += p_desc[DESC_OFFSET_LEN];
199  p_desc = tu_desc_next(p_desc);
200 
201  // data endpoints expected to be in pairs
202  for(uint32_t i=0; i<2; i++)
203  {
204  tusb_desc_endpoint_t const *ep_desc = (tusb_desc_endpoint_t const *) p_desc;
205  TU_ASSERT(TUSB_DESC_ENDPOINT == ep_desc->bDescriptorType);
206  TU_ASSERT(TUSB_XFER_BULK == ep_desc->bmAttributes.xfer);
207 
208  TU_ASSERT(usbh_edpt_open(rhport, dev_addr, ep_desc));
209 
210  if ( tu_edpt_dir(ep_desc->bEndpointAddress) == TUSB_DIR_IN )
211  {
212  p_cdc->ep_in = ep_desc->bEndpointAddress;
213  }else
214  {
215  p_cdc->ep_out = ep_desc->bEndpointAddress;
216  }
217 
218  (*p_length) += p_desc[DESC_OFFSET_LEN];
219  p_desc = tu_desc_next( p_desc );
220  }
221  }
222 
223  return true;
224 }
